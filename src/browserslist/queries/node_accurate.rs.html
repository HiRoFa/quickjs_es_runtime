4pub(super) fn node_accurate(version: &str, opts: &Opts) -> QueryResult {
5    let distribs = NODE_VERSIONS
6        .iter()
7        .rev()
8        .find(|v| v.split('.').zip(version.split('.')).all(|(a, b)| a == b))
9        .map(|version| vec![Distrib::new("node", *version)]);
10    if opts.ignore_unknown_versions {
11        Ok(distribs.unwrap_or_default())
12    } else {
13        distribs.ok_or_else(|| Error::UnknownNodejsVersion(version.to_string()))
14    }
15}
