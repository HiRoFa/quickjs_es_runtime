8pub(super) fn last_n_major_browsers(count: usize, opts: &Opts) -> QueryResult {
9    let distribs = CANIUSE_BROWSERS
10        .keys()
11        .filter_map(|name| get_browser_stat(name, opts.mobile_to_desktop))
12        .flat_map(|(name, stat)| {
13            let count = count_filter_versions(name, opts.mobile_to_desktop, count);
14
15            let minimum: u32 = stat
16                .version_list
17                .iter()
18                .filter(|version| version.release_date.is_some())
19                .rev()
20                .map(|version| version.version.split('.').next().unwrap())
21                .dedup()
22                .nth(count - 1)
23                .and_then(|minimum| minimum.parse().ok())
24                .unwrap_or(0);
25
26            stat.version_list
27                .iter()
28                .filter(|version| version.release_date.is_some())
29                .map(|version| version.version)
30                .filter(move |version| {
31                    version.split('.').next().unwrap().parse().unwrap_or(0) >= minimum
32                })
33                .rev()
34                .map(move |version| Distrib::new(name, version))
35        })
36        .collect();
37
38    Ok(distribs)
39}
