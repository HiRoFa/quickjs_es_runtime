5pub(super) fn percentage_by_region(
6    comparator: Comparator,
7    popularity: f32,
8    region: &str,
9) -> QueryResult {
10    let normalized_region = if region.len() == 2 {
11        region.to_uppercase()
12    } else {
13        region.to_lowercase()
14    };
15
16    if let Some(region_data) = get_usage_by_region(&normalized_region) {
17        let distribs = region_data
18            .iter()
19            .filter(|(_, _, usage)| match comparator {
20                Comparator::Greater => *usage > popularity,
21                Comparator::Less => *usage < popularity,
22                Comparator::GreaterOrEqual => *usage >= popularity,
23                Comparator::LessOrEqual => *usage <= popularity,
24            })
25            .map(|(name, version, _)| Distrib::new(name, version))
26            .collect();
27        Ok(distribs)
28    } else {
29        Err(Error::UnknownRegion(region.to_string()))
30    }
31}
