10#[cfg(target_arch = "wasm32")]
11pub(super) fn extends(pkg: &str, opts: &Opts) -> QueryResult {
12    if opts.dangerous_extend {
13        Err(Error::UnsupportedExtends)
14    } else {
15        check_extend_name(pkg).map(|_| Default::default())
16    }
17}

19#[cfg(not(target_arch = "wasm32"))]
20pub(super) fn extends(pkg: &str, opts: &Opts) -> QueryResult {
21    use crate::{config, resolve};
22    use std::{env, process};
23
24    let dangerous_extend =
25        opts.dangerous_extend || env::var("BROWSERSLIST_DANGEROUS_EXTEND").is_ok();
26    if !dangerous_extend {
27        check_extend_name(pkg)?;
28    }
29
30    let mut command = process::Command::new("node");
31    command.args(["-p", &format!("JSON.stringify(require('{pkg}'))")]);
32    #[cfg(test)]
33    command.current_dir(&*BASE_TEST_DIR);
34    let output = command
35        .output()
36        .map_err(|_| Error::UnsupportedExtends)?
37        .stdout;
38    let config = serde_json::from_str(&String::from_utf8_lossy(&output))
39        .map_err(|_| Error::FailedToResolveExtend(pkg.to_string()))?;
40
41    resolve(config::load_with_config(config, opts)?, opts)
42}

44fn check_extend_name(pkg: &str) -> Result<(), Error> {
45    let unscoped = pkg
46        .strip_prefix('@')
47        .and_then(|s| s.find('/').and_then(|i| s.get(i + 1..)))
48        .unwrap_or(pkg);
49    if !(unscoped.starts_with("browserslist-config-")
50        || (pkg.starts_with('@') && unscoped == "browserslist-config"))
51    {
52        return Err(Error::InvalidExtendName(
53            "Browserslist config needs `browserslist-config-` prefix.",
54        ));
55    }
56    if unscoped.contains('.') {
57        return Err(Error::InvalidExtendName(
58            "`.` not allowed in Browserslist config name.",
59        ));
60    }
61    if pkg.contains("node_modules") {
62        return Err(Error::InvalidExtendName(
63            "`node_modules` not allowed in Browserslist config.",
64        ));
65    }
66
67    Ok(())
68}
