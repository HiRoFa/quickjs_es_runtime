1use serde::{Deserialize, Deserializer, Serialize};
2use swc_atoms::{atom, Atom};
3#[derive(Debug, Clone, Serialize, Deserialize, Eq, PartialEq)]
4#[serde(rename_all = "camelCase")]
5#[serde(deny_unknown_fields)]
6pub struct RefreshOptions {
7    #[serde(default = "default_refresh_reg")]
8    pub refresh_reg: Atom,
9    #[serde(default = "default_refresh_sig")]
10    pub refresh_sig: Atom,
11    #[serde(default = "default_emit_full_signatures")]
12    pub emit_full_signatures: bool,
13}
14
15fn default_refresh_reg() -> Atom {
16    atom!("$RefreshReg$")
17}
18fn default_refresh_sig() -> Atom {
19    atom!("$RefreshSig$")
20}
21fn default_emit_full_signatures() -> bool {
22    // Prefer to hash when we can
23    // This makes it deterministically compact, even if there's
24    // e.g. a useState initializer with some code inside.
25    false
26}
27
28impl Default for RefreshOptions {
29    fn default() -> Self {
30        RefreshOptions {
31            refresh_reg: default_refresh_reg(),
32            refresh_sig: default_refresh_sig(),
33            emit_full_signatures: default_emit_full_signatures(),
34        }
35    }
36}
37
38#[derive(Deserialize)]
39#[serde(untagged)]
40enum BoolOrRefresh {
41    Bool(bool),
42    Refresh(RefreshOptions),
43}
44pub fn deserialize_refresh<'de, D>(deserializer: D) -> Result<Option<RefreshOptions>, D::Error>
45where
46    D: Deserializer<'de>,
47{
48    match BoolOrRefresh::deserialize(deserializer)? {
49        BoolOrRefresh::Refresh(refresh) => Ok(Some(refresh)),
50        BoolOrRefresh::Bool(true) => Ok(Some(Default::default())),
51        BoolOrRefresh::Bool(false) => Ok(None),
52    }
53}
