24    /// This method do
25    ///
26    /// - inject helpers
27    /// - rename chunks
28    /// - invoke fixer
29    pub(super) fn finalize(
30        &self,
31        bundles: Vec<Bundle>,
32        unresolved_mark: Mark,
33    ) -> Result<Vec<Bundle>, Error> {
34        self.run(|| {
35            let mut new = Vec::with_capacity(bundles.len());
36            let mut renamed = FxHashMap::default();
37
38            for mut bundle in bundles {
39                bundle.module = self.optimize(bundle.module);
40
41                if !self.config.disable_hygiene {
42                    bundle.module.visit_mut_with(&mut hygiene());
43                }
44
45                bundle.module = self.may_wrap_with_iife(bundle.module);
46
47                if !self.config.disable_fixer {
48                    bundle.module.visit_mut_with(&mut fixer(None));
49                }
50
51                {
52                    // Inject swc helpers
53                    let swc_helpers = *self
54                        .scope
55                        .get_module(bundle.id)
56                        .expect("module should exist at this point")
57                        .swc_helpers
58                        .lock();
59
60                    HELPERS.set(&Helpers::from_data(swc_helpers), || {
61                        bundle
62                            .module
63                            .visit_mut_with(&mut inject_helpers(unresolved_mark));
64                    });
65                }
66
67                match bundle.kind {
68                    BundleKind::Named { .. } => {
69                        // Inject helpers
70                        let helpers = self
71                            .scope
72                            .get_module(bundle.id)
73                            .expect("module should exist at this point")
74                            .helpers;
75
76                        helpers.add_to(&mut bundle.module.body);
77
78                        new.push(bundle);
79                    }
80                    BundleKind::Lib { name } => {
81                        let hash = calc_hash(self.cm.clone(), &bundle.module)?;
82                        let mut new_name = PathBuf::from(name);
83                        let key = new_name.clone();
84                        let file_name = new_name
85                            .file_name()
86                            .map(|path| -> PathBuf {
87                                let path = Path::new(path);
88                                let ext = path.extension();
89                                if let Some(ext) = ext {
90                                    return format!(
91                                        "{}-{}.{}",
92                                        path.file_stem().unwrap().to_string_lossy(),
93                                        hash,
94                                        ext.to_string_lossy()
95                                    )
96                                    .into();
97                                }
98                                format!("{}-{}", path.file_stem().unwrap().to_string_lossy(), hash,)
99                                    .into()
100                            })
101                            .expect("javascript file should have name");
102                        new_name.pop();
103                        new_name = new_name.join(file_name.clone());
104
105                        renamed.insert(key, new_name.to_string_lossy().to_string());
106
107                        new.push(Bundle {
108                            kind: BundleKind::Named {
109                                name: file_name.display().to_string(),
110                            },
111                            ..bundle
112                        })
113                    }
114                    _ => new.push(bundle),
115                }
116            }
117
118            if new.len() == 1 {
119                return Ok(new);
120            }
121
122            new = new.move_map(|bundle| {
123                let path = match &*self.scope.get_module(bundle.id).unwrap().fm.name {
124                    FileName::Real(ref v) => v.clone(),
125                    _ => {
126                        tracing::error!("Cannot rename: not a real file");
127                        return bundle;
128                    }
129                };
130
131                let module = {
132                    // Change imports
133                    let mut v = Renamer {
134                        resolver: &self.resolver,
135                        base: &path,
136                        renamed: &renamed,
137                    };
138                    bundle.module.fold_with(&mut v)
139                };
140
141                Bundle { module, ..bundle }
142            });
143
144            Ok(new)
145        })
146    }

148    fn may_wrap_with_iife(&self, module: Module) -> Module {
149        if self.config.module != ModuleType::Iife {
150            return module;
151        }
152
153        let is_async = contains_top_level_await(&module);
154
155        // Properties of returned object
156        let mut props = Vec::new();
157
158        let mut body = BlockStmt {
159            span: module.span,
160            stmts: module
161                .body
162                .into_iter()
163                .filter_map(|item| {
164                    let decl = match item {
165                        ModuleItem::ModuleDecl(v) => v,
166                        ModuleItem::Stmt(stmt) => return Some(stmt),
167                    };
168
169                    match decl {
170                        ModuleDecl::ExportNamed(NamedExport { src: Some(..), .. })
171                        | ModuleDecl::TsImportEquals(_)
172                        | ModuleDecl::TsExportAssignment(_)
173                        | ModuleDecl::TsNamespaceExport(_)
174                        | ModuleDecl::Import(_) => None,
175
176                        ModuleDecl::ExportDecl(export) => {
177                            match &export.decl {
178                                Decl::Class(ClassDecl { ident, .. })
179                                | Decl::Fn(FnDecl { ident, .. }) => {
180                                    props.push(PropOrSpread::Prop(Box::new(Prop::Shorthand(
181                                        ident.clone(),
182                                    ))));
183                                }
184                                Decl::Var(decl) => {
185                                    let ids: Vec<Ident> = find_pat_ids(decl);
186                                    props.extend(
187                                        ids.into_iter()
188                                            .map(Prop::Shorthand)
189                                            .map(Box::new)
190                                            .map(PropOrSpread::Prop),
191                                    );
192                                }
193                                _ => unreachable!(),
194                            }
195
196                            Some(export.decl.into())
197                        }
198
199                        ModuleDecl::ExportNamed(NamedExport {
200                            specifiers,
201                            src: None,
202                            ..
203                        }) => {
204                            for s in specifiers {
205                                match s {
206                                    ExportSpecifier::Namespace(..) => {
207                                        // unreachable
208                                    }
209                                    ExportSpecifier::Default(s) => {
210                                        props.push(PropOrSpread::Prop(Box::new(Prop::KeyValue(
211                                            KeyValueProp {
212                                                key: PropName::Ident(IdentName::new(
213                                                    atom!("default"),
214                                                    DUMMY_SP,
215                                                )),
216                                                value: s.exported.into(),
217                                            },
218                                        ))));
219                                    }
220                                    ExportSpecifier::Named(s) => match s.exported {
221                                        Some(ModuleExportName::Ident(exported)) => {
222                                            let orig = match s.orig {
223                                                ModuleExportName::Ident(ident) => ident,
224                                                ModuleExportName::Str(..) => unimplemented!(
225                                                    "module string names unimplemented"
226                                                ),
227                                            };
228                                            props.push(PropOrSpread::Prop(Box::new(
229                                                Prop::KeyValue(KeyValueProp {
230                                                    key: PropName::Ident(exported.into()),
231                                                    value: orig.into(),
232                                                }),
233                                            )));
234                                        }
235                                        Some(ModuleExportName::Str(..)) => {
236                                            unimplemented!("module string names unimplemented")
237                                        }
238                                        None => {
239                                            let orig = match s.orig {
240                                                ModuleExportName::Ident(ident) => ident,
241                                                ModuleExportName::Str(..) => unimplemented!(
242                                                    "module string names unimplemented"
243                                                ),
244                                            };
245                                            props.push(PropOrSpread::Prop(Box::new(
246                                                Prop::Shorthand(orig),
247                                            )));
248                                        }
249                                    },
250                                }
251                            }
252
253                            None
254                        }
255
256                        ModuleDecl::ExportDefaultDecl(export) => match export.decl {
257                            DefaultDecl::Class(expr) => {
258                                let ident = expr.ident;
259                                let ident =
260                                    ident.unwrap_or_else(|| private_ident!("_default_decl"));
261
262                                props.push(PropOrSpread::Prop(Box::new(Prop::KeyValue(
263                                    KeyValueProp {
264                                        key: PropName::Ident(IdentName::new(
265                                            atom!("default"),
266                                            export.span,
267                                        )),
268                                        value: ident.clone().into(),
269                                    },
270                                ))));
271
272                                Some(
273                                    ClassDecl {
274                                        ident,
275                                        class: expr.class,
276                                        declare: false,
277                                    }
278                                    .into(),
279                                )
280                            }
281                            DefaultDecl::Fn(expr) => {
282                                let ident = expr.ident;
283                                let ident =
284                                    ident.unwrap_or_else(|| private_ident!("_default_decl"));
285
286                                props.push(PropOrSpread::Prop(Box::new(Prop::KeyValue(
287                                    KeyValueProp {
288                                        key: PropName::Ident(IdentName::new(
289                                            atom!("default"),
290                                            export.span,
291                                        )),
292                                        value: ident.clone().into(),
293                                    },
294                                ))));
295
296                                Some(
297                                    FnDecl {
298                                        ident,
299                                        function: expr.function,
300                                        declare: false,
301                                    }
302                                    .into(),
303                                )
304                            }
305                            DefaultDecl::TsInterfaceDecl(_) => None,
306                        },
307                        ModuleDecl::ExportDefaultExpr(export) => {
308                            let default_var = private_ident!("default");
309                            props.push(PropOrSpread::Prop(Box::new(Prop::Shorthand(
310                                default_var.clone(),
311                            ))));
312                            let var = VarDeclarator {
313                                span: DUMMY_SP,
314                                name: default_var.into(),
315                                init: Some(export.expr),
316                                definite: false,
317                            };
318                            Some(
319                                VarDecl {
320                                    span: DUMMY_SP,
321                                    kind: VarDeclKind::Const,
322                                    declare: false,
323                                    decls: vec![var],
324                                    ..Default::default()
325                                }
326                                .into(),
327                            )
328                        }
329
330                        ModuleDecl::ExportAll(_) => None,
331                    }
332                })
333                .collect(),
334            ..Default::default()
335        };
336        body.stmts.push(
337            ReturnStmt {
338                span: DUMMY_SP,
339                arg: Some(
340                    ObjectLit {
341                        span: DUMMY_SP,
342                        props,
343                    }
344                    .into(),
345                ),
346            }
347            .into(),
348        );
349
350        let f = Function {
351            is_generator: false,
352            is_async,
353            span: DUMMY_SP,
354            body: Some(body),
355            ..Default::default()
356        };
357
358        let invoked_fn_expr = FnExpr {
359            ident: None,
360            function: Box::new(f),
361        };
362
363        let iife = CallExpr {
364            span: DUMMY_SP,
365            callee: invoked_fn_expr.as_callee(),
366            args: Default::default(),
367            ..Default::default()
368        }
369        .into();
370
371        Module {
372            span: DUMMY_SP,
373            shebang: None,
374            body: vec![ModuleItem::Stmt(Stmt::Expr(ExprStmt {
375                span: DUMMY_SP,
376                expr: iife,
377            }))],
378        }
379    }

382/// Import renamer. This pass changes import path.
383struct Renamer<'a, R>
384where
385    R: Resolve,
386{
387    resolver: R,
388    base: &'a PathBuf,
389    renamed: &'a FxHashMap<PathBuf, String>,
390}
391
392impl<R> Fold for Renamer<'_, R>
393where
394    R: Resolve,
395{
396    noop_fold_type!();
397
398    fn fold_import_decl(&mut self, import: ImportDecl) -> ImportDecl {
399        let resolved = match self
400            .resolver
401            .resolve(&FileName::Real(self.base.clone()), &import.src.value)
402        {
403            Ok(v) => match v.filename {
404                FileName::Real(v) => v,
405                _ => panic!("rename_bundles called with non-path module"),
406            },
407            Err(_) => return import,
408        };
409
410        if let Some(v) = self.renamed.get(&resolved) {
411            // We use parent because RelativePath uses ../common-[hash].js
412            // if we use `entry-a.js` as a base.
413            //
414            // entry-a.js
415            // common.js
416            let base = self
417                .base
418                .parent()
419                .unwrap_or(self.base)
420                .as_os_str()
421                .to_string_lossy();
422            let base = RelativePath::new(&*base);
423            let v = base.relative(v);
424            let value = v.as_str();
425            return ImportDecl {
426                src: Box::new(Str {
427                    value: if value.starts_with('.') {
428                        value.into()
429                    } else {
430                        format!("./{value}").into()
431                    },
432                    ..*import.src
433                }),
434                ..import
435            };
436        }
437
438        import
439    }
