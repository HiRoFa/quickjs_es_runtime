20    /// This method de-globs imports if possible and colorizes imported values.
21    pub(super) fn extract_import_info(
22        &self,
23        path: &FileName,
24        module: &mut Module,
25        module_local_mark: Mark,
26    ) -> RawImports {
27        self.run(|| {
28            let mut v = ImportHandler {
29                module_ctxt: SyntaxContext::empty().apply_mark(module_local_mark),
30                path,
31                bundler: self,
32                top_level: false,
33                info: Default::default(),
34                usages: Default::default(),
35                imported_idents: Default::default(),
36                deglob_phase: false,
37                idents_to_deglob: Default::default(),
38                in_obj_of_member: false,
39            };
40            module.body.visit_mut_with(&mut v);
41            v.deglob_phase = true;
42            module.body.visit_mut_with(&mut v);
43
44            v.info
45        })
46    }
47
48    pub(super) fn resolve(
49        &self,
50        base: &FileName,
51        module_specifier: &str,
52    ) -> Result<Lrc<FileName>, Error> {
53        self.run(|| {
54            let path = self
55                .resolver
56                .resolve(base, module_specifier)
57                .map(|v| v.filename)
58                .with_context(|| format!("failed to resolve {module_specifier} from {base}"))?;
59
60            let path = Lrc::new(path);
61
62            Ok(path)
63        })
64    }
65}

98/// This type implements two operation (analysis, deglobbing) to reduce binary
99/// size.
100struct ImportHandler<'a, 'b, L, R>
101where
102    L: Load,
103    R: Resolve,
104{
105    /// The [SyntaxContext] for the top level module items.
106    //// The top level module items includes imported bindings.
107    module_ctxt: SyntaxContext,
108    path: &'a FileName,
109    bundler: &'a Bundler<'b, L, R>,
110    top_level: bool,
111    info: RawImports,
112
113    /// HashMap from the local identifier of a namespace import to used
114    /// properties.
115    usages: FxHashMap<Id, Vec<Id>>,
116
117    /// While deglobbing, we also marks imported identifiers.
118    imported_idents: FxHashMap<Id, SyntaxContext>,
119
120    deglob_phase: bool,
121    idents_to_deglob: FxHashSet<Id>,
122
123    /// `true` while folding objects of a member expression.
124    ///
125    /// This is used to distinguish usage of `a` in `console.log(a)` and
126    /// `a.join()`.
127    in_obj_of_member: bool,
128}

130impl RawImports {
131    fn insert(&mut self, import: &ImportDecl) {
132        for prev in self.imports.iter_mut() {
133            if prev.src.value == import.src.value {
134                prev.specifiers.extend(import.specifiers.clone());
135                return;
136            }
137        }
138
139        self.imports.push(import.clone());
140    }
141}

143impl<L, R> ImportHandler<'_, '_, L, R>
144where
145    L: Load,
146    R: Resolve,
147{
148    /// Returns (local, export)
149    fn ctxt_for(&self, src: &Atom) -> Option<(SyntaxContext, SyntaxContext)> {
150        // Don't apply mark if it's a core module.
151        if self.bundler.is_external(src) {
152            return None;
153        }
154        let path = self.bundler.resolve(self.path, src).ok()?;
155        let (_, local_mark, export_mark) = self.bundler.scope.module_id_gen.gen(&path);
156
157        Some((
158            SyntaxContext::empty().apply_mark(local_mark),
159            SyntaxContext::empty().apply_mark(export_mark),
160        ))
161    }
162
163    fn mark_as_wrapping_required(&self, src: &Atom) {
164        // Don't apply mark if it's a core module.
165        if self.bundler.is_external(src) {
166            return;
167        }
168        let path = self.bundler.resolve(self.path, src);
169        let path = match path {
170            Ok(v) => v,
171            Err(_) => return,
172        };
173        let (id, _, _) = self.bundler.scope.module_id_gen.gen(&path);
174
175        self.bundler.scope.mark_as_wrapping_required(id);
176    }
177
178    fn mark_as_cjs(&self, src: &Atom) {
179        let path = self.bundler.resolve(self.path, src);
180        let path = match path {
181            Ok(v) => v,
182            Err(_) => return,
183        };
184        let (id, _, _) = self.bundler.scope.module_id_gen.gen(&path);
185
186        self.bundler.scope.mark_as_cjs(id);
187    }
188
189    fn add_forced_ns_for(&mut self, id: Id) {
190        self.idents_to_deglob.remove(&id);
191        self.imported_idents.remove(&id);
192        self.usages.remove(&id);
193
194        if let Some(src) = self
195            .info
196            .imports
197            .iter()
198            .find(|import| {
199                import.specifiers.iter().any(|specifier| match specifier {
200                    ImportSpecifier::Namespace(ns) => ns.local.sym == id.0 && ns.local.ctxt == id.1,
201                    _ => false,
202                })
203            })
204            .map(|import| import.src.value.clone())
205        {
206            self.info.forced_ns.insert(src);
207        }
208    }
209
210    fn find_require(&mut self, e: &mut Expr) {
211        match e {
212            Expr::Call(e) if e.args.len() == 1 => {
213                let src = match e.args.first().unwrap() {
214                    ExprOrSpread { spread: None, expr } => match &**expr {
215                        Expr::Lit(Lit::Str(s)) => s,
216                        _ => return,
217                    },
218                    _ => return,
219                };
220
221                match &mut e.callee {
222                    Callee::Expr(callee)
223                        if self.bundler.config.require && callee.is_ident_ref_to("require") =>
224                    {
225                        if self.bundler.is_external(&src.value) {
226                            return;
227                        }
228                        if let Expr::Ident(i) = &mut **callee {
229                            self.mark_as_cjs(&src.value);
230                            if let Some((_, export_ctxt)) = self.ctxt_for(&src.value) {
231                                i.ctxt = export_ctxt;
232                            }
233                        }
234
235                        let span = callee.span();
236
237                        let decl = ImportDecl {
238                            span,
239                            specifiers: Vec::new(),
240                            src: Box::new(src.clone()),
241                            type_only: false,
242                            with: None,
243                            phase: Default::default(),
244                        };
245
246                        if self.top_level {
247                            self.info.insert(&decl);
248                            return;
249                        }
250
251                        self.info.lazy_imports.push(decl);
252                    }
253
254                    // TODO: Uncomment this after implementing an option to make swc_bundler
255                    // includes dynamic imports
256                    //
257                    //
258                    // ExprOrSuper::Expr(ref e) => match &**e {
259                    //     Expr::Ident(Ident {
260                    //         sym: "import",
261                    //         ..
262                    //     }) => {
263                    //         self.info.dynamic_imports.push(src.clone());
264                    //     }
265                    //     _ => {}
266                    // },
267                    _ => {}
268                }
269            }
270            _ => {}
271        }
272    }
273
274    fn analyze_usage(&mut self, e: &mut Expr) {
275        if let Expr::Member(e) = e {
276            if let Expr::Ident(obj) = &*e.obj {
277                if !self.imported_idents.contains_key(&obj.to_id()) {
278                    // If it's not imported, just abort the usage analysis.
279                    return;
280                }
281
282                if e.prop.is_computed() {
283                    // If a module is accessed with unknown key, we should import
284                    // everything from it.
285                    self.add_forced_ns_for(obj.to_id());
286                    return;
287                }
288
289                // Store usages of obj
290                let import = self.info.imports.iter().find(|import| {
291                    for s in &import.specifiers {
292                        if let ImportSpecifier::Namespace(n) = s {
293                            return obj.sym == n.local.sym
294                                && (obj.ctxt == self.module_ctxt || obj.ctxt == n.local.ctxt);
295                        }
296                    }
297
298                    false
299                });
300                let import = match import {
301                    Some(v) => v,
302                    None => return,
303                };
304
305                let mark = self.ctxt_for(&import.src.value);
306                let exported_ctxt = match mark {
307                    None => return,
308                    Some(ctxts) => ctxts.1,
309                };
310                let prop = match &e.prop {
311                    MemberProp::Ident(i) => {
312                        let mut i = Ident::from(i.clone());
313                        i.ctxt = exported_ctxt;
314                        i
315                    }
316                    _ => unreachable!(
317                        "Non-computed member expression with property other than ident is invalid"
318                    ),
319                };
320
321                self.usages
322                    .entry(obj.to_id())
323                    .or_default()
324                    .push(prop.to_id());
325            }
326        }
327    }
328
329    fn try_deglob(&mut self, e: &mut Expr) {
330        let me = match e {
331            Expr::Member(e) => e,
332            _ => return,
333        };
334        if me.prop.is_computed() {
335            return;
336        }
337
338        let obj = match &*me.obj {
339            Expr::Ident(obj) => obj,
340            _ => return,
341        };
342
343        let usages = self.usages.get(&obj.to_id());
344
345        match usages {
346            Some(..) => {}
347            _ => return,
348        };
349
350        let mut prop = match &me.prop {
351            MemberProp::Ident(v) => Ident::from(v.clone()),
352            _ => return,
353        };
354        prop.ctxt = self.imported_idents.get(&obj.to_id()).copied().unwrap();
355
356        *e = prop.into();
357    }
358}

360impl<L, R> VisitMut for ImportHandler<'_, '_, L, R>
361where
362    L: Load,
363    R: Resolve,
364{
365    noop_visit_mut_type!(fail);

367    fn visit_mut_export_named_specifier(&mut self, s: &mut ExportNamedSpecifier) {
368        let orig = match &s.orig {
369            ModuleExportName::Ident(ident) => ident,
370            ModuleExportName::Str(..) => unimplemented!("module string names unimplemented"),
371        };
372
373        self.add_forced_ns_for(orig.to_id());
374
375        match &mut s.exported {
376            Some(ModuleExportName::Ident(exported)) => {
377                // PR 3139 (https://github.com/swc-project/swc/pull/3139) removes the syntax context from any named exports from other sources.
378                exported.ctxt = self.module_ctxt;
379            }
380            Some(ModuleExportName::Str(..)) => unimplemented!("module string names unimplemented"),
381            None => {
382                let exported = Ident::new(orig.sym.clone(), orig.span, self.module_ctxt);
383                s.exported = Some(ModuleExportName::Ident(exported));
384            }
385        }
386    }
387
388    fn visit_mut_expr(&mut self, e: &mut Expr) {
389        e.visit_mut_children_with(self);
390
391        if !self.deglob_phase {
392            // Firstly, we check for usages of imported namespaces.
393            // Code like below are handled by this check.
394            //
395            // import * as log from './log';
396            // console.log(log)
397            // console.log(log.getLogger())
398            if !self.in_obj_of_member {
399                if let Expr::Ident(i) = &e {
400                    if !self.in_obj_of_member {
401                        self.add_forced_ns_for(i.to_id());
402                        return;
403                    }
404                }
405            }
406
407            self.analyze_usage(e);
408            self.find_require(e);
409        } else {
410            self.try_deglob(e);
411        }
412    }
413
414    fn visit_mut_import_decl(&mut self, import: &mut ImportDecl) {
415        // Ignore if it's a core module.
416        if self.bundler.is_external(&import.src.value) {
417            return;
418        }
419
420        if !self.deglob_phase {
421            if let Some((_, export_ctxt)) = self.ctxt_for(&import.src.value) {
422                // Firstly we attach proper syntax contexts.
423                ExportMetadata {
424                    export_ctxt: Some(export_ctxt),
425                    ..Default::default()
426                }
427                .encode(&mut import.with);
428
429                // Then we store list of imported identifiers.
430                for specifier in &mut import.specifiers {
431                    match specifier {
432                        ImportSpecifier::Named(n) => {
433                            self.imported_idents.insert(n.local.to_id(), export_ctxt);
434                            match &mut n.imported {
435                                Some(ModuleExportName::Ident(imported)) => {
436                                    imported.ctxt = export_ctxt;
437                                }
438                                Some(ModuleExportName::Str(..)) => {
439                                    unimplemented!("module string names unimplemented")
440                                }
441                                None => {
442                                    let mut imported: Ident = n.local.clone();
443                                    imported.ctxt = export_ctxt;
444                                    n.imported = Some(ModuleExportName::Ident(imported));
445                                }
446                            }
447                        }
448                        ImportSpecifier::Default(n) => {
449                            self.imported_idents.insert(n.local.to_id(), n.local.ctxt);
450                        }
451                        ImportSpecifier::Namespace(n) => {
452                            self.imported_idents.insert(n.local.to_id(), export_ctxt);
453                        }
454                    }
455                }
456            }
457
458            self.info.insert(import);
459            return;
460        }
461
462        // Now we are in deglobbing phase.
463
464        // We cannot deglob this.
465        if self.info.forced_ns.contains(&import.src.value) {
466            return;
467        }
468
469        // deglob namespace imports
470        if import.specifiers.len() == 1 {
471            if let ImportSpecifier::Namespace(ns) = &import.specifiers[0] {
472                //
473                let specifiers = self
474                    .usages
475                    .get(&ns.local.to_id())
476                    .cloned()
477                    .map(|ids| {
478                        //
479                        let specifiers: Vec<_> = ids
480                            .into_iter()
481                            .map(|id| {
482                                self.idents_to_deglob.insert(id.clone());
483                                ImportSpecifier::Named(ImportNamedSpecifier {
484                                    span: DUMMY_SP,
485                                    local: Ident::new(id.0, DUMMY_SP, id.1),
486                                    imported: None,
487                                    is_type_only: false,
488                                })
489                            })
490                            .collect();
491
492                        for import_info in &mut self.info.imports {
493                            if import_info.src != import.src {
494                                continue;
495                            }
496
497                            import_info.specifiers.extend(specifiers.clone());
498                        }
499
500                        specifiers
501                    })
502                    .unwrap_or_else(Vec::new);
503
504                if !specifiers.is_empty() {
505                    import.specifiers = specifiers;
506                    return;
507                }
508
509                // We failed to found property usage.
510                self.info.forced_ns.insert(import.src.value.clone());
511            }
512        }
513    }
514
515    fn visit_mut_member_expr(&mut self, e: &mut MemberExpr) {
516        let old = self.in_obj_of_member;
517        self.in_obj_of_member = true;
518        e.obj.visit_mut_with(self);
519
520        if let MemberProp::Computed(c) = &mut e.prop {
521            self.in_obj_of_member = false;
522            c.visit_mut_with(self);
523        }
524
525        self.in_obj_of_member = old;
526    }
527
528    fn visit_mut_module_items(&mut self, items: &mut Vec<ModuleItem>) {
529        self.top_level = true;
530        items.visit_mut_children_with(self);
531
532        items.retain_mut(|item| match item {
533            ModuleItem::Stmt(Stmt::Empty(..)) => false,
534            ModuleItem::Stmt(Stmt::Decl(Decl::Var(var))) => {
535                var.decls.retain(|d| !matches!(d.name, Pat::Invalid(..)));
536
537                !var.decls.is_empty()
538            }
539
540            _ => true,
541        });
542
543        if self.deglob_phase {
544            let mut wrapping_required = Vec::new();
545            for import in self.info.imports.iter_mut() {
546                let use_ns = self.info.forced_ns.contains(&import.src.value)
547                    || self
548                        .bundler
549                        .config
550                        .external_modules
551                        .contains(&import.src.value);
552
553                if use_ns {
554                    wrapping_required.push(import.src.value.clone());
555                } else {
556                    // De-glob namespace imports
557                    import
558                        .specifiers
559                        .retain(|s| !matches!(s, ImportSpecifier::Namespace(_)));
560                }
561            }
562
563            for id in wrapping_required {
564                self.mark_as_wrapping_required(&id);
565            }
566        }
567    }
568
569    fn visit_mut_stmts(&mut self, items: &mut Vec<Stmt>) {
570        self.top_level = false;
571        items.visit_mut_children_with(self)
572    }
573
574    fn visit_mut_super_prop_expr(&mut self, e: &mut SuperPropExpr) {
575        let old = self.in_obj_of_member;
576
577        if let SuperProp::Computed(c) = &mut e.prop {
578            self.in_obj_of_member = false;
579            c.visit_mut_with(self);
580        }
581
582        self.in_obj_of_member = old;
583    }

585    /// ```js
586    /// const { readFile } = required('fs');
587    /// ```
588    ///
589    /// is treated as
590    ///
591    /// ```js
592    /// import { readFile } from 'fs';
593    /// ```
594    fn visit_mut_var_declarator(&mut self, node: &mut VarDeclarator) {
595        node.visit_mut_children_with(self);
596
597        if let Some(init) = &mut node.init {
598            match &mut **init {
599                Expr::Call(CallExpr {
600                    span,
601                    callee: Callee::Expr(ref mut callee),
602                    ref args,
603                    ..
604                }) if self.bundler.config.require
605                    && callee.is_ident_ref_to("require")
606                    && args.len() == 1 =>
607                {
608                    let span = *span;
609                    let src = match args.first().unwrap() {
610                        ExprOrSpread { spread: None, expr } => match &**expr {
611                            Expr::Lit(Lit::Str(s)) => s.clone(),
612                            _ => return,
613                        },
614                        _ => return,
615                    };
616                    // Ignore core modules.
617                    if self.bundler.config.external_modules.contains(&src.value) {
618                        return;
619                    }
620
621                    self.mark_as_cjs(&src.value);
622
623                    if let Expr::Ident(i) = &mut **callee {
624                        if let Some((_, export_ctxt)) = self.ctxt_for(&src.value) {
625                            i.ctxt = export_ctxt;
626                        }
627                    }
628
629                    let ids: Vec<Ident> = find_pat_ids(&node.name);
630
631                    let decl = ImportDecl {
632                        span,
633                        specifiers: ids
634                            .into_iter()
635                            .map(|ident| {
636                                ImportSpecifier::Named(ImportNamedSpecifier {
637                                    span,
638                                    local: ident,
639                                    imported: None,
640                                    is_type_only: false,
641                                })
642                            })
643                            .collect(),
644                        src: Box::new(src),
645                        type_only: false,
646                        with: None,
647                        phase: Default::default(),
648                    };
649
650                    // if self.top_level {
651                    //     self.info.imports.push(decl);
652                    //     node.init = None;
653                    //     node.name = Pat::Invalid(Invalid { span: DUMMY_SP });
654                    //     return node;
655                    // }
656
657                    self.info.lazy_imports.push(decl);
658                }
659
660                _ => {}
661            }
662        }
663    }
