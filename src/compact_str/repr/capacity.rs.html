1use crate::repr::HEAP_MASK;
2
3// how many bytes a `usize` occupies
4const USIZE_SIZE: usize = core::mem::size_of::<usize>();
5
6/// Used to generate [`CAPACITY_IS_ON_THE_HEAP`]
7#[allow(non_snake_case)]
8const fn CAP_ON_HEAP_FLAG() -> [u8; USIZE_SIZE] {
9    // all bytes 255, with the last being HEAP_MASK
10    let mut flag = [255; USIZE_SIZE];
11    flag[USIZE_SIZE - 1] = HEAP_MASK;
12    flag
13}
14
15/// State that describes the capacity as being stored on the heap.
16///
17/// All bytes `255`, with the last being [`HEAP_MASK`], using the same amount of bytes as `usize`
18/// Example (64-bit): `[255, 255, 255, 255, 255, 255, 255, 254]`
19const CAPACITY_IS_ON_THE_HEAP: [u8; USIZE_SIZE] = CAP_ON_HEAP_FLAG();
20
21// how many bytes we can use for capacity
22const SPACE_FOR_CAPACITY: usize = USIZE_SIZE - 1;
23// the maximum value we're able to store, e.g. on 64-bit arch this is 2^56 - 2
24//
25// note: Preferably we'd used usize.pow(..) here, but that's not a `const fn`, so we need to use
26// bitshift operators, and there's a lint against using them in this pattern, which IMO isn't a
27// great lint
28pub const MAX_VALUE: usize = 2usize.pow(SPACE_FOR_CAPACITY as u32 * 8) - 2;
29
30/// An integer type that uses `core::mem::size_of::<usize>() - 1` bytes to store the capacity of
31/// a heap buffer.
32///
33/// Assumming a 64-bit arch, a [`super::BoxString`] uses 8 bytes for a pointer, 8 bytes for a
34/// length, and then needs 1 byte for a discriminant. We need to store the capacity somewhere, and
35/// we could store it on the heap, but we also have 7 unused bytes. [`Capacity`] handles storing a
36/// value in these 7 bytes, returning an error if it's not possible, at which point we'll store the
37/// capacity on the heap.
38///
39/// # Max Values
40/// * __64-bit:__ `(2 ^ (7 * 8)) - 2 = 72_057_594_037_927_934 ~= 64 petabytes`
41/// * __32-bit:__ `(2 ^ (3 * 8)) - 2 = 16_777_214             ~= 16 megabytes`
42///
43/// Practically speaking, on a 64-bit architecture we'll never need to store the capacity on the
44/// heap, because with it's impossible to create a string that is 64 petabytes or larger. But for
45/// 32-bit architectures we need to be able to store a capacity larger than 16 megabytes, since a
46/// string larger than 16 megabytes probably isn't that uncommon.
47#[derive(Copy, Clone, Debug, PartialEq, Eq)]
48#[cfg_attr(target_pointer_width = "64", repr(align(8)))]
49#[cfg_attr(target_pointer_width = "32", repr(align(4)))]
50pub struct Capacity([u8; USIZE_SIZE]);
51
52static_assertions::assert_eq_size!(Capacity, usize);
53static_assertions::assert_eq_align!(Capacity, usize);
54
55impl Capacity {
56    #[inline]
57    pub const fn new(capacity: usize) -> Self {
58        cfg_if::cfg_if! {
59            if #[cfg(target_pointer_width = "64")] {
60                // on 64-bit arches we can always fit the capacity inline
61                debug_assert!(capacity <= MAX_VALUE);
62
63                let mut bytes = capacity.to_le_bytes();
64                bytes[core::mem::size_of::<usize>() - 1] = HEAP_MASK;
65                Capacity(bytes)
66            } else if #[cfg(target_pointer_width = "32")] {
67                // on 32-bit arches we might need to store the capacity on the heap
68                if capacity > MAX_VALUE {
69                    // if we need the last byte to encode this capacity then we need to put the capacity on
70                    // the heap. return an Error so `BoxString` can do the right thing
71                    Capacity(CAPACITY_IS_ON_THE_HEAP)
72                } else {
73                    // otherwise, we can store this capacity inline! Set the last byte to be our `HEAP_MASK`
74                    // for our discriminant, using the leading bytes to store the actual value
75                    let mut bytes = capacity.to_le_bytes();
76                    bytes[core::mem::size_of::<usize>() - 1] = HEAP_MASK;
77                    Capacity(bytes)
78                }
79            } else {
80                compile_error!("Unsupported target_pointer_width");
81            }
82        }
83    }
84
85    /// Re-interprets a [`Capacity`] as a `usize`
86    ///
87    /// # SAFETY:
88    /// * `self` must be less than or equal to [`MAX_VALUE`]
89    #[inline(always)]
90    pub unsafe fn as_usize(&self) -> usize {
91        let mut usize_buf = [0u8; USIZE_SIZE];
92        // SAFETY:
93        // * `src` is valid for reads of `SPACE_FOR_CAPACITY` because it is less than `USIZE_SIZE`
94        // * `dst` is valid for reads of `SPACE_FOR_CAPACITY` because it is less than `USIZE_SIZE`
95        // * `src` and `dst` do not overlap because we created `usize_buf`
96        core::ptr::copy_nonoverlapping(self.0.as_ptr(), usize_buf.as_mut_ptr(), SPACE_FOR_CAPACITY);
97        usize::from_le_bytes(usize_buf)
98    }
99
100    /// Returns whether or not this [`Capacity`] has a value that indicates the capacity is being
101    /// stored on the heap
102    #[inline(always)]
103    pub fn is_heap(&self) -> bool {
104        self.0 == CAPACITY_IS_ON_THE_HEAP
105    }
106}

108#[cfg(test)]
109mod tests {
110    use rayon::prelude::*;
111
112    use super::Capacity;
113
114    #[test]
115    fn test_zero_roundtrips() {
116        let og = 0;
117        let cap = Capacity::new(og);
118        let after = unsafe { cap.as_usize() };
119
120        assert_eq!(og, after);
121    }
122
123    #[test]
124    fn test_max_value() {
125        let available_bytes = (core::mem::size_of::<usize>() - 1) as u32;
126        let max_value = 2usize.pow(available_bytes * 8) - 2;
127
128        #[cfg(target_pointer_width = "64")]
129        assert_eq!(max_value, 72057594037927934);
130        #[cfg(target_pointer_width = "32")]
131        assert_eq!(max_value, 16777214);
132
133        let cap = Capacity::new(max_value);
134        let after = unsafe { cap.as_usize() };
135
136        assert_eq!(max_value, after);
137    }
138
139    #[cfg(target_pointer_width = "32")]
140    #[test]
141
142    fn test_invalid_value() {
143        let invalid_val = usize::MAX;
144        let cap = Capacity::new(invalid_val);
145        let after = unsafe { cap.as_usize() };
146
147        // anything greater than or equal to 16777215, should "resolve" to 16777215
148        assert_eq!(16777215, after);
149    }
150
151    #[test]
152    #[cfg_attr(miri, ignore)]
153    fn test_all_valid_32bit_values() {
154        #[cfg(target_pointer_width = "32")]
155        assert_eq!(16_777_214, super::MAX_VALUE);
156
157        (0..=16_777_214).into_par_iter().for_each(|i| {
158            let cap = Capacity::new(i);
159            let val = unsafe { cap.as_usize() };
160
161            assert_eq!(val, i, "value roundtriped to wrong value?");
162        });
163    }
