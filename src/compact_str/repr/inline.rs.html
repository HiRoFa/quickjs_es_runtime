4    Repr,
5    LENGTH_MASK,

9/// A buffer stored on the stack whose size is equal to the stack size of `String`
10#[repr(transparent)]
11pub struct InlineBuffer(pub [u8; MAX_SIZE]);
12static_assertions::assert_eq_size!(InlineBuffer, Repr);
13
14impl InlineBuffer {
15    /// Construct a new [`InlineString`]. A string that lives in a small buffer on the stack
16    ///
17    /// SAFETY:
18    /// * The caller must guarantee that the length of `text` is less than [`MAX_SIZE`]
19    #[inline]
20    pub unsafe fn new(text: &str) -> Self {
21        debug_assert!(text.len() <= MAX_SIZE);
22
23        let len = text.len();
24        let mut buffer = [0u8; MAX_SIZE];
25
26        // set the length in the last byte
27        buffer[MAX_SIZE - 1] = len as u8 | LENGTH_MASK;
28
29        // copy the string into our buffer
30        //
31        // note: in the case where len == MAX_SIZE, we'll overwrite the len, but that's okay because
32        // when reading the length we can detect that the last byte is part of UTF-8 and return a
33        // length of MAX_SIZE
34        //
35        // SAFETY:
36        // * src (`text`) is valid for `len` bytes because `len` comes from `text`
37        // * dst (`buffer`) is valid for `len` bytes because we assert src is less than MAX_SIZE
38        // * src and dst don't overlap because we created dst
39        //
40        ptr::copy_nonoverlapping(text.as_ptr(), buffer.as_mut_ptr(), len);
41
42        InlineBuffer(buffer)
43    }
44
45    #[inline]
46    pub const fn new_const(text: &str) -> Self {
47        if text.len() > MAX_SIZE {
48            panic!("Provided string has a length greater than our MAX_SIZE");
49        }
50
51        let len = text.len();
52        let mut buffer = [0u8; MAX_SIZE];
53
54        // set the length
55        buffer[MAX_SIZE - 1] = len as u8 | LENGTH_MASK;
56
57        // Note: for loops aren't allowed in `const fn`, hence the while.
58        // Note: Iterating forward results in badly optimized code, because the compiler tries to
59        //       unroll the loop.
60        let text = text.as_bytes();
61        let mut i = len;
62        while i > 0 {
63            buffer[i - 1] = text[i - 1];
64            i -= 1;
65        }
66
67        InlineBuffer(buffer)
68    }
69
70    /// Returns an empty [`InlineBuffer`]
71    #[inline(always)]
72    pub const fn empty() -> Self {
73        Self::new_const("")
74    }
75
76    /// Consumes the [`InlineBuffer`] returning the entire underlying array and the length of the
77    /// string that it contains
78    #[inline]
79    #[cfg(feature = "smallvec")]
80    pub fn into_array(self) -> ([u8; MAX_SIZE], usize) {
81        let mut buffer = self.0;
82
83        let length = core::cmp::min(
84            (buffer[MAX_SIZE - 1].wrapping_sub(LENGTH_MASK)) as usize,
85            MAX_SIZE,
86        );
87
88        let last_byte_ref = &mut buffer[MAX_SIZE - 1];
89
90        // unset the last byte of the buffer if it's just storing the length of the string
91        //
92        // Note: we should never add an `else` statement here, keeping the conditional simple allows
93        // the compiler to optimize this to a conditional-move instead of a branch
94        if length < MAX_SIZE {
95            *last_byte_ref = 0;
96        }
97
98        (buffer, length)
99    }
100
101    /// Set's the length of the content for this [`InlineBuffer`]
102    ///
103    /// # SAFETY:
104    /// * The caller must guarantee that `len` bytes in the buffer are valid UTF-8
105    #[inline]
106    pub unsafe fn set_len(&mut self, len: usize) {
107        debug_assert!(len <= MAX_SIZE);
108
109        // If `length` == MAX_SIZE, then we infer the length to be the capacity of the buffer. We
110        // can infer this because the way we encode length doesn't overlap with any valid UTF-8
111        // bytes
112        if len < MAX_SIZE {
113            self.0[MAX_SIZE - 1] = len as u8 | LENGTH_MASK;
114        }
115    }
116
117    #[inline(always)]
118    pub fn copy(&self) -> Self {
119        InlineBuffer(self.0)
120    }
121}

127    #[test]
128    #[ignore] // we run this in CI, but unless you're compiling in release, this takes a while
129    fn test_unused_utf8_bytes() {
130        // test to validate for all char the first and last bytes are never within a specified range
131        // note: according to the UTF-8 spec it shouldn't be, but we double check that here
132        (0..u32::MAX).into_par_iter().for_each(|i| {
133            if let Ok(c) = char::try_from(i) {
134                let mut buf = [0_u8; 4];
135                c.encode_utf8(&mut buf);
136
137                // check ranges for first byte
138                match buf[0] {
139                    x @ 128..=191 => panic!("first byte within 128..=191, {}", x),
140                    x @ 248..=255 => panic!("first byte within 248..=255, {}", x),
141                    _ => (),
142                }
143
144                // check ranges for last byte
145                match buf[c.len_utf8() - 1] {
146                    x @ 192..=255 => panic!("last byte within 192..=255, {}", x),
147                    _ => (),
148                }
149            }
150        })
151    }

153    #[cfg(feature = "smallvec")]
154    mod smallvec {
155        use quickcheck_macros::quickcheck;
156
157        use crate::repr::{
158            InlineBuffer,
159            MAX_SIZE,
160        };
161
162        #[test]
163        fn test_into_array() {
164            let s = "hello world!";
165
166            let inline = unsafe { InlineBuffer::new(s) };
167            let (array, length) = inline.into_array();
168
169            assert_eq!(s.len(), length);
170
171            // all bytes after the length should be 0
172            assert!(array[length..].iter().all(|b| *b == 0));
173
174            // taking a string slice should give back the same string as the original
175            let ex_s = unsafe { std::str::from_utf8_unchecked(&array[..length]) };
176            assert_eq!(s, ex_s);
177        }
178
179        #[quickcheck]
180        #[cfg_attr(miri, ignore)]
181        fn quickcheck_into_array(s: String) {
182            let mut total_length = 0;
183            let s: String = s
184                .chars()
185                .take_while(|c| {
186                    total_length += c.len_utf8();
187                    total_length < MAX_SIZE
188                })
189                .collect();
190
191            let inline = unsafe { InlineBuffer::new(&s) };
192            let (array, length) = inline.into_array();
193            assert_eq!(s.len(), length);
194
195            // all bytes after the length should be 0
196            assert!(array[length..].iter().all(|b| *b == 0));
197
198            // taking a string slice should give back the same string as the original
199            let ex_s = unsafe { std::str::from_utf8_unchecked(&array[..length]) };
200            assert_eq!(s, ex_s);
201        }
202    }
