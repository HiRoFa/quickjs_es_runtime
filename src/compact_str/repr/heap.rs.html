13/// The minimum size we'll allocate on the heap is one usize larger than our max inline size
14const MIN_HEAP_SIZE: usize = MAX_SIZE + mem::size_of::<usize>();

16const UNKNOWN: usize = 0;
17pub type StrBuffer = [u8; UNKNOWN];

19/// [`HeapBuffer`] grows at an amortized rates of 1.5x
20///
21/// Note: this is different than [`std::string::String`], which grows at a rate of 2x. It's debated
22/// which is better, for now we'll stick with a rate of 1.5x
23#[inline(always)]
24pub fn amortized_growth(cur_len: usize, additional: usize) -> usize {
25    let required = cur_len.saturating_add(additional);
26    let amortized = cur_len.saturating_mul(3) / 2;
27    amortized.max(required)
28}

30#[repr(C)]
31pub struct HeapBuffer {
32    pub ptr: ptr::NonNull<u8>,
33    pub len: usize,
34    pub cap: Capacity,
35}
36static_assertions::assert_eq_size!(HeapBuffer, Repr);

38impl HeapBuffer {
39    /// Create a [`HeapBuffer`] with the provided text
40    #[inline]
41    pub fn new(text: &str) -> Self {
42        let len = text.len();
43        let (cap, ptr) = allocate_ptr(len);
44
45        // copy our string into the buffer we just allocated
46        //
47        // SAFETY: We know both `src` and `dest` are valid for respectively reads and writes of
48        // length `len` because `len` comes from `src`, and `dest` was allocated to be at least that
49        // length. We also know they're non-overlapping because `dest` is newly allocated
50        unsafe { ptr.as_ptr().copy_from_nonoverlapping(text.as_ptr(), len) };
51
52        HeapBuffer { ptr, len, cap }
53    }
54
55    /// Create an empty [`HeapBuffer`] with a specific capacity
56    #[inline]
57    pub fn with_capacity(capacity: usize) -> Self {
58        let len = 0;
59        let (cap, ptr) = allocate_ptr(capacity);
60
61        HeapBuffer { ptr, len, cap }
62    }
63
64    /// Create a [`HeapBuffer`] with `text` that has _at least_ `additional` bytes of capacity
65    ///
66    /// To prevent frequent re-allocations, this method will create a [`HeapBuffer`] with a capacity
67    /// of `text.len() + additional` or `text.len() * 1.5`, whichever is greater
68    #[inline]
69    pub fn with_additional(text: &str, additional: usize) -> Self {
70        let len = text.len();
71        let new_capacity = amortized_growth(len, additional);
72        let (cap, ptr) = allocate_ptr(new_capacity);
73
74        // copy our string into the buffer we just allocated
75        //
76        // SAFETY: We know both `src` and `dest` are valid for respectively reads and writes of
77        // length `len` because `len` comes from `src`, and `dest` was allocated to be at least that
78        // length. We also know they're non-overlapping because `dest` is newly allocated
79        unsafe { ptr.as_ptr().copy_from_nonoverlapping(text.as_ptr(), len) };
80
81        HeapBuffer { ptr, len, cap }
82    }
83
84    /// Return the capacity of the [`HeapBuffer`]
85    #[inline]
86    pub fn capacity(&self) -> usize {
87        #[cold]
88        fn read_capacity_from_heap(this: &HeapBuffer) -> usize {
89            // re-adjust the pointer to include the capacity that's on the heap
90            let adj_ptr: *const u8 = this.ptr.as_ptr().wrapping_sub(mem::size_of::<usize>());
91            let mut buf = [0u8; mem::size_of::<usize>()];
92            // SAFETY: `src` and `dst` don't overlap, and are valid for usize number of bytes
93            unsafe {
94                ptr::copy_nonoverlapping(adj_ptr, buf.as_mut_ptr(), mem::size_of::<usize>());
95            }
96            usize::from_ne_bytes(buf)
97        }
98
99        if self.cap.is_heap() {
100            read_capacity_from_heap(self)
101        } else {
102            // SAFETY: Checked above that the capacity is on the stack
103            unsafe { self.cap.as_usize() }
104        }
105    }
106
107    /// Try to grow the [`HeapBuffer`] by reallocating, returning an error if we fail
108    pub fn realloc(&mut self, new_capacity: usize) -> Result<usize, ()> {
109        let new_cap = Capacity::new(new_capacity);
110
111        // We can't reallocate to a size less than our length, or else we'd clip the string
112        if new_capacity < self.len {
113            return Err(());
114        }
115
116        // HeapBuffer doesn't support 0 byte heap sizes
117        if new_capacity == 0 {
118            return Err(());
119        }
120
121        // Always allocate at least MIN_HEAP_SIZE
122        let new_capacity = cmp::max(new_capacity, MIN_HEAP_SIZE);
123
124        let (new_cap, new_ptr) = match (self.cap.is_heap(), new_cap.is_heap()) {
125            // both current and new capacity can be stored inline
126            (false, false) => {
127                // SAFETY: checked above that our capacity is valid
128                let cap = unsafe { self.cap.as_usize() };
129
130                // current capacity is the same as the new, nothing to do!
131                if cap == new_capacity {
132                    return Ok(new_capacity);
133                }
134
135                let cur_layout = inline_capacity::layout(cap);
136                let new_layout = inline_capacity::layout(new_capacity);
137                let new_size = new_layout.size();
138
139                // It's possible `new_size` could overflow since inline_capacity::layout pads for
140                // alignment
141                if new_size < new_capacity {
142                    return Err(());
143                }
144
145                // SAFETY:
146                // * We're using the same allocator that we used for `ptr`
147                // * The layout is the same because we checked that the capacity is inline
148                // * `new_size` will be > 0, we return early if the requested capacity is 0
149                // * Checked above if `new_size` overflowed when rounding to alignment
150                match ptr::NonNull::new(unsafe {
151                    std::alloc::realloc(self.ptr.as_ptr(), cur_layout, new_size)
152                }) {
153                    Some(ptr) => (new_cap, ptr),
154                    None => return Err(()),
155                }
156            }
157            // both current and new capacity need to be stored on the heap
158            (true, true) => {
159                let cur_layout = heap_capacity::layout(self.capacity());
160                let new_layout = heap_capacity::layout(new_capacity);
161                let new_size = new_layout.size();
162
163                // alloc::realloc requires that size > 0
164                debug_assert!(new_size > 0);
165
166                // It's possible `new_size` could overflow since heap_capacity::layout requires a
167                // few additional bytes
168                if new_size < new_capacity {
169                    return Err(());
170                }
171
172                // move our pointer back one WORD since our capacity is behind it
173                let raw_ptr = self.ptr.as_ptr();
174                let adj_ptr = raw_ptr.wrapping_sub(mem::size_of::<usize>());
175
176                // SAFETY:
177                // * We're using the same allocator that we used for `ptr`
178                // * The layout is the same because we checked that the capacity is on the heap
179                // * `new_size` will be > 0, we return early if the requested capacity is 0
180                // * Checked above if `new_size` overflowed when rounding to alignment
181                let cap_ptr = unsafe { std::alloc::realloc(adj_ptr, cur_layout, new_size) };
182                // Check if reallocation succeeded
183                if cap_ptr.is_null() {
184                    return Err(());
185                }
186
187                // Our allocation succeeded! Write the new capacity
188                //
189                // SAFTEY:
190                // * `src` and `dst` are both valid for reads of `usize` number of bytes
191                // * `src` and `dst` don't overlap because we created `src`
192                unsafe {
193                    ptr::copy_nonoverlapping(
194                        new_capacity.to_ne_bytes().as_ptr(),
195                        cap_ptr,
196                        mem::size_of::<usize>(),
197                    )
198                };
199
200                // Finally, adjust our pointer backup so it points at the string content
201                let str_ptr = cap_ptr.wrapping_add(mem::size_of::<usize>());
202                // SAFETY: We checked above to make sure the pointer was non-null
203                let ptr = unsafe { ptr::NonNull::new_unchecked(str_ptr) };
204
205                (new_cap, ptr)
206            }
207            // capacity is currently inline or on the heap, but needs to move, can't realloc because
208            // we'd need to change the layout!
209            (false, true) | (true, false) => return Err(()),
210        };
211
212        // set our new pointer and new capacity
213        self.ptr = new_ptr;
214        self.cap = new_cap;
215
216        Ok(new_capacity)
217    }

222    /// * The caller must guarantee that `len` bytes in the buffer are valid UTF-8

226    }

228    /// Deallocates the memory owned by the provided [`HeapBuffer`]

231        deallocate_ptr(self.ptr, self.cap);
232    }

235impl Clone for HeapBuffer {
236    fn clone(&self) -> Self {
237        // Create a new HeapBuffer with the same capacity as the original
238        let mut new = Self::with_capacity(self.capacity());
239
240        // SAFETY:
241        // * `src` and `dst` don't overlap because we just created `dst`
242        // * `src` and `dst` are both valid for `self.len` bytes because self.len < capacity
243        unsafe {
244            new.ptr
245                .as_ptr()
246                .copy_from_nonoverlapping(self.ptr.as_ptr(), self.len)
247        };
248        // SAFETY:
249        // * We copied the text from self, which is valid UTF-8
250        unsafe { new.set_len(self.len) };
251
252        new
253    }
254}

262/// Allocates a buffer on the heap that we can use to store a string, optionally stores the capacity
263/// of said buffer on the heap.
264///
265/// Returns a [`Capacity`] that either indicates the capacity is stored on the heap, or is stored
266/// in the `Capacity` itself.
267#[inline]
268pub fn allocate_ptr(capacity: usize) -> (Capacity, ptr::NonNull<u8>) {
269    // We allocate at least MIN_HEAP_SIZE bytes because we need to allocate at least one byte
270    let capacity = capacity.max(MIN_HEAP_SIZE);
271    let cap = Capacity::new(capacity);
272
273    // HeapBuffer doesn't support 0 sized allocations, we should always allocate at least
274    // MIN_HEAP_SIZE bytes
275    debug_assert!(capacity > 0);
276
277    #[cold]
278    fn allocate_with_capacity_on_heap(capacity: usize) -> ptr::NonNull<u8> {
279        // write our capacity onto the heap
280        let ptr = heap_capacity::alloc(capacity);
281        // SAFETY:
282        // * `src` and `dst` don't overlap and are both valid for `usize` bytes
283        unsafe {
284            ptr::copy_nonoverlapping(
285                capacity.to_ne_bytes().as_ptr(),
286                ptr.as_ptr(),
287                mem::size_of::<usize>(),
288            )
289        };
290        let raw_ptr = ptr.as_ptr().wrapping_add(core::mem::size_of::<usize>());
291        // SAFETY: We know `raw_ptr` is non-null because we just created it
292        unsafe { ptr::NonNull::new_unchecked(raw_ptr) }
293    }
294
295    let ptr = if cap.is_heap() {
296        allocate_with_capacity_on_heap(capacity)
297    } else {
298        unsafe { inline_capacity::alloc(capacity) }
299    };
300
301    (cap, ptr)
302}

304/// Deallocates a buffer on the heap, handling when the capacity is also stored on the heap
305#[inline]
306pub fn deallocate_ptr(ptr: ptr::NonNull<u8>, cap: Capacity) {
307    #[cold]
308    fn deallocate_with_capacity_on_heap(ptr: ptr::NonNull<u8>) {
309        // re-adjust the pointer to include the capacity that's on the heap
310        let adj_ptr = ptr.as_ptr().wrapping_sub(mem::size_of::<usize>());
311        // read the capacity from the heap so we know how much to deallocate
312        let mut buf = [0u8; mem::size_of::<usize>()];
313        // SAFETY: `src` and `dst` don't overlap, and are valid for usize number of bytes
314        unsafe {
315            ptr::copy_nonoverlapping(adj_ptr, buf.as_mut_ptr(), mem::size_of::<usize>());
316        }
317        let capacity = usize::from_ne_bytes(buf);
318        // SAFETY: We know the pointer is not null since we got it as a NonNull
319        let ptr = unsafe { ptr::NonNull::new_unchecked(adj_ptr) };
320        // SAFETY: We checked above that our capacity is on the heap, and we readjusted the
321        // pointer to reference the capacity
322        unsafe { heap_capacity::dealloc(ptr, capacity) }
323    }
324
325    if cap.is_heap() {
326        deallocate_with_capacity_on_heap(ptr);
327    } else {
328        // SAFETY: Our capacity is always inline on 64-bit archs
329        unsafe { inline_capacity::dealloc(ptr, cap.as_usize()) }
330    }
331}

333mod heap_capacity {
334    use core::ptr;
335    use std::alloc;
336
337    use super::StrBuffer;
338
339    #[inline]
340    pub fn alloc(capacity: usize) -> ptr::NonNull<u8> {
341        let layout = layout(capacity);
342        debug_assert!(layout.size() > 0);
343
344        // SAFETY: `alloc(...)` has undefined behavior if the layout is zero-sized. We know the
345        // layout can't be zero-sized though because we're always at least allocating one `usize`
346        let raw_ptr = unsafe { alloc::alloc(layout) };
347
348        // Check to make sure our pointer is non-null, some allocators return null pointers instead
349        // of panicking
350        match ptr::NonNull::new(raw_ptr) {
351            Some(ptr) => ptr,
352            None => alloc::handle_alloc_error(layout),
353        }
354    }
355
356    /// Deallocates a pointer which references a `HeapBuffer` whose capacity is on the heap
357    ///
358    /// # Saftey
359    /// * `ptr` must point to the start of a `HeapBuffer` whose capacity is on the heap. i.e. we
360    ///   must have `ptr -> [cap<usize> ; string<bytes>]`
361    pub unsafe fn dealloc(ptr: ptr::NonNull<u8>, capacity: usize) {
362        let layout = layout(capacity);
363        alloc::dealloc(ptr.as_ptr(), layout);
364    }
365
366    #[repr(C)]
367    struct HeapBufferInnerHeapCapacity {
368        capacity: usize,
369        buffer: StrBuffer,
370    }
371
372    #[inline(always)]
373    pub fn layout(capacity: usize) -> alloc::Layout {
374        let buffer_layout = alloc::Layout::array::<u8>(capacity).expect("valid capacity");
375        alloc::Layout::new::<HeapBufferInnerHeapCapacity>()
376            .extend(buffer_layout)
377            .expect("valid layout")
378            .0
379            .pad_to_align()
380    }
381}

389    /// # SAFETY:
390    /// * `capacity` must be > 0
391    #[inline]
392    pub unsafe fn alloc(capacity: usize) -> ptr::NonNull<u8> {
393        let layout = layout(capacity);
394        debug_assert!(layout.size() > 0);
395
396        // SAFETY: `alloc(...)` has undefined behavior if the layout is zero-sized. We specify that
397        // `capacity` must be > 0 as a constraint to uphold the safety of this method. If capacity
398        // is greater than 0, then our layout will be non-zero-sized.
399        let raw_ptr = alloc::alloc(layout);
400
401        // Check to make sure our pointer is non-null, some allocators return null pointers instead
402        // of panicking
403        match ptr::NonNull::new(raw_ptr) {
404            Some(ptr) => ptr,
405            None => alloc::handle_alloc_error(layout),
406        }
407    }

413    pub unsafe fn dealloc(ptr: ptr::NonNull<u8>, capacity: usize) {
414        let layout = layout(capacity);
415        alloc::dealloc(ptr.as_ptr(), layout);
