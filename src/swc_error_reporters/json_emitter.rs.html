42impl Emitter for JsonEmitter {
43    fn emit(&mut self, db: &DiagnosticBuilder) {
44        let d = &**db;
45
46        let children = d
47            .children
48            .iter()
49            .map(|d| todo!("json subdiagnostic: {d:?}"))
50            .collect::<Vec<_>>();
51
52        let error_code = match &d.code {
53            Some(DiagnosticId::Error(s)) => Some(&**s),
54            Some(DiagnosticId::Lint(s)) => Some(&**s),
55            None => None,
56        };
57
58        let loc = d
59            .span
60            .primary_span()
61            .and_then(|span| self.cm.try_lookup_char_pos(span.lo()).ok());
62
63        let snippet = d
64            .span
65            .primary_span()
66            .and_then(|span| self.cm.span_to_snippet(span).ok());
67
68        let filename = if self.config.skip_filename {
69            None
70        } else {
71            loc.as_ref().map(|loc| loc.file.name.to_string())
72        };
73
74        let error = JsonDiagnostic {
75            code: error_code,
76            message: &d.message[0].0,
77            snippet: snippet.as_deref(),
78            filename: filename.as_deref(),
79            line: loc.as_ref().map(|loc| loc.line),
80            column: loc.as_ref().map(|loc| loc.col_display),
81            children,
82        };
83
84        let result = serde_json::to_string(&error).unwrap();
85
86        self.wr.write_str(&result).unwrap();
87        writeln!(self.wr).unwrap();
88
89        self.diagnostics.push(result);
90    }
91
92    fn take_diagnostics(&mut self) -> Vec<String> {
93        std::mem::take(&mut self.diagnostics)
94    }
95}
