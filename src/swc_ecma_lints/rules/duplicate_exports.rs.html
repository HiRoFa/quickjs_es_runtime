11pub fn duplicate_exports() -> Box<dyn Rule> {
12    visitor_rule(DuplicateExports::default())
13}
14
15#[derive(Debug, Default)]
16struct DuplicateExports {
17    exports: FxHashMap<Atom, Span>,
18    export_assign: Option<Span>,
19}
20
21impl DuplicateExports {
22    /// Add an export.
23    fn add(&mut self, id: &Ident) {
24        match self.exports.entry(id.sym.clone()) {
25            Entry::Occupied(mut prev) => {
26                let name = &id.sym;
27
28                HANDLER.with(|handler| {
29                    handler
30                        .struct_span_err(
31                            id.span,
32                            &format!("the name `{}` is exported multiple times", name),
33                        )
34                        .span_label(*prev.get(), "previous exported here")
35                        .span_label(id.span, "exported more than once")
36                        .note("Exported identifiers must be unique")
37                        .emit();
38                });
39
40                // Next span.
41                *prev.get_mut() = id.span;
42            }
43            Entry::Vacant(e) => {
44                e.insert(id.span);
45            }
46        }
47
48        self.check_no_coexist();
49    }
50
51    fn add_export_assign(&mut self, span: Span) {
52        if let Some(prev_span) = self.export_assign {
53            // TS2300
54            HANDLER.with(|handler| {
55                handler
56                    .struct_span_err(span, "multiple `export =` found")
57                    .span_label(prev_span, "previous `export =` declared here")
58                    .emit()
59            });
60        }
61
62        self.export_assign = Some(span);
63
64        self.check_no_coexist();
65    }
66
67    fn check_no_coexist(&self) {
68        if let Some(span) = self.export_assign {
69            if !self.exports.is_empty() {
70                // TS2309
71                HANDLER.with(|handler| {
72                    handler
73                        .struct_span_err(span, r#"An export assignment cannot be used in a module with other exported elements."#)
74                        .emit()
75                });
76            }
77        }
78    }
79}
80
81impl Visit for DuplicateExports {
82    noop_visit_type!();
83
84    fn visit_export_default_decl(&mut self, d: &ExportDefaultDecl) {
85        if match &d.decl {
86            DefaultDecl::Fn(FnExpr { function: f, .. }) if f.body.is_none() => true,
87            DefaultDecl::TsInterfaceDecl(..) => true,
88            _ => false,
89        } {
90            return;
91        }
92
93        d.visit_children_with(self);
94
95        self.add(&Ident::new_no_ctxt("default".into(), d.span));
96    }
97
98    fn visit_export_default_expr(&mut self, d: &ExportDefaultExpr) {
99        d.visit_children_with(self);
100
101        match &*d.expr {
102            Expr::Fn(FnExpr { function: f, .. }) if f.body.is_none() => return,
103            _ => {}
104        }
105
106        self.add(&Ident::new_no_ctxt("default".into(), d.span));
107    }
108
109    fn visit_export_default_specifier(&mut self, s: &ExportDefaultSpecifier) {
110        self.add(&s.exported);
111    }
112
113    fn visit_export_named_specifier(&mut self, s: &ExportNamedSpecifier) {
114        let exported = match &s.exported {
115            Some(ModuleExportName::Ident(ident)) => Some(ident),
116            Some(ModuleExportName::Str(..)) => return,
117            _ => None,
118        };
119        let orig = match &s.orig {
120            ModuleExportName::Ident(ident) => ident,
121            ModuleExportName::Str(..) => return,
122        };
123        self.add(exported.as_ref().unwrap_or(&orig));
124    }
125
126    fn visit_export_namespace_specifier(&mut self, s: &ExportNamespaceSpecifier) {
127        match &s.name {
128            ModuleExportName::Ident(name) => self.add(name),
129            ModuleExportName::Str(..) => {}
130        };
131    }
132
133    /// Noop. Just to ensure that the visitor do not recurse into stmt.
134    fn visit_stmt(&mut self, _: &Stmt) {}
135
136    fn visit_ts_export_assignment(&mut self, n: &TsExportAssignment) {
137        self.add_export_assign(n.span);
138    }
139
140    fn visit_ts_import_equals_decl(&mut self, n: &TsImportEqualsDecl) {
141        if n.is_export && !n.is_type_only {
142            self.add(&n.id)
143        }
144    }
145
146    fn visit_ts_module_decl(&mut self, d: &TsModuleDecl) {
147        if !d.declare {
148            let old = mem::take(self);
149            d.visit_children_with(self);
150            *self = old;
151        }
152    }
153}
