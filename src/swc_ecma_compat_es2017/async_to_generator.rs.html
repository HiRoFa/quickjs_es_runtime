18/// `@babel/plugin-transform-async-to-generator`
19///
20/// ## In
21///
22/// ```js
23/// async function foo() {
24///   await bar();
25/// }
26/// ```
27///
28/// ## Out
29///
30/// ```js
31/// var _async_to_generator = function (fn) {
32///   ...
33/// };
34/// var foo = _async_to_generator(function* () {
35///   yield bar();
36/// });
37/// ```
38pub fn async_to_generator(c: Config, unresolved_mark: Mark) -> impl Pass {
39    visit_mut_pass(AsyncToGenerator {
40        c,
41        in_subclass: false,
42        unresolved_ctxt: SyntaxContext::empty().apply_mark(unresolved_mark),
43    })
44}

95#[swc_trace]
96impl AsyncToGenerator {
97    fn visit_mut_stmt_like<T>(&mut self, stmts: &mut Vec<T>)
98    where
99        T: StmtLike + VisitMutWith<Actual>,
100        Vec<T>: VisitMutWith<Self>,
101    {
102        let mut stmts_updated = Vec::with_capacity(stmts.len());
103
104        for mut stmt in stmts.drain(..) {
105            let hoister = FnEnvHoister::new(self.unresolved_ctxt);
106
107            let mut actual = Actual {
108                c: self.c,
109                in_subclass: self.in_subclass,
110                hoister,
111                unresolved_ctxt: self.unresolved_ctxt,
112                extra_stmts: Vec::new(),
113            };
114
115            stmt.visit_mut_with(&mut actual);
116            stmts_updated.extend(actual.hoister.to_stmt().into_iter().map(T::from));
117            stmts_updated.push(stmt);
118            stmts_updated.extend(actual.extra_stmts.into_iter().map(T::from));
119        }
120
121        *stmts = stmts_updated;
122        stmts.visit_mut_children_with(self);
123    }
124}

142        c.params.visit_mut_children_with(self);
143
144        if let Some(BlockStmt { span: _, stmts, .. }) = &mut c.body {

159                }
160            } else {
161                let decl = mem::replace(&mut self.hoister, old_rep).to_stmt();
162
163                if let Some(decl) = decl {
164                    prepend_stmt(stmts, decl)
165                }

171        if m.function.body.is_none() {
172            return;
173        }
174
175        m.visit_mut_children_with(self);
176

178            return;

288            return;
289        }

392    fn visit_mut_var_declarator(&mut self, var: &mut VarDeclarator) {
393        if let VarDeclarator {
394            name: Pat::Ident(id),
395            init: Some(init),
396            ..
397        } = var
398        {
399            match init.as_ref() {
400                Expr::Fn(FnExpr {
401                    ident: None,
402                    ref function,
403                }) if function.is_async || function.is_generator => {
404                    self.visit_mut_expr_with_binding(init, Some(Ident::from(&*id)), false);
405                    return;
406                }
407
408                Expr::Arrow(arrow_expr) if arrow_expr.is_async || arrow_expr.is_generator => {
409                    self.visit_mut_expr_with_binding(init, Some(Ident::from(&*id)), false);

421#[swc_trace]
422impl Actual {
423    fn visit_mut_expr_with_binding(
424        &mut self,
425        expr: &mut Expr,
426        binding_ident: Option<Ident>,
427        in_iife: bool,
428    ) {
429        expr.visit_mut_children_with(self);
430
431        match expr {
432            Expr::Arrow(arrow_expr @ ArrowExpr { is_async: true, .. }) => {
433                arrow_expr.visit_mut_with(&mut self.hoister);
434
435                let mut wrapper = FunctionWrapper::from(arrow_expr.take());
436                wrapper.ignore_function_name = self.c.ignore_function_name;
437                wrapper.ignore_function_length = self.c.ignore_function_length;
438                wrapper.binding_ident = binding_ident;
439
440                let fn_expr = wrapper.function.expect_fn_expr();
441
442                wrapper.function = make_fn_ref(fn_expr);
443                *expr = wrapper.into();
444                if !in_iife {
445                    expr.set_span(PURE_SP);
446                }
447            }
448
449            Expr::Fn(fn_expr) if fn_expr.function.is_async => {
450                let mut wrapper = FunctionWrapper::from(fn_expr.take());
451                wrapper.ignore_function_name = self.c.ignore_function_name;
452                wrapper.ignore_function_length = self.c.ignore_function_length;
453                wrapper.binding_ident = binding_ident;
454
455                let fn_expr = wrapper.function.expect_fn_expr();
456
457                wrapper.function = make_fn_ref(fn_expr);
458
459                *expr = wrapper.into();
460                if !in_iife {
461                    expr.set_span(PURE_SP);
462                }
463            }
464
465            _ => {}
466        }
467    }

470/// Creates
471///
472/// `_async_to_generator(function*() {})` from `async function() {}`;
473#[tracing::instrument(level = "info", skip_all)]
474fn make_fn_ref(mut expr: FnExpr) -> Expr {
475    {
476        let param_ids: Vec<Id> = find_pat_ids(&expr.function.params);
477        let mapping = param_ids
478            .into_iter()
479            .map(|id| (id, SyntaxContext::empty().apply_mark(Mark::new())))
480            .collect();
481
482        expr.function.visit_mut_with(&mut Remapper::new(&mapping));
483    }
484
485    expr.function.body.visit_mut_with(&mut AsyncFnBodyHandler {
486        is_async_generator: expr.function.is_generator,
487    });
488
489    assert!(expr.function.is_async);
490    expr.function.is_async = false;
491
492    let helper = if expr.function.is_generator {
493        helper!(wrap_async_generator)
494    } else {
495        helper!(async_to_generator)
496    };
497
498    expr.function.is_generator = true;
499
500    let span = expr.span();
501
502    CallExpr {
503        span,
504        callee: helper,
505        args: vec![expr.as_arg()],
506        ..Default::default()
507    }
508    .into()
509}

511struct AsyncFnBodyHandler {
512    is_async_generator: bool,
513}
514
515macro_rules! noop {
516    ($name:ident, $T:path) => {
517        /// Don't recurse into function.
518        fn $name(&mut self, _f: &mut $T) {}
519    };
520}

523impl VisitMut for AsyncFnBodyHandler {
524    noop_visit_mut_type!(fail);
525
526    noop!(visit_mut_fn_expr, FnExpr);
527
528    noop!(visit_mut_constructor, Constructor);
529
530    noop!(visit_mut_arrow_expr, ArrowExpr);
531
532    noop!(visit_mut_fn_decl, FnDecl);
533
534    fn visit_mut_expr(&mut self, expr: &mut Expr) {
535        expr.visit_mut_children_with(self);
536
537        match expr {
538            Expr::Yield(YieldExpr {
539                span,
540                arg: Some(arg),
541                delegate: true,
542            }) => {
543                let callee = helper!(async_generator_delegate);
544                let arg = CallExpr {
545                    span: *span,
546                    callee,
547                    args: vec![
548                        CallExpr {
549                            span: DUMMY_SP,
550                            callee: helper!(async_iterator),
551                            args: vec![arg.take().as_arg()],
552                            ..Default::default()
553                        }
554                        .as_arg(),
555                        helper_expr!(await_async_generator).as_arg(),
556                    ],
557                    ..Default::default()
558                }
559                .into();
560                *expr = YieldExpr {
561                    span: *span,
562                    delegate: true,
563                    arg: Some(arg),
564                }
565                .into()
566            }
567
568            Expr::Await(AwaitExpr { span, arg }) => {
569                if self.is_async_generator {
570                    let callee = helper!(await_async_generator);
571                    let arg = CallExpr {
572                        span: *span,
573                        callee,
574                        args: vec![arg.take().as_arg()],
575                        ..Default::default()
576                    }
577                    .into();
578                    *expr = YieldExpr {
579                        span: *span,
580                        delegate: false,
581                        arg: Some(arg),
582                    }
583                    .into()
584                } else {
585                    *expr = YieldExpr {
586                        span: *span,
587                        delegate: false,
588                        arg: Some(arg.take()),
589                    }
590                    .into()
591                }
592            }
593            _ => {}
594        }
595    }

598        s.visit_mut_children_with(self);
599
600        handle_await_for(s, self.is_async_generator);
601    }

636#[tracing::instrument(level = "info", skip_all)]
637fn handle_await_for(stmt: &mut Stmt, is_async_generator: bool) {
638    let s = match stmt {
639        Stmt::ForOf(s @ ForOfStmt { is_await: true, .. }) => s.take(),
640        _ => return,
641    };
642
643    let value = private_ident!("_value");
644    let iterator = private_ident!("_iterator");
645    let iterator_error = private_ident!("_iteratorError");
646    let step = private_ident!("_step");
647    let did_iteration_error = private_ident!("_didIteratorError");
648    let iterator_abrupt_completion = private_ident!("_iteratorAbruptCompletion");
649    let err_param = private_ident!("err");
650
651    let try_body = {
652        let body_span = s.body.span();
653        let orig_body = match *s.body {
654            Stmt::Block(s) => s.stmts,
655            _ => vec![*s.body],
656        };
657
658        let mut for_loop_body = Vec::new();
659        {
660            // let value = _step.value;
661            let value_var = VarDeclarator {
662                span: DUMMY_SP,
663                name: value.clone().into(),
664                init: Some(step.clone().make_member(quote_ident!("value")).into()),
665                definite: false,
666            };
667            for_loop_body.push(
668                VarDecl {
669                    span: DUMMY_SP,
670                    kind: VarDeclKind::Let,
671                    declare: false,
672                    decls: vec![value_var],
673                    ..Default::default()
674                }
675                .into(),
676            );
677        }
678
679        match s.left {
680            ForHead::VarDecl(v) => {
681                let var = v.decls.into_iter().next().unwrap();
682                let var_decl = VarDeclarator {
683                    span: DUMMY_SP,
684                    name: var.name,
685                    init: Some(value.into()),
686                    definite: false,
687                };
688                for_loop_body.push(
689                    VarDecl {
690                        span: DUMMY_SP,
691                        kind: VarDeclKind::Const,
692                        declare: false,
693                        decls: vec![var_decl],
694                        ..Default::default()
695                    }
696                    .into(),
697                );
698            }
699            ForHead::Pat(p) => {
700                for_loop_body.push(
701                    ExprStmt {
702                        span: DUMMY_SP,
703                        expr: AssignExpr {
704                            span: DUMMY_SP,
705                            op: op!("="),
706                            left: p.try_into().unwrap(),
707                            right: Box::new(value.into()),
708                        }
709                        .into(),
710                    }
711                    .into(),
712                );
713            }
714
715            ForHead::UsingDecl(..) => {
716                unreachable!("using declaration must be removed by previous pass")
717            }
718        }
719
720        for_loop_body.extend(orig_body);
721
722        let for_loop_body = BlockStmt {
723            span: body_span,
724            stmts: for_loop_body,
725            ..Default::default()
726        };
727
728        let mut init_var_decls = Vec::new();
729        // _iterator = _async_iterator(lol())
730        init_var_decls.push(VarDeclarator {
731            span: DUMMY_SP,
732            name: iterator.clone().into(),
733            init: {
734                let callee = helper!(async_iterator);
735
736                Some(
737                    CallExpr {
738                        span: DUMMY_SP,
739                        callee,
740                        args: vec![s.right.as_arg()],
741                        ..Default::default()
742                    }
743                    .into(),
744                )
745            },
746            definite: false,
747        });
748        init_var_decls.push(VarDeclarator {
749            span: DUMMY_SP,
750            name: step.clone().into(),
751            init: None,
752            definite: false,
753        });
754
755        let for_stmt = ForStmt {
756            span: s.span,
757            // var _iterator = _async_iterator(lol()), _step;
758            init: Some(
759                VarDecl {
760                    span: DUMMY_SP,
761                    kind: VarDeclKind::Var,
762                    declare: false,
763                    decls: init_var_decls,
764                    ..Default::default()
765                }
766                .into(),
767            ),
768            // _iteratorAbruptCompletion = !(_step = yield _iterator.next()).done
769            test: {
770                let iter_next = iterator.clone().make_member(quote_ident!("next"));
771                let iter_next = CallExpr {
772                    span: DUMMY_SP,
773                    callee: iter_next.as_callee(),
774                    args: Default::default(),
775                    ..Default::default()
776                };
777
778                let yield_arg = if is_async_generator {
779                    CallExpr {
780                        span: DUMMY_SP,
781                        callee: helper!(await_async_generator),
782                        args: vec![iter_next.as_arg()],
783                        ..Default::default()
784                    }
785                    .into()
786                } else {
787                    iter_next.into()
788                };
789
790                let assign_to_step: Expr = AssignExpr {
791                    span: DUMMY_SP,
792                    op: op!("="),
793                    left: step.into(),
794                    right: YieldExpr {
795                        span: DUMMY_SP,
796                        arg: Some(yield_arg),
797                        delegate: false,
798                    }
799                    .into(),
800                }
801                .into();
802
803                let right = UnaryExpr {
804                    span: DUMMY_SP,
805                    op: op!("!"),
806                    arg: assign_to_step.make_member(quote_ident!("done")).into(),
807                }
808                .into();
809
810                let left = iterator_abrupt_completion.clone().into();
811
812                Some(
813                    AssignExpr {
814                        span: DUMMY_SP,
815                        op: op!("="),
816                        left,
817                        right,
818                    }
819                    .into(),
820                )
821            },
822            // _iteratorNormalCompletion = true
823            update: Some(
824                AssignExpr {
825                    span: DUMMY_SP,
826                    op: op!("="),
827                    left: iterator_abrupt_completion.clone().into(),
828                    right: false.into(),
829                }
830                .into(),
831            ),
832            body: Box::new(Stmt::Block(for_loop_body)),
833        }
834        .into();
835
836        BlockStmt {
837            span: body_span,
838            stmts: vec![for_stmt],
839            ..Default::default()
840        }
841    };
842
843    let catch_clause = {
844        // _didIteratorError = true;
845        let mark_as_errorred = ExprStmt {
846            span: DUMMY_SP,
847            expr: AssignExpr {
848                span: DUMMY_SP,
849                op: op!("="),
850                left: did_iteration_error.clone().into(),
851                right: true.into(),
852            }
853            .into(),
854        }
855        .into();
856        // _iteratorError = err;
857        let store_error = ExprStmt {
858            span: DUMMY_SP,
859            expr: AssignExpr {
860                span: DUMMY_SP,
861                op: op!("="),
862                left: iterator_error.clone().into(),
863                right: Box::new(err_param.clone().into()),
864            }
865            .into(),
866        }
867        .into();
868
869        CatchClause {
870            span: DUMMY_SP,
871            param: Some(err_param.into()),
872            body: BlockStmt {
873                stmts: vec![mark_as_errorred, store_error],
874                ..Default::default()
875            },
876        }
877    };
878
879    let finally_block = {
880        let throw_iterator_error = ThrowStmt {
881            span: DUMMY_SP,
882            arg: iterator_error.clone().into(),
883        }
884        .into();
885        let throw_iterator_error = IfStmt {
886            span: DUMMY_SP,
887            test: did_iteration_error.clone().into(),
888            cons: Box::new(Stmt::Block(BlockStmt {
889                span: DUMMY_SP,
890                stmts: vec![throw_iterator_error],
891                ..Default::default()
892            })),
893            alt: None,
894        }
895        .into();
896
897        let iterator_return: Expr = CallExpr {
898            span: DUMMY_SP,
899            callee: iterator
900                .clone()
901                .make_member(quote_ident!("return"))
902                .as_callee(),
903            args: Vec::new(),
904            ..Default::default()
905        }
906        .into();
907
908        // yield _iterator.return();
909        // or
910        // yield _awaitAsyncGenerator(_iterator.return());
911        let yield_stmt = ExprStmt {
912            span: DUMMY_SP,
913            expr: YieldExpr {
914                span: DUMMY_SP,
915                delegate: false,
916                arg: Some(if is_async_generator {
917                    CallExpr {
918                        span: DUMMY_SP,
919                        callee: helper!(await_async_generator),
920                        args: vec![iterator_return.as_arg()],
921                        ..Default::default()
922                    }
923                    .into()
924                } else {
925                    iterator_return.into()
926                }),
927            }
928            .into(),
929        }
930        .into();
931
932        let conditional_yield = IfStmt {
933            span: DUMMY_SP,
934            // _iteratorAbruptCompletion && _iterator.return != null
935            test: BinExpr {
936                span: DUMMY_SP,
937                op: op!("&&"),
938                // _iteratorAbruptCompletion
939                left: Box::new(iterator_abrupt_completion.clone().into()),
940                // _iterator.return != null
941                right: Box::new(
942                    BinExpr {
943                        span: DUMMY_SP,
944                        op: op!("!="),
945                        left: iterator.make_member(quote_ident!("return")).into(),
946                        right: Null { span: DUMMY_SP }.into(),
947                    }
948                    .into(),
949                ),
950            }
951            .into(),
952            cons: Box::new(Stmt::Block(BlockStmt {
953                stmts: vec![yield_stmt],
954                ..Default::default()
955            })),
956            alt: None,
957        }
958        .into();
959        let body = BlockStmt {
960            stmts: vec![conditional_yield],
961            ..Default::default()
962        };
963
964        let inner_try = TryStmt {
965            span: DUMMY_SP,
966            block: body,
967            handler: None,
968            finalizer: Some(BlockStmt {
969                stmts: vec![throw_iterator_error],
970                ..Default::default()
971            }),
972        }
973        .into();
974        BlockStmt {
975            stmts: vec![inner_try],
976            ..Default::default()
977        }
978    };
979
980    let try_stmt = TryStmt {
981        span: s.span,
982        block: try_body,
983        handler: Some(catch_clause),
984        finalizer: Some(finally_block),
985    };
986
987    let stmts = vec![
988        VarDecl {
989            kind: VarDeclKind::Var,
990            decls: vec![
991                // var _iteratorAbruptCompletion = false;
992                VarDeclarator {
993                    span: DUMMY_SP,
994                    name: iterator_abrupt_completion.into(),
995                    init: Some(false.into()),
996                    definite: false,
997                },
998                // var _didIteratorError = false;
999                VarDeclarator {
1000                    span: DUMMY_SP,
1001                    name: did_iteration_error.into(),
1002                    init: Some(false.into()),
1003                    definite: false,
1004                },
1005                // var _iteratorError;
1006                VarDeclarator {
1007                    span: DUMMY_SP,
1008                    name: iterator_error.into(),
1009                    init: None,
1010                    definite: false,
1011                },
1012            ],
1013            ..Default::default()
1014        }
1015        .into(),
1016        try_stmt.into(),
1017    ];
1018
1019    *stmt = BlockStmt {
1020        span: s.span,
1021        stmts,
1022        ..Default::default()
1023    }
1024    .into()
1025}
