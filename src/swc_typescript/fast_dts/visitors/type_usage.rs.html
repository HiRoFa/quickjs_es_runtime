20bitflags! {
21    #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
22    pub struct SymbolFlags: u8 {
23        const Value = 1 << 0;
24        const Type = 1 << 1;
25    }
26}

28#[derive(Debug, Clone, PartialEq, Eq, Hash)]
29struct Symbol {
30    id: Id,
31    kind: SymbolFlags,
32}
33
34impl Symbol {
35    fn new(id: Id, kind: SymbolFlags) -> Self {
36        Symbol { id, kind }
37    }

40bitflags! {
41    #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
42    pub struct Context: u8 {
43        const InTypeQuery = 1 << 0;
44    }

47pub struct TypeUsageAnalyzer<'a> {
48    /// The graph may contain multiple symbol nodes with the same `Id` and
49    /// different `SymbolFlags`. This is ok because they will be merged into the
50    /// final result `UsedRefs`
51    graph: DiGraph<Symbol, ()>,
52    nodes: FxHashMap<Symbol, NodeIndex>,
53    /// Global scope + nested ts module block scope
54    scope_entries: Vec<NodeIndex>,
55    /// Dts will only consider referred nodes and ignore binding nodes
56    references: FxHashSet<NodeIndex>,
57    /// Current source node that all the nested idents should depend on
58    source: Option<NodeIndex>,
59    /// Used for stripping those with @internal
60    internal_annotations: Option<&'a FxHashSet<BytePos>>,
61    /// Used for analyzing usages without parameter drilling
62    ctx: Context,
63}
64
65#[derive(Debug, Clone, Default)]
66pub struct UsedRefs(FxHashMap<Id, SymbolFlags>);
67
68impl UsedRefs {
69    pub fn add_usage(&mut self, id: Id, kind: SymbolFlags) {
70        match self.0.entry(id) {
71            Entry::Occupied(mut occupied_entry) => {
72                let value = occupied_entry.get_mut();
73                *value = value.union(kind);
74            }
75            Entry::Vacant(vacant_entry) => {
76                vacant_entry.insert(kind);
77            }
78        }
79    }
80
81    pub fn used_as_type(&self, id: &Id) -> bool {
82        self.0
83            .get(id)
84            .is_some_and(|ref_type| ref_type.contains(SymbolFlags::Type))
85    }
86
87    pub fn used_as_value(&self, id: &Id) -> bool {
88        self.0
89            .get(id)
90            .is_some_and(|ref_type| ref_type.contains(SymbolFlags::Value))
91    }
92
93    pub fn used(&self, id: &Id) -> bool {
94        self.0.get(id).is_some_and(|ref_type| !ref_type.is_empty())
95    }
96
97    pub fn extend(&mut self, other: Self) {
98        for (id, kind) in other.0 {
99            self.add_usage(id, kind);
100        }
101    }
102}

104impl TypeUsageAnalyzer<'_> {
105    pub fn analyze(
106        module_items: &Vec<ModuleItem>,
107        internal_annotations: Option<&FxHashSet<BytePos>>,
108    ) -> UsedRefs {
109        // Create a fake entry reprensting global scope
110        let mut graph = Graph::default();
111        let entry = graph.add_node(Symbol::new(
112            ("".into(), SyntaxContext::empty()),
113            SymbolFlags::empty(),
114        ));
115
116        let mut analyzer = TypeUsageAnalyzer {
117            graph,
118            nodes: FxHashMap::default(),
119            scope_entries: vec![entry],
120            references: FxHashSet::default(),
121            source: None,
122            internal_annotations,
123            ctx: Context::empty(),
124        };
125        module_items.visit_with(&mut analyzer);
126
127        // Reachability: used_refs = all references - unreachable references
128        let mut used_refs = UsedRefs::default();
129        let mut bfs = Bfs::new(&analyzer.graph, entry);
130        bfs.next(&analyzer.graph);
131        while let Some(node_id) = bfs.next(&analyzer.graph) {
132            if analyzer.references.contains(&node_id) {
133                let symbol = analyzer
134                    .graph
135                    .node_weight(node_id)
136                    .expect("unexpected invalid node id");
137                used_refs.add_usage(symbol.id.clone(), symbol.kind);
138            }
139        }
140        used_refs
141    }
142
143    /// All the nested nodes may be connected by the passed id nodes
144    fn with_source<F: FnMut(&mut TypeUsageAnalyzer)>(&mut self, id: Option<NodeIndex>, mut f: F) {
145        // If id is None, we use the nearest scope
146        let old_source = self
147            .source
148            .replace(id.unwrap_or(*self.scope_entries.last().expect("No scope")));
149        f(self);
150        self.source = old_source;
151    }
152
153    /// All the nested idents will depend on the usage of the passed ident
154    /// This is usually called when meeting a binding ident
155    fn with_source_ident<F: FnMut(&mut TypeUsageAnalyzer)>(&mut self, symbol: Symbol, f: F) {
156        self.add_edge(symbol.clone(), false);
157        let id = *self
158            .nodes
159            .entry(symbol.clone())
160            .or_insert_with(|| self.graph.add_node(symbol));
161        self.with_source(Some(id), f);
162    }
163
164    /// Add a dependency edge from current source to the reference
165    /// If `is_ref` is false, the reference is a binding ident such as a
166    /// variable declaration, which means it is not a usage.
167    fn add_edge(&mut self, reference: Symbol, is_ref: bool) {
168        if let Some(source) = self.source {
169            let target_id = *self
170                .nodes
171                .entry(reference.clone())
172                .or_insert_with(|| self.graph.add_node(reference));
173            self.graph.add_edge(source, target_id, ());
174            if is_ref {
175                self.references.insert(target_id);
176            }
177        }
178    }
179
180    fn has_internal_annotation(&self, pos: BytePos) -> bool {
181        if let Some(internal_annotations) = &self.internal_annotations {
182            return internal_annotations.contains(&pos);
183        }
184        false
185    }
186}

188impl Visit for TypeUsageAnalyzer<'_> {
189    fn visit_ts_property_signature(&mut self, node: &TsPropertySignature) {
190        if let Some(ident) = node.key.get_root_ident() {
191            self.add_edge(Symbol::new(ident.to_id(), SymbolFlags::Value), true);
192        }
193        node.visit_children_with(self);
194    }
195
196    fn visit_ts_expr_with_type_args(&mut self, node: &TsExprWithTypeArgs) {
197        if let Some(ident) = node.expr.get_root_ident() {
198            self.add_edge(Symbol::new(ident.to_id(), SymbolFlags::Type), true);
199        }
200        node.visit_children_with(self);
201    }
202
203    fn visit_ts_type_query_expr(&mut self, node: &TsTypeQueryExpr) {
204        // In `typeof A.B.C`, A should be a value
205        if node.is_ts_entity_name() {
206            self.ctx.insert(Context::InTypeQuery);
207            node.visit_children_with(self);
208            self.ctx.remove(Context::InTypeQuery);
209        } else {
210            node.visit_children_with(self);
211        }
212    }
213
214    fn visit_ts_entity_name(&mut self, node: &TsEntityName) {
215        match node {
216            TsEntityName::TsQualifiedName(ts_qualified_name) => {
217                ts_qualified_name.left.visit_with(self);
218            }
219            TsEntityName::Ident(ident) => {
220                let flag = if self.ctx.contains(Context::InTypeQuery) {
221                    SymbolFlags::Value
222                } else {
223                    SymbolFlags::Type
224                };
225                self.add_edge(Symbol::new(ident.to_id(), flag), true);
226            }
227        };
228    }
229
230    fn visit_decl(&mut self, node: &Decl) {
231        match node {
232            Decl::Class(class_decl) => {
233                self.with_source_ident(
234                    Symbol::new(class_decl.ident.to_id(), SymbolFlags::all()),
235                    |this| class_decl.class.visit_with(this),
236                );
237            }
238            Decl::Fn(fn_decl) => {
239                self.with_source_ident(
240                    Symbol::new(fn_decl.ident.to_id(), SymbolFlags::Value),
241                    |this| fn_decl.function.visit_with(this),
242                );
243            }
244            Decl::Var(var_decl) => {
245                for decl in &var_decl.decls {
246                    // When the name of the decl is a binding ident, the usage of its type
247                    // annotation should also depend on the usage of the ident name
248                    if let Some(name) = decl.name.as_ident() {
249                        name.id.visit_with(self);
250                        self.with_source_ident(
251                            Symbol::new(name.id.to_id(), SymbolFlags::Value),
252                            |this| {
253                                name.type_ann.visit_with(this);
254                                decl.init.visit_with(this);
255                            },
256                        );
257                    } else {
258                        decl.name.visit_with(self);
259                    }
260                }
261            }
262            Decl::Using(using_decl) => {
263                for decl in &using_decl.decls {
264                    // When the name of the decl is a binding ident, the usage of its type
265                    // annotation should also depend on the usage of the ident name
266                    if let Some(name) = decl.name.as_ident() {
267                        name.id.visit_with(self);
268                        self.with_source_ident(
269                            Symbol::new(name.to_id(), SymbolFlags::Value),
270                            |this| {
271                                name.type_ann.visit_with(this);
272                                decl.init.visit_with(this);
273                            },
274                        );
275                    } else {
276                        decl.name.visit_with(self);
277                    }
278                }
279            }
280            Decl::TsInterface(ts_interface_decl) => {
281                self.with_source_ident(
282                    Symbol::new(ts_interface_decl.id.to_id(), SymbolFlags::Type),
283                    |this| {
284                        ts_interface_decl.body.visit_with(this);
285                        ts_interface_decl.extends.visit_with(this);
286                        ts_interface_decl.type_params.visit_with(this);
287                    },
288                );
289            }
290            Decl::TsTypeAlias(ts_type_alias_decl) => {
291                self.with_source_ident(
292                    Symbol::new(ts_type_alias_decl.id.to_id(), SymbolFlags::Type),
293                    |this| {
294                        ts_type_alias_decl.type_ann.visit_with(this);
295                        ts_type_alias_decl.type_params.visit_with(this);
296                    },
297                );
298            }
299            Decl::TsEnum(ts_enum_decl) => {
300                self.with_source_ident(
301                    Symbol::new(ts_enum_decl.id.to_id(), SymbolFlags::all()),
302                    |this| {
303                        ts_enum_decl.members.visit_with(this);
304                    },
305                );
306            }
307            Decl::TsModule(ts_module_decl) => {
308                if ts_module_decl.global || ts_module_decl.id.is_str() {
309                    // Here we enter global scope
310                    self.with_source(Some(self.scope_entries[0]), |this| {
311                        ts_module_decl.body.visit_with(this)
312                    });
313                } else if let Some(ident) = ts_module_decl.id.as_ident() {
314                    // Push a new scope and set current scope to None, which indicates that
315                    // non-exported elements in ts module block are unreachable
316                    let symbol = Symbol::new(ident.to_id(), SymbolFlags::Type);
317                    self.add_edge(symbol.clone(), false);
318                    let id = *self
319                        .nodes
320                        .entry(symbol.clone())
321                        .or_insert_with(|| self.graph.add_node(symbol));
322                    self.scope_entries.push(id);
323                    let old_source = self.source.take();
324                    ts_module_decl.body.visit_with(self);
325                    self.source = old_source;
326                    self.scope_entries.pop();
327                }
328            }
329        }
330    }
331
332    fn visit_export_decl(&mut self, node: &ExportDecl) {
333        self.with_source(self.source, |this| {
334            node.visit_children_with(this);
335        });
336    }
337
338    fn visit_named_export(&mut self, node: &NamedExport) {
339        self.with_source(self.source, |this| {
340            for specifier in &node.specifiers {
341                if let Some(name) = specifier.as_named() {
342                    if let ModuleExportName::Ident(ident) = &name.orig {
343                        // We should add egdes for all three possible namespaces because there's no
344                        // mechanism to merge SymbolFlags with same Ids
345                        this.add_edge(Symbol::new(ident.to_id(), SymbolFlags::Type), true);
346                        if !node.type_only {
347                            this.add_edge(Symbol::new(ident.to_id(), SymbolFlags::Value), true);
348                            this.add_edge(Symbol::new(ident.to_id(), SymbolFlags::all()), true);
349                        }
350                    }
351                }
352            }
353        });
354    }
355
356    fn visit_export_default_decl(&mut self, node: &ExportDefaultDecl) {
357        self.with_source(self.source, |this| {
358            node.visit_children_with(this);
359        });
360    }
361
362    fn visit_export_default_expr(&mut self, node: &ExportDefaultExpr) {
363        self.with_source(self.source, |this| {
364            if let Some(ident) = node.expr.as_ident() {
365                // We should add egdes for all three possible namespaces because there's no
366                // mechanism to merge SymbolFlags with same Ids
367                this.add_edge(Symbol::new(ident.to_id(), SymbolFlags::Type), true);
368                this.add_edge(Symbol::new(ident.to_id(), SymbolFlags::Value), true);
369                this.add_edge(Symbol::new(ident.to_id(), SymbolFlags::all()), true);
370            }
371            node.visit_children_with(this);
372        });
373    }
374
375    fn visit_ts_export_assignment(&mut self, node: &TsExportAssignment) {
376        self.with_source(self.source, |this| {
377            node.visit_children_with(this);
378        });
379    }
380
381    fn visit_arrow_expr(&mut self, node: &ArrowExpr) {
382        // Skip body
383        node.params.visit_with(self);
384        node.type_params.visit_with(self);
385        node.return_type.visit_with(self);
386    }
387
388    fn visit_function(&mut self, node: &Function) {
389        // Skip body
390        node.params.visit_with(self);
391        node.decorators.visit_with(self);
392        node.type_params.visit_with(self);
393        node.return_type.visit_with(self);
394    }
395
396    fn visit_class(&mut self, node: &Class) {
397        if let Some(super_class) = &node.super_class {
398            if let Some(ident) = super_class.get_root_ident() {
399                self.add_edge(Symbol::new(ident.to_id(), SymbolFlags::Value), true);
400            }
401        }
402        node.visit_children_with(self);
403    }
404
405    fn visit_class_member(&mut self, node: &ClassMember) {
406        if self.has_internal_annotation(node.span_lo()) {
407            return;
408        }
409
410        let is_private = match node {
411            ClassMember::Constructor(constructor) => constructor
412                .accessibility
413                .is_some_and(|accessibility| accessibility == Accessibility::Private),
414            ClassMember::Method(class_method) => class_method
415                .accessibility
416                .is_some_and(|accessibility| accessibility == Accessibility::Private),
417            ClassMember::PrivateMethod(_) => true,
418            ClassMember::ClassProp(class_prop) => class_prop
419                .accessibility
420                .is_some_and(|accessibility| accessibility == Accessibility::Private),
421            ClassMember::PrivateProp(_) => true,
422            ClassMember::TsIndexSignature(_) => false,
423            ClassMember::Empty(_) => false,
424            ClassMember::StaticBlock(_) => false,
425            ClassMember::AutoAccessor(auto_accessor) => {
426                auto_accessor
427                    .accessibility
428                    .is_some_and(|accessibility| accessibility == Accessibility::Private)
429                    || auto_accessor.key.is_private()
430            }
431        };
432
433        if is_private {
434            return;
435        }
436
437        node.visit_children_with(self);
438    }
439
440    fn visit_ts_type_element(&mut self, node: &TsTypeElement) {
441        if self.has_internal_annotation(node.span_lo()) {
442            return;
443        }
444        node.visit_children_with(self);
445    }
446
447    fn visit_module_items(&mut self, node: &[ModuleItem]) {
448        for item in node {
449            // Skip statements and internals
450            if item.as_stmt().is_some_and(|stmt| !stmt.is_decl())
451                || self.has_internal_annotation(item.span_lo())
452            {
453                continue;
454            }
455            item.visit_children_with(self);
456        }
457    }
