14impl FastDts {
15    pub(crate) fn infer_type_from_expr(&mut self, e: &Expr) -> Option<Box<TsType>> {
16        match e {
17            Expr::Ident(ident) if ident.sym.as_str() == "undefined" => {
18                Some(ts_keyword_type(TsKeywordTypeKind::TsUndefinedKeyword))
19            }
20            Expr::Lit(lit) => match lit {
21                Lit::Str(_) => Some(ts_keyword_type(TsKeywordTypeKind::TsStringKeyword)),
22                Lit::Bool(_) => Some(ts_keyword_type(TsKeywordTypeKind::TsBooleanKeyword)),
23                Lit::Num(_) => Some(ts_keyword_type(TsKeywordTypeKind::TsNumberKeyword)),
24                Lit::BigInt(_) => Some(ts_keyword_type(TsKeywordTypeKind::TsBigIntKeyword)),
25                Lit::Null(_) => Some(ts_keyword_type(TsKeywordTypeKind::TsNullKeyword)),
26                Lit::Regex(_) | Lit::JSXText(_) => None,
27            },
28            Expr::Tpl(_) => Some(ts_keyword_type(TsKeywordTypeKind::TsStringKeyword)),
29            Expr::Fn(fn_expr) => self.transform_fn_to_ts_type(
30                &fn_expr.function,
31                fn_expr.ident.as_ref().map(|ident| ident.span),
32            ),
33            Expr::Arrow(arrow_expr) => self.transform_arrow_expr_to_ts_type(arrow_expr),
34            Expr::Array(arr) => {
35                self.array_inferred(arr.span);
36                Some(ts_keyword_type(TsKeywordTypeKind::TsUnknownKeyword))
37            }
38            Expr::Object(obj) => self.transform_object_to_ts_type(obj, false),
39            Expr::Class(class) => {
40                self.inferred_type_of_class_expression(class.span());
41                Some(ts_keyword_type(TsKeywordTypeKind::TsUnknownKeyword))
42            }
43            Expr::Paren(expr) => self.infer_type_from_expr(&expr.expr),
44            Expr::TsNonNull(non_null) => self.infer_type_from_expr(&non_null.expr),
45            Expr::TsSatisfies(satisifies) => self.infer_type_from_expr(&satisifies.expr),
46            Expr::TsConstAssertion(assertion) => self.transform_expr_to_ts_type(&assertion.expr),
47            Expr::TsAs(ts_as) => Some(ts_as.type_ann.clone()),
48            Expr::TsTypeAssertion(type_assertion) => Some(type_assertion.type_ann.clone()),
49            Expr::Unary(unary) if Self::can_infer_unary_expr(unary) => {
50                self.infer_type_from_expr(&unary.arg)
51            }
52            _ => None,
53        }
54    }
55
56    pub(crate) fn infer_function_return_type(
57        &mut self,
58        function: &Function,
59    ) -> Option<Box<TsTypeAnn>> {
60        if function.return_type.is_some() {
61            return function.return_type.clone();
62        }
63
64        if function.is_async || function.is_generator {
65            return None;
66        }
67
68        function
69            .body
70            .as_ref()
71            .and_then(|body| ReturnTypeInferrer::infer(self, &body.stmts))
72            .map(type_ann)
73    }
74
75    pub(crate) fn infer_arrow_return_type(&mut self, arrow: &ArrowExpr) -> Option<Box<TsTypeAnn>> {
76        if arrow.return_type.is_some() {
77            return arrow.return_type.clone();
78        }
79
80        if arrow.is_async || arrow.is_generator {
81            return None;
82        }
83
84        match arrow.body.as_ref() {
85            BlockStmtOrExpr::BlockStmt(block_stmt) => {
86                ReturnTypeInferrer::infer(self, &block_stmt.stmts)
87            }
88            BlockStmtOrExpr::Expr(expr) => self.infer_type_from_expr(expr),
89        }
90        .map(type_ann)
91    }
92
93    pub(crate) fn need_to_infer_type_from_expression(expr: &Expr) -> bool {
94        match expr {
95            Expr::Lit(lit) => !(lit.is_str() || lit.is_num() || lit.is_big_int() || lit.is_bool()),
96            Expr::Tpl(tpl) => !tpl.exprs.is_empty(),
97            Expr::Unary(unary) => !Self::can_infer_unary_expr(unary),
98            _ => true,
99        }
100    }
101
102    pub(crate) fn can_infer_unary_expr(unary: &UnaryExpr) -> bool {
103        let is_arithmetic = matches!(unary.op, UnaryOp::Plus | UnaryOp::Minus);
104        let is_number_lit = match unary.arg.as_ref() {
105            Expr::Lit(lit) => lit.is_num() || lit.is_big_int(),
106            _ => false,
107        };
108        is_arithmetic && is_number_lit
109    }

112#[derive(Default)]
113pub struct ReturnTypeInferrer {
114    value_bindings: Vec<Ident>,
115    type_bindings: Vec<Ident>,
116    return_expr: Option<Option<Box<Expr>>>,
117    return_expr_count: u8,
118}
119
120impl ReturnTypeInferrer {
121    pub fn infer(transformer: &mut FastDts, stmts: &[Stmt]) -> Option<Box<TsType>> {
122        let mut visitor = ReturnTypeInferrer::default();
123        stmts.visit_children_with(&mut visitor);
124
125        let expr = visitor.return_expr??;
126        let Some(mut expr_type) = transformer.infer_type_from_expr(&expr) else {
127            return if expr.is_fn_expr() || expr.is_arrow() {
128                Some(ts_keyword_type(TsKeywordTypeKind::TsUnknownKeyword))
129            } else {
130                None
131            };
132        };
133
134        if let Some((ref_name, is_value)) = match expr_type.as_ref() {
135            TsType::TsTypeRef(type_ref) => Some((type_ref.type_name.as_ident()?, false)),
136            TsType::TsTypeQuery(type_query) => {
137                Some((type_query.expr_name.as_ts_entity_name()?.as_ident()?, true))
138            }
139            _ => None,
140        } {
141            let is_defined = if is_value {
142                visitor.value_bindings.contains(ref_name)
143            } else {
144                visitor.type_bindings.contains(ref_name)
145            };
146
147            if is_defined {
148                transformer.type_containing_private_name(ref_name.sym.as_str(), ref_name.span);
149            }
150        }
151
152        if visitor.return_expr_count > 1 {
153            if expr_type.is_ts_fn_or_constructor_type() {
154                expr_type = Box::new(TsType::TsParenthesizedType(TsParenthesizedType {
155                    span: DUMMY_SP,
156                    type_ann: expr_type,
157                }));
158            }
159
160            expr_type = Box::new(TsType::TsUnionOrIntersectionType(
161                TsUnionOrIntersectionType::TsUnionType(TsUnionType {
162                    span: DUMMY_SP,
163                    types: vec![
164                        expr_type,
165                        ts_keyword_type(TsKeywordTypeKind::TsUndefinedKeyword),
166                    ],
167                }),
168            ))
169        }
170
171        Some(expr_type)
172    }
173}

175impl Visit for ReturnTypeInferrer {
176    fn visit_arrow_expr(&mut self, _node: &ArrowExpr) {}
177
178    fn visit_function(&mut self, _node: &Function) {}
179
180    fn visit_class(&mut self, _node: &Class) {}
181
182    fn visit_binding_ident(&mut self, node: &BindingIdent) {
183        self.value_bindings.push(node.id.clone());
184    }
185
186    fn visit_ts_type_alias_decl(&mut self, node: &TsTypeAliasDecl) {
187        self.type_bindings.push(node.id.clone());
188    }
189
190    fn visit_return_stmt(&mut self, node: &ReturnStmt) {
191        self.return_expr_count += 1;
192        if self.return_expr_count > 1 {
193            if let Some(return_expr) = &self.return_expr {
194                if return_expr.is_some() {
195                    self.return_expr = None;
196                    return;
197                }
198            } else {
199                return;
200            }
201        }
202        self.return_expr = Some(node.arg.clone());
203    }
