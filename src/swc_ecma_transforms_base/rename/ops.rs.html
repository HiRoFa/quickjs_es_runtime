30    fn keep_class_name(&mut self, ident: &mut Ident, class: &mut Class) -> Option<ClassExpr> {
31        if !self.config.keep_class_names {
32            return None;
33        }
34
35        let mut orig_name = ident.clone();
36        orig_name.ctxt = SyntaxContext::empty();
37
38        {
39            // Remove span hygiene of the class.
40            let mut rename = FxHashMap::<Id, V>::default();
41
42            rename.insert(ident.to_id(), V::new_private(ident.sym.clone()));
43
44            let mut operator = Operator {
45                rename: &rename,
46                config: self.config.clone(),
47                extra: Default::default(),
48            };
49
50            class.visit_mut_with(&mut operator);
51        }
52
53        let _ = self.rename_ident(ident);
54        class.visit_mut_with(self);
55
56        let class_expr = ClassExpr {
57            ident: Some(orig_name),
58            class: Box::new(class.take()),
59        };
60
61        Some(class_expr)
62    }
63}

238    fn visit_mut_module_item(&mut self, item: &mut ModuleItem) {
239        let span = item.span();
240
241        macro_rules! export {
242            ($orig:expr, $ident:expr) => {
243                self.extra
244                    .push(ModuleItem::ModuleDecl(ModuleDecl::ExportNamed(
245                        NamedExport {
246                            span,
247                            specifiers: vec![ExportSpecifier::Named(ExportNamedSpecifier {
248                                span: DUMMY_SP,
249                                orig: $ident,
250                                exported: Some($orig),
251                                is_type_only: false,
252                            })],
253                            src: None,
254                            type_only: false,
255                            with: None,
256                        },
257                    )));
258            };
259        }
260
261        match item {
262            ModuleItem::ModuleDecl(ModuleDecl::ExportDecl(ExportDecl {
263                span,
264                decl:
265                    Decl::Class(ClassDecl {
266                        ident,
267                        class,
268                        declare,
269                    }),
270            })) => {
271                let mut ident = ident.take();
272                let mut class = class.take();
273
274                class.visit_mut_with(self);
275                let orig_ident = ident.clone();
276                match self.rename_ident(&mut ident) {
277                    Ok(..) => {
278                        *item = ClassDecl {
279                            ident: ident.clone(),
280                            class: class.take(),
281                            declare: *declare,
282                        }
283                        .into();
284                        export!(
285                            ModuleExportName::Ident(orig_ident),
286                            ModuleExportName::Ident(ident.take())
287                        );
288                    }
289                    Err(..) => {
290                        *item = ExportDecl {
291                            span: *span,
292                            decl: ClassDecl {
293                                ident: ident.take(),
294                                class: class.take(),
295                                declare: *declare,
296                            }
297                            .into(),
298                        }
299                        .into()
300                    }
301                }
302            }
303            ModuleItem::ModuleDecl(ModuleDecl::ExportDecl(ExportDecl {
304                span,
305                decl:
306                    Decl::Fn(FnDecl {
307                        ident,
308                        function,
309                        declare,
310                    }),
311            })) => {
312                let mut ident = ident.take();
313                let mut function = function.take();
314
315                function.visit_mut_with(self);
316                let orig_ident = ident.clone();
317                match self.rename_ident(&mut ident) {
318                    Ok(..) => {
319                        *item = FnDecl {
320                            ident: ident.clone(),
321                            function,
322                            declare: *declare,
323                        }
324                        .into();
325                        export!(
326                            ModuleExportName::Ident(orig_ident),
327                            ModuleExportName::Ident(ident)
328                        );
329                    }
330                    Err(..) => {
331                        *item = ExportDecl {
332                            span: *span,
333                            decl: FnDecl {
334                                ident,
335                                function,
336                                declare: *declare,
337                            }
338                            .into(),
339                        }
340                        .into()
341                    }
342                }
343            }
344            ModuleItem::ModuleDecl(ModuleDecl::ExportDecl(ExportDecl {
345                decl: Decl::Var(var),
346                ..
347            })) => {
348                let decls = var.decls.take();
349
350                let mut renamed: Vec<ExportSpecifier> = Vec::new();
351                let decls = decls.move_map(|mut decl| {
352                    decl.name.visit_mut_with(&mut VarFolder {
353                        orig: self,
354                        renamed: &mut renamed,
355                    });
356                    decl.init.visit_mut_with(self);
357                    decl
358                });
359
360                if renamed.is_empty() {
361                    *item = ExportDecl {
362                        span,
363                        decl: VarDecl {
364                            decls,
365                            ..*var.take()
366                        }
367                        .into(),
368                    }
369                    .into();
370                    return;
371                }
372                *item = VarDecl {
373                    decls,
374                    ..*var.take()
375                }
376                .into();
377                self.extra.push(
378                    NamedExport {
379                        span,
380                        specifiers: renamed,
381                        src: None,
382                        type_only: false,
383                        with: None,
384                    }
385                    .into(),
386                );
387            }
388            _ => {
389                item.visit_mut_children_with(self);
390            }
391        }
392    }

394    fn visit_mut_module_items(&mut self, nodes: &mut Vec<ModuleItem>) {
395        use std::mem::take;
396
397        #[cfg(feature = "concurrent")]
398        if nodes.len() >= 8 * cpu_count() {
399            ::swc_common::GLOBALS.with(|globals| {
400                use par_iter::prelude::*;
401
402                let (visitor, new_nodes) = take(nodes)
403                    .into_par_iter()
404                    .map(|mut node| {
405                        ::swc_common::GLOBALS.set(globals, || {
406                            let mut visitor = Parallel::create(&*self);
407                            node.visit_mut_with(&mut visitor);
408
409                            let mut nodes = Vec::with_capacity(4);
410
411                            ParExplode::after_one_module_item(&mut visitor, &mut nodes);
412
413                            nodes.push(node);
414
415                            (visitor, nodes)
416                        })
417                    })
418                    .reduce(
419                        || (Parallel::create(&*self), Vec::new()),
420                        |mut a, b| {
421                            Parallel::merge(&mut a.0, b.0);
422
423                            a.1.extend(b.1);
424
425                            a
426                        },
427                    );
428
429                Parallel::merge(self, visitor);
430
431                {
432                    self.after_module_items(nodes);
433                }
434
435                *nodes = new_nodes;
436            });
437
438            return;
439        }
440
441        let mut buf = Vec::with_capacity(nodes.len());
442
443        for mut node in take(nodes) {
444            let mut visitor = Parallel::create(&*self);
445            node.visit_mut_with(&mut visitor);
446            buf.push(node);
447            visitor.after_one_module_item(&mut buf);
448        }
449
450        self.after_module_items(&mut buf);
451
452        *nodes = buf;
453    }

612struct VarFolder<'a, 'b, V>

614    V: RenamedVariable,

616    orig: &'a mut Operator<'b, V>,

632    fn visit_mut_ident(&mut self, i: &mut Ident) {
633        let orig = i.clone();
634        if self.orig.rename_ident(i).is_ok() {
635            self.renamed
636                .push(ExportSpecifier::Named(ExportNamedSpecifier {
637                    span: i.span,
638                    exported: Some(ModuleExportName::Ident(orig)),
639                    orig: ModuleExportName::Ident(i.clone()),
640                    is_type_only: false,
641                }));
642        }
643    }

646impl<V> Operator<'_, V>
647where
648    V: RenamedVariable,
649{
650    /// Returns `Ok(renamed_ident)` if ident should be renamed.
651    fn rename_ident(&mut self, ident: &mut Ident) -> Result<(), ()> {
652        if let Some(new_id) = self.rename.get(&ident.to_id()) {
653            let (new_sym, new_ctxt) = new_id.to_id();
654
655            if new_sym == ident.sym {
656                return Err(());
657            }
658
659            ident.ctxt = new_ctxt;
660            ident.sym = new_sym;
661            return Ok(());
662        }
663
664        Err(())
665    }
