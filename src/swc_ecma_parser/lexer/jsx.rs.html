7    pub(super) fn read_jsx_token(&mut self) -> LexResult<Option<Token>> {
8        debug_assert!(self.syntax.jsx());
9
10        let start = self.input.cur_pos();
11        let mut chunk_start = self.input.cur_pos();
12        let mut value = String::new();
13
14        loop {
15            let cur = match self.input.cur() {
16                Some(c) => c,
17                None => {
18                    let start = self.state.start;
19                    self.error(start, SyntaxError::UnterminatedJSXContents)?
20                }
21            };
22            let cur_pos = self.input.cur_pos();
23
24            match cur {
25                '<' if self.had_line_break_before_last() && self.is_str("<<<<<< ") => {
26                    let span = Span::new(cur_pos, cur_pos + BytePos(7));
27
28                    self.emit_error_span(span, SyntaxError::TS1185);
29                    self.skip_line_comment(6);
30                    self.skip_space::<true>();
31                    return self.read_token();
32                }
33                '<' | '{' => {
34                    //
35                    if cur_pos == self.state.start {
36                        if cur == '<' && self.state.is_expr_allowed {
37                            unsafe {
38                                // Safety: cur() was Some('<')
39                                self.input.bump();
40                            }
41                            return Ok(Some(Token::JSXTagStart));
42                        }
43                        return self.read_token();
44                    }
45
46                    let value = if value.is_empty() {
47                        // Fast path: We don't need to allocate extra buffer for value
48                        let s = unsafe {
49                            // Safety: We already checked for the range
50                            self.input.slice(chunk_start, cur_pos)
51                        };
52                        self.atoms.atom(s)
53                    } else {
54                        value.push_str(unsafe {
55                            // Safety: We already checked for the range
56                            self.input.slice(chunk_start, cur_pos)
57                        });
58                        self.atoms.atom(value)
59                    };
60
61                    let raw = {
62                        let s = unsafe {
63                            // Safety: We already checked for the range
64                            self.input.slice(start, cur_pos)
65                        };
66                        self.atoms.atom(s)
67                    };
68
69                    return Ok(Some(Token::JSXText { raw, value }));
70                }
71                '>' => {
72                    self.emit_error(
73                        cur_pos,
74                        SyntaxError::UnexpectedTokenWithSuggestions {
75                            candidate_list: vec!["`{'>'}`", "`&gt;`"],
76                        },
77                    );
78                    unsafe {
79                        // Safety: cur() was Some('>')
80                        self.input.bump()
81                    }
82                }
83                '}' => {
84                    self.emit_error(
85                        cur_pos,
86                        SyntaxError::UnexpectedTokenWithSuggestions {
87                            candidate_list: vec!["`{'}'}`", "`&rbrace;`"],
88                        },
89                    );
90                    unsafe {
91                        // Safety: cur() was Some('}')
92                        self.input.bump()
93                    }
94                }
95                '&' => {
96                    value.push_str(unsafe {
97                        // Safety: We already checked for the range
98                        self.input.slice(chunk_start, cur_pos)
99                    });
100
101                    let jsx_entity = self.read_jsx_entity()?;
102
103                    value.push(jsx_entity.0);
104                    chunk_start = self.input.cur_pos();
105                }
106
107                _ => {
108                    if cur.is_line_terminator() {
109                        value.push_str(unsafe {
110                            // Safety: We already checked for the range
111                            self.input.slice(chunk_start, cur_pos)
112                        });
113                        match self.read_jsx_new_line(true)? {
114                            Either::Left(s) => value.push_str(s),
115                            Either::Right(c) => value.push(c),
116                        }
117                        chunk_start = self.input.cur_pos();
118                    } else {
119                        unsafe {
120                            // Safety: cur() was Some(c)
121                            self.input.bump()
122                        }
123                    }
124                }
125            }
126        }
127    }

129    pub(super) fn read_jsx_entity(&mut self) -> LexResult<(char, String)> {
130        debug_assert!(self.syntax.jsx());
131
132        fn from_code(s: &str, radix: u32) -> LexResult<char> {
133            // TODO(kdy1): unwrap -> Err
134            let c = char::from_u32(
135                u32::from_str_radix(s, radix).expect("failed to parse string as number"),
136            )
137            .expect("failed to parse number as char");
138
139            Ok(c)
140        }
141
142        fn is_hex(s: &str) -> bool {
143            s.chars().all(|c| c.is_ascii_hexdigit())
144        }
145
146        fn is_dec(s: &str) -> bool {
147            s.chars().all(|c| c.is_ascii_digit())
148        }
149
150        let mut s = SmartString::<LazyCompact>::default();
151
152        let c = self.input.cur();
153        debug_assert_eq!(c, Some('&'));
154        unsafe {
155            // Safety: cur() was Some('&')
156            self.input.bump();
157        }
158
159        let start_pos = self.input.cur_pos();
160
161        for _ in 0..10 {
162            let c = match self.input.cur() {
163                Some(c) => c,
164                None => break,
165            };
166            unsafe {
167                // Safety: cur() was Some(c)
168                self.input.bump();
169            }
170
171            if c == ';' {
172                if let Some(stripped) = s.strip_prefix('#') {
173                    if stripped.starts_with('x') {
174                        if is_hex(&s[2..]) {
175                            let value = from_code(&s[2..], 16)?;
176
177                            return Ok((value, format!("&{};", s)));
178                        }
179                    } else if is_dec(stripped) {
180                        let value = from_code(stripped, 10)?;
181
182                        return Ok((value, format!("&{};", s)));
183                    }
184                } else if let Some(entity) = xhtml(&s) {
185                    return Ok((entity, format!("&{};", s)));
186                }
187
188                break;
189            }
190
191            s.push(c)
192        }
193
194        unsafe {
195            // Safety: start_pos is a valid position because we got it from self.input
196            self.input.reset_to(start_pos);
197        }
198
199        Ok(('&', "&".to_string()))
200    }

202    pub(super) fn read_jsx_new_line(
203        &mut self,
204        normalize_crlf: bool,
205    ) -> LexResult<Either<&'static str, char>> {
206        debug_assert!(self.syntax.jsx());
207
208        let ch = self.input.cur().unwrap();
209        unsafe {
210            // Safety: cur() was Some(ch)
211            self.input.bump();
212        }
213
214        let out = if ch == '\r' && self.input.cur() == Some('\n') {
215            unsafe {
216                // Safety: cur() was Some('\n')
217                self.input.bump();
218            }
219            Either::Left(if normalize_crlf { "\n" } else { "\r\n" })
220        } else {
221            Either::Right(ch)
222        };
223        let cur_pos = self.input.cur_pos();
224        self.state.cur_line += 1;
225        self.state.line_start = cur_pos;
226
227        Ok(out)
228    }

230    pub(super) fn read_jsx_str(&mut self, quote: char) -> LexResult<Token> {
231        debug_assert!(self.syntax.jsx());
232
233        let start = self.input.cur_pos();
234
235        unsafe {
236            // Safety: cur() was Some(quote)
237            self.input.bump(); // `quote`
238        }
239
240        let mut out = String::new();
241        let mut chunk_start = self.input.cur_pos();
242
243        loop {
244            let ch = match self.input.cur() {
245                Some(c) => c,
246                None => {
247                    let start = self.state.start;
248                    self.emit_error(start, SyntaxError::UnterminatedStrLit);
249                    break;
250                }
251            };
252
253            let cur_pos = self.input.cur_pos();
254
255            if ch == '\\' {
256                let value = unsafe {
257                    // Safety: We already checked for the range
258                    self.input.slice(chunk_start, cur_pos)
259                };
260
261                out.push_str(value);
262                out.push('\\');
263
264                self.bump();
265
266                chunk_start = self.input.cur_pos();
267
268                continue;
269            }
270
271            if ch == quote {
272                break;
273            }
274
275            if ch == '&' {
276                let value = unsafe {
277                    // Safety: We already checked for the range
278                    self.input.slice(chunk_start, cur_pos)
279                };
280
281                out.push_str(value);
282
283                let jsx_entity = self.read_jsx_entity()?;
284
285                out.push(jsx_entity.0);
286
287                chunk_start = self.input.cur_pos();
288            } else if ch.is_line_terminator() {
289                let value = unsafe {
290                    // Safety: We already checked for the range
291                    self.input.slice(chunk_start, cur_pos)
292                };
293
294                out.push_str(value);
295
296                match self.read_jsx_new_line(false)? {
297                    Either::Left(s) => {
298                        out.push_str(s);
299                    }
300                    Either::Right(c) => {
301                        out.push(c);
302                    }
303                }
304
305                chunk_start = cur_pos + BytePos(ch.len_utf8() as _);
306            } else {
307                unsafe {
308                    // Safety: cur() was Some(ch)
309                    self.input.bump();
310                }
311            }
312        }
313
314        let value = if out.is_empty() {
315            // Fast path: We don't need to allocate
316
317            let cur_pos = self.input.cur_pos();
318            let value = unsafe {
319                // Safety: We already checked for the range
320                self.input.slice(chunk_start, cur_pos)
321            };
322
323            self.atoms.atom(value)
324        } else {
325            let cur_pos = self.input.cur_pos();
326            let value = unsafe {
327                // Safety: We already checked for the range
328                self.input.slice(chunk_start, cur_pos)
329            };
330
331            out.push_str(value);
332
333            self.atoms.atom(out)
334        };
335
336        // it might be at the end of the file when
337        // the string literal is unterminated
338        if self.input.peek_ahead().is_some() {
339            unsafe {
340                // Safety: We called peek_ahead() which means cur() was Some
341                self.input.bump();
342            }
343        }
344
345        let end = self.input.cur_pos();
346        let raw = unsafe {
347            // Safety: Both of `start` and `end` are generated from `cur_pos()`
348            self.input.slice(start, end)
349        };
350
351        Ok(Token::Str {
352            value,
353            raw: self.atoms.atom(raw),
354        })
355    }

357    /// Read a JSX identifier (valid tag or attribute name).
358    ///
359    /// Optimized version since JSX identifiers can"t contain
360    /// escape characters and so can be read as single slice.
361    /// Also assumes that first character was already checked
362    /// by isIdentifierStart in readToken.
363    pub(super) fn read_jsx_word(&mut self) -> LexResult<Token> {
364        debug_assert!(self.syntax.jsx());
365        debug_assert!(self.input.cur().is_some());
366        debug_assert!(self.input.cur().unwrap().is_ident_start());
367
368        let mut first = true;
369        let slice = self.input.uncons_while(|c| {
370            if first {
371                first = false;
372                c.is_ident_start()
373            } else {
374                c.is_ident_part() || c == '-'
375            }
376        });
377
378        Ok(Token::JSXName {
379            name: self.atoms.atom(slice),
380        })
381    }

384macro_rules! xhtml {
385    (
386        $(
387            $i:ident : $s:expr,
388        )*
389    ) => {
390        fn xhtml(s: &str) -> Option<char> {
391            match s{
392                $(stringify!($i) => Some($s),)*
393                _ => None,
394            }
395        }
