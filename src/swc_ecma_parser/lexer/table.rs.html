42const EOF: ByteHandler = Some(|lexer| {
43    lexer.input.bump_bytes(1);
44

48const ERR: ByteHandler = Some(|lexer| {
49    let c = unsafe {
50        // Safety: Byte handler is only called for non-last chracters
51        lexer.input.cur().unwrap_unchecked()
52    };
53
54    let start = lexer.cur_pos();
55    unsafe {
56        // Safety: Byte handler is only called for non-last chracters
57        lexer.input.bump();
58    }
59    lexer.error_span(pos_span(start), SyntaxError::UnexpectedChar { c })?
60});

285/// Numbers

299/// Unicode
300const UNI: ByteHandler = Some(|lexer| {
301    let c = unsafe {
302        // Safety: Byte handler is only called for non-last chracters
303        lexer.input.cur().unwrap_unchecked()
304    };
305
306    // Identifier or keyword. '\uXXXX' sequences are allowed in
307    // identifiers, so '\' also dispatches to that.
308    if c == '\\' || c.is_ident_start() {
309        return lexer.read_ident_unknown().map(Some);
310    }
311
312    let start = lexer.cur_pos();
313    unsafe {
314        // Safety: Byte handler is only called for non-last chracters
315        lexer.input.bump();
316    }
317    lexer.error_span(pos_span(start), SyntaxError::UnexpectedChar { c })?
318});

338macro_rules! single_char {
339    ($name:ident, $c:literal, $token:ident) => {
340        const $name: ByteHandler = Some(|lexer| {
341            lexer.input.bump_bytes(1);
342            Ok(Some(Token::$token))
343        });
344    };
345}

362/// `^`
363const CRT: ByteHandler = Some(|lexer| {
364    // Bitwise xor
365    lexer.input.bump_bytes(1);
366    Ok(Some(if lexer.input.cur_as_ascii() == Some(b'=') {
367        lexer.input.bump_bytes(1);
368        Token::AssignOp(AssignOp::BitXorAssign)
369    } else {
370        Token::BinOp(BinOpToken::BitXor)
371    }))
372});
