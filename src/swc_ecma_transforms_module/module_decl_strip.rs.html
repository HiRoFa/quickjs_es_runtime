51impl VisitMut for ModuleDeclStrip {
52    noop_visit_mut_type!(fail);
53
54    fn visit_mut_module_items(&mut self, n: &mut Vec<ModuleItem>) {
55        let mut list = Vec::with_capacity(n.len());
56
57        for item in n.drain(..) {
58            match item {
59                ModuleItem::Stmt(stmt) => list.push(stmt.into()),
60
61                ModuleItem::ModuleDecl(mut module_decl) => {
62                    // collect link meta
63                    module_decl.visit_mut_with(self);
64                    self.has_module_decl = true;
65
66                    // emit stmt
67                    match module_decl {
68                        ModuleDecl::Import(..) => continue,
69                        ModuleDecl::ExportDecl(ExportDecl { decl, .. }) => {
70                            list.push(decl.into());
71                        }
72                        ModuleDecl::ExportNamed(..) => continue,
73                        ModuleDecl::ExportDefaultDecl(ExportDefaultDecl { decl, .. }) => match decl
74                        {
75                            DefaultDecl::Class(class_expr) => {
76                                list.extend(class_expr.as_class_decl().map(From::from))
77                            }
78                            DefaultDecl::Fn(fn_expr) => {
79                                list.extend(fn_expr.as_fn_decl().map(From::from))
80                            }
81                            DefaultDecl::TsInterfaceDecl(_) => continue,
82                        },
83                        ModuleDecl::ExportDefaultExpr(..) => {
84                            list.extend(self.export_default.take().map(From::from))
85                        }
86                        ModuleDecl::ExportAll(..) => continue,
87                        ModuleDecl::TsImportEquals(..) => continue,
88                        ModuleDecl::TsExportAssignment(..) => continue,
89                        ModuleDecl::TsNamespaceExport(..) => continue,
90                    };
91                }
92            };
93        }
94
95        *n = list;
96    }
97
98    // collect all static import
99    fn visit_mut_import_decl(&mut self, n: &mut ImportDecl) {
100        if n.type_only {
101            return;
102        }
103
104        let ImportDecl {
105            specifiers, src, ..
106        } = n.take();
107
108        self.link
109            .entry(src.value)
110            .or_default()
111            .mut_dummy_span(src.span)
112            .extend(specifiers.into_iter().map(From::from));
113    }
114
115    /// ```javascript
116    /// export const foo = 1, bar = 2, { baz } = { baz: 3 };
117    /// export let a = 1, [b] = [2];
118    /// export function x() {}
119    /// export class y {}
120    /// ```
121    /// ->
122    /// ```javascript
123    /// const foo = 1, bar = 2, { baz } = { baz: 3 };
124    /// let a = 1, [b] = [2];
125    /// function x() {}
126    /// class y {}
127    /// ```
128    fn visit_mut_export_decl(&mut self, n: &mut ExportDecl) {
129        match &n.decl {
130            Decl::Class(ClassDecl { ident, .. }) | Decl::Fn(FnDecl { ident, .. }) => {
131                self.export.insert(
132                    ident.sym.clone(),
133                    ExportItem::new((ident.span, ident.ctxt), ident.clone()),
134                );
135            }
136
137            Decl::Var(v) => {
138                self.export.extend(
139                    find_pat_ids::<_, Ident>(&v.decls)
140                        .into_iter()
141                        .map(|id| (id.sym.clone(), ExportItem::new((id.span, id.ctxt), id))),
142                );
143            }
144            _ => {}
145        };
146    }
147
148    /// ```javascript
149    /// export { foo, foo as bar, foo as "baz" };
150    /// export { "foo", foo as bar, "foo" as "baz" } from "mod";
151    /// export * as foo from "mod";
152    /// export * as "bar" from "mod";
153    /// ```
154    fn visit_mut_named_export(&mut self, n: &mut NamedExport) {
155        if n.type_only {
156            return;
157        }
158
159        let NamedExport {
160            specifiers, src, ..
161        } = n.take();
162
163        if let Some(src) = src {
164            self.link
165                .entry(src.value)
166                .or_default()
167                .mut_dummy_span(src.span)
168                .extend(specifiers.into_iter().map(From::from));
169        } else {
170            self.export.extend(specifiers.into_iter().map(|e| match e {
171                ExportSpecifier::Namespace(..) => {
172                    unreachable!("`export *` without src is invalid")
173                }
174                ExportSpecifier::Default(..) => {
175                    unreachable!("`export foo` without src is invalid")
176                }
177                ExportSpecifier::Named(ExportNamedSpecifier { orig, exported, .. }) => {
178                    let orig = match orig {
179                        ModuleExportName::Ident(id) => id,
180                        ModuleExportName::Str(_) => {
181                            unreachable!(r#"`export {{ "foo" }}` without src is invalid"#)
182                        }
183                    };
184
185                    if let Some(exported) = exported {
186                        let (export_name, export_name_span) = match exported {
187                            ModuleExportName::Ident(Ident {
188                                ctxt, span, sym, ..
189                            }) => (sym, (span, ctxt)),
190                            ModuleExportName::Str(Str { span, value, .. }) => {
191                                (value, (span, Default::default()))
192                            }
193                        };
194
195                        (export_name, ExportItem::new(export_name_span, orig))
196                    } else {
197                        (
198                            orig.sym.clone(),
199                            ExportItem::new((orig.span, orig.ctxt), orig),
200                        )
201                    }
202                }
203            }))
204        }
205    }
206
207    /// ```javascript
208    /// export default class foo {};
209    /// export default class {};
210    /// export default function bar () {};
211    /// export default function () {};
212    /// ```
213    /// ->
214    /// ```javascript
215    /// class foo {};
216    /// class _default {};
217    /// function bar () {};
218    /// function _default () {};
219    /// ```
220    fn visit_mut_export_default_decl(&mut self, n: &mut ExportDefaultDecl) {
221        match &mut n.decl {
222            DefaultDecl::Class(class_expr) => {
223                let ident = class_expr
224                    .ident
225                    .get_or_insert_with(|| private_ident!(n.span, "_default"))
226                    .clone();
227
228                self.export.insert(
229                    atom!("default"),
230                    ExportItem::new((n.span, Default::default()), ident),
231                );
232            }
233            DefaultDecl::Fn(fn_expr) => {
234                let ident = fn_expr
235                    .ident
236                    .get_or_insert_with(|| private_ident!(n.span, "_default"))
237                    .clone();
238
239                self.export.insert(
240                    atom!("default"),
241                    ExportItem::new((n.span, Default::default()), ident),
242                );
243            }
244            DefaultDecl::TsInterfaceDecl(_) => {}
245        }
246    }
247
248    /// ```javascript
249    /// export default foo;
250    /// export default 1
251    /// ```
252    /// ->
253    /// ```javascript
254    /// var _default = foo;
255    /// var _default = 1;
256    /// ```
257    fn visit_mut_export_default_expr(&mut self, n: &mut ExportDefaultExpr) {
258        let ident = private_ident!(n.span, "_default");
259
260        self.export.insert(
261            atom!("default"),
262            ExportItem::new((n.span, Default::default()), ident.clone()),
263        );
264
265        self.export_default = Some(
266            n.expr
267                .take()
268                .into_var_decl(self.const_var_kind, ident.into())
269                .into(),
270        );
271    }
272
273    /// ```javascript
274    /// export * from "mod";
275    /// ```
276    fn visit_mut_export_all(&mut self, n: &mut ExportAll) {
277        let Str {
278            value: src_key,
279            span: src_span,
280            ..
281        } = *n.take().src;
282
283        self.link
284            .entry(src_key)
285            .or_default()
286            .mut_dummy_span(src_span)
287            .insert(LinkSpecifier::ExportStar);
288    }
289
290    /// ```javascript
291    /// import foo = require("mod");
292    /// export import foo = require("mod");
293    /// ```
294    fn visit_mut_ts_import_equals_decl(&mut self, n: &mut TsImportEqualsDecl) {
295        if n.is_type_only {
296            return;
297        }
298
299        let TsImportEqualsDecl {
300            id,
301            module_ref,
302            is_export,
303            ..
304        } = n;
305
306        if let TsModuleRef::TsExternalModuleRef(TsExternalModuleRef {
307            expr:
308                Str {
309                    span,
310                    value: src_key,
311                    ..
312                },
313            ..
314        }) = module_ref
315        {
316            if *is_export {
317                self.export.insert(
318                    id.sym.clone(),
319                    ExportItem::new((id.span, id.ctxt), id.clone()),
320                );
321            }
322
323            self.link
324                .entry(src_key.clone())
325                .or_default()
326                .mut_dummy_span(*span)
327                .insert(LinkSpecifier::ImportEqual(id.to_id()));
328        }
329    }
330
331    /// ```javascript
332    /// export = expr;
333    /// ```
334    fn visit_mut_ts_export_assignment(&mut self, n: &mut TsExportAssignment) {
335        self.export_assign.get_or_insert(n.expr.take());
336    }

339#[derive(Debug, PartialEq, Eq, Hash)]
340pub enum LinkSpecifier {
341    ///```javascript
342    /// import "mod";
343    /// import {} from "mod",
344    /// import { type foo } from "mod";
345    /// export {} from "mod";
346    /// export { type foo } from "mod";
347    /// ```
348    Empty,
349
350    /// ```javascript
351    /// import { imported as local, local } from "mod";
352    /// import { "imported" as local } from "mod";
353    /// ```
354    /// Note: imported will never be `default`
355    ImportNamed { imported: Option<Atom>, local: Id },
356
357    /// ```javascript
358    /// import foo from "mod";
359    /// ```
360    ImportDefault(Id),
361
362    /// ```javascript
363    /// import * as foo from "mod";
364    /// ```
365    ImportStarAs(Id),
366
367    /// ```javascript
368    /// export { orig, orig as exported } from "mod";
369    /// export { "orig", "orig" as "exported" } from "mod";
370    /// ```
371    /// Note: orig will never be `default`
372    ExportNamed {
373        orig: (Atom, SpanCtx),
374        exported: Option<(Atom, SpanCtx)>,
375    },
376
377    /// ```javascript
378    /// export { default } from "foo";
379    /// export { "default" } from "foo";
380    /// export { default as foo } from "mod";
381    /// ```
382    /// (default_span, local_sym, local_span)
383    ExportDefaultAs(SpanCtx, Atom, SpanCtx),
384
385    /// ```javascript
386    /// export * as foo from "mod";
387    /// export * as "bar" from "mod";
388    /// ```
389    ExportStarAs(Atom, SpanCtx),
390
391    /// ```javascript
392    /// export * from "mod";
393    /// ```
394    ExportStar,
395
396    /// ```javascript
397    /// import foo = require("foo");
398    /// ```
399    ImportEqual(Id),
400}

402impl From<ImportSpecifier> for LinkSpecifier {
403    fn from(i: ImportSpecifier) -> Self {
404        match i {
405            ImportSpecifier::Namespace(ImportStarAsSpecifier { local, .. }) => {
406                Self::ImportStarAs(local.to_id())
407            }
408
409            ImportSpecifier::Default(ImportDefaultSpecifier { local, .. }) => {
410                Self::ImportDefault(local.to_id())
411            }
412            ImportSpecifier::Named(ImportNamedSpecifier {
413                is_type_only: false,
414                local,
415                imported:
416                    Some(ModuleExportName::Ident(Ident { sym: s, .. }))
417                    | Some(ModuleExportName::Str(Str { value: s, .. })),
418                ..
419            }) if &*s == "default" => Self::ImportDefault(local.to_id()),
420
421            ImportSpecifier::Named(ImportNamedSpecifier {
422                is_type_only: false,
423                local,
424                imported,
425                ..
426            }) => {
427                let imported = imported.map(|e| match e {
428                    ModuleExportName::Ident(Ident { sym, .. }) => sym,
429                    ModuleExportName::Str(Str { value, .. }) => value,
430                });
431
432                Self::ImportNamed {
433                    local: local.to_id(),
434                    imported,
435                }
436            }
437            _ => Self::Empty,
438        }
439    }
440}

442impl From<ExportSpecifier> for LinkSpecifier {
443    fn from(e: ExportSpecifier) -> Self {
444        match e {
445            ExportSpecifier::Namespace(ExportNamespaceSpecifier {
446                name:
447                    ModuleExportName::Ident(Ident { span, sym, .. })
448                    | ModuleExportName::Str(Str {
449                        span, value: sym, ..
450                    }),
451                ..
452            }) => Self::ExportStarAs(sym, (span, SyntaxContext::empty())),
453
454            ExportSpecifier::Default(ExportDefaultSpecifier { exported }) => {
455                // https://github.com/tc39/proposal-export-default-from
456                Self::ExportDefaultAs(
457                    (exported.span, exported.ctxt),
458                    exported.sym,
459                    (exported.span, exported.ctxt),
460                )
461            }
462
463            ExportSpecifier::Named(ExportNamedSpecifier {
464                is_type_only: false,
465                orig,
466                exported,
467                ..
468            }) => {
469                let orig = match orig {
470                    ModuleExportName::Ident(Ident { span, sym, .. })
471                    | ModuleExportName::Str(Str {
472                        span, value: sym, ..
473                    }) => (sym, (span, SyntaxContext::empty().apply_mark(Mark::new()))),
474                };
475
476                let exported = exported.map(|exported| match exported {
477                    ModuleExportName::Ident(Ident { span, sym, .. })
478                    | ModuleExportName::Str(Str {
479                        span, value: sym, ..
480                    }) => (sym, (span, SyntaxContext::empty().apply_mark(Mark::new()))),
481                });
482
483                match (&*orig.0, orig.1) {
484                    ("default", default_span) => {
485                        let (sym, span) = exported.unwrap_or(orig);
486
487                        Self::ExportDefaultAs(default_span, sym, span)
488                    }
489                    _ => Self::ExportNamed { orig, exported },
490                }
491            }
492            _ => Self::Empty,
493        }
494    }
495}

500use bitflags::bitflags;

502bitflags! {
503    #[derive(Default, Debug)]
504    pub struct LinkFlag: u8 {
505        const NAMED = 1 << 0;
506        const DEFAULT = 1 << 1;
507        const NAMESPACE = Self::NAMED.bits() | Self::DEFAULT.bits();
508        const EXPORT_STAR = 1 << 2;
509        const IMPORT_EQUAL = 1 << 3;
510    }
511}
512
513impl LinkFlag {
514    pub fn interop(&self) -> bool {
515        self.intersects(Self::DEFAULT)
516    }
517
518    pub fn has_named(&self) -> bool {
519        self.intersects(Self::NAMED)
520    }
521
522    pub fn namespace(&self) -> bool {
523        self.contains(Self::NAMESPACE)
524    }
525
526    pub fn need_raw_import(&self) -> bool {
527        self.interop() && self.intersects(Self::IMPORT_EQUAL)
528    }
529
530    pub fn export_star(&self) -> bool {
531        self.intersects(Self::EXPORT_STAR)
532    }
533}

535impl From<&LinkSpecifier> for LinkFlag {
536    fn from(s: &LinkSpecifier) -> Self {
537        match s {
538            LinkSpecifier::Empty => Self::empty(),
539            LinkSpecifier::ImportStarAs(..) => Self::NAMESPACE,
540            LinkSpecifier::ImportDefault(..) => Self::DEFAULT,
541            LinkSpecifier::ImportNamed { .. } => Self::NAMED,
542
543            LinkSpecifier::ExportStarAs(..) => Self::NAMESPACE,
544            LinkSpecifier::ExportDefaultAs(..) => Self::DEFAULT,
545            LinkSpecifier::ExportNamed { .. } => Self::NAMED,
546
547            LinkSpecifier::ImportEqual(..) => Self::IMPORT_EQUAL,
548            LinkSpecifier::ExportStar => Self::EXPORT_STAR,
549        }
550    }
551}

553impl From<&ImportSpecifier> for LinkFlag {
554    fn from(i: &ImportSpecifier) -> Self {
555        match i {
556            ImportSpecifier::Namespace(..) => Self::NAMESPACE,
557
558            ImportSpecifier::Default(ImportDefaultSpecifier { .. }) => Self::DEFAULT,
559
560            ImportSpecifier::Named(ImportNamedSpecifier {

562                imported:
563                    Some(ModuleExportName::Ident(Ident { sym: default, .. }))
564                    | Some(ModuleExportName::Str(Str { value: default, .. })),
565                ..
566            }) if &**default == "default" => Self::DEFAULT,
567
568            ImportSpecifier::Named(ImportNamedSpecifier {
569                is_type_only: false,
570                ..
571            }) => Self::NAMED,
572
573            _ => Self::empty(),
574        }
575    }

578impl From<&ExportSpecifier> for LinkFlag {

581            ExportSpecifier::Namespace(..) => Self::NAMESPACE,

585            ExportSpecifier::Named(ExportNamedSpecifier {

587                orig:
588                    ModuleExportName::Ident(Ident { sym: s, .. })
589                    | ModuleExportName::Str(Str { value: s, .. }),

593            ExportSpecifier::Named(ExportNamedSpecifier {
594                is_type_only: false,
595                ..
596            }) => Self::NAMED,
597
598            _ => Self::empty(),
599        }

628/// Reduce self to generate ImportMap and ExportObjPropList
629pub(crate) trait LinkSpecifierReducer {
630    fn reduce(
631        self,
632        import_map: &mut ImportMap,
633        export_obj_prop_list: &mut ExportObjPropList,
634        mod_ident: &Ident,
635        raw_mod_ident: &Option<Ident>,
636        ref_to_mod_ident: &mut bool,
637        // do not emit `mod.default`, emit `mod` instead
638        default_nowrap: bool,
639    );
640}
641
642impl LinkSpecifierReducer for FxHashSet<LinkSpecifier> {
643    fn reduce(
644        self,
645        import_map: &mut ImportMap,
646        export_obj_prop_list: &mut ExportObjPropList,
647        mod_ident: &Ident,
648        raw_mod_ident: &Option<Ident>,
649        ref_to_mod_ident: &mut bool,
650        default_nowrap: bool,
651    ) {
652        self.into_iter().for_each(|s| match s {
653            LinkSpecifier::ImportNamed { imported, local } => {
654                *ref_to_mod_ident = true;
655
656                import_map.insert(
657                    local.clone(),
658                    (mod_ident.clone(), imported.or(Some(local.0))),
659                );
660            }
661            LinkSpecifier::ImportDefault(id) => {
662                *ref_to_mod_ident = true;
663
664                import_map.insert(
665                    id,
666                    (
667                        mod_ident.clone(),
668                        (!default_nowrap).then(|| atom!("default")),
669                    ),
670                );
671            }
672            LinkSpecifier::ImportStarAs(id) => {
673                *ref_to_mod_ident = true;
674
675                import_map.insert(id, (mod_ident.clone(), None));
676            }
677            LinkSpecifier::ExportNamed { orig, exported } => {
678                *ref_to_mod_ident = true;
679
680                // ```javascript
681                // export { foo as bar } from "mod";
682                //
683                // import { foo } from "mod";
684                // export { foo as bar };
685                // ```
686
687                // foo -> mod.foo
688                import_map.insert(
689                    (orig.0.clone(), orig.1 .1),
690                    (mod_ident.clone(), Some(orig.0.clone())),
691                );
692
693                let (export_name, export_name_span) = exported.unwrap_or_else(|| orig.clone());
694
695                // bar -> foo
696                export_obj_prop_list.push((
697                    export_name,
698                    ExportItem::new(export_name_span, quote_ident!(orig.1 .1, orig.1 .0, orig.0)),
699                ))
700            }
701            LinkSpecifier::ExportDefaultAs(_, key, span) => {
702                *ref_to_mod_ident = true;
703
704                // ```javascript
705                // export { default as foo } from "mod";
706                //
707                // import { default as foo } from "mod";
708                // export { foo };
709                // ```
710
711                // foo -> mod.default
712                import_map.insert(
713                    (key.clone(), span.1),
714                    (mod_ident.clone(), Some(atom!("default"))),
715                );
716
717                export_obj_prop_list.push((
718                    key.clone(),
719                    ExportItem::new(span, quote_ident!(span.1, span.0, key)),
720                ));
721            }
722            LinkSpecifier::ExportStarAs(key, span) => {
723                *ref_to_mod_ident = true;
724
725                export_obj_prop_list.push((key, ExportItem::new(span, mod_ident.clone())));
726            }
727            LinkSpecifier::ExportStar => {}
728            LinkSpecifier::ImportEqual(id) => {
729                *ref_to_mod_ident = true;
730
731                import_map.insert(
732                    id,
733                    (
734                        raw_mod_ident.clone().unwrap_or_else(|| mod_ident.clone()),
735                        None,
736                    ),
737                );
738            }
739            LinkSpecifier::Empty => {}
740        })
741    }
742}
