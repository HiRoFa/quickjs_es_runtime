20// UTF-8 ranges and tags for encoding characters
21// Copied from 48d5fe9ec560b53b1f5069219b0d62015e1de5ba^:src/libcore/char.rs
22const TAG_CONT: u8 = 0b1000_0000;
23const TAG_TWO_B: u8 = 0b1100_0000;
24const TAG_THREE_B: u8 = 0b1110_0000;
25const TAG_FOUR_B: u8 = 0b1111_0000;
26const MAX_ONE_B: u32 = 0x80;
27const MAX_TWO_B: u32 = 0x800;
28const MAX_THREE_B: u32 = 0x10000;
29
30/// Copied from 48d5fe9ec560b53b1f5069219b0d62015e1de5ba^:src/libcore/char.rs
31#[inline]
32fn encode_utf8_raw(code: u32, dst: &mut [u8]) -> Option<usize> {
33    // Marked #[inline] to allow llvm optimizing it away
34    if code < MAX_ONE_B && !dst.is_empty() {
35        dst[0] = code as u8;
36        Some(1)
37    } else if code < MAX_TWO_B && dst.len() >= 2 {
38        dst[0] = (code >> 6 & 0x1f) as u8 | TAG_TWO_B;
39        dst[1] = (code & 0x3f) as u8 | TAG_CONT;
40        Some(2)
41    } else if code < MAX_THREE_B && dst.len() >= 3 {
42        dst[0] = (code >> 12 & 0x0f) as u8 | TAG_THREE_B;
43        dst[1] = (code >> 6 & 0x3f) as u8 | TAG_CONT;
44        dst[2] = (code & 0x3f) as u8 | TAG_CONT;
45        Some(3)
46    } else if dst.len() >= 4 {
47        dst[0] = (code >> 18 & 0x07) as u8 | TAG_FOUR_B;
48        dst[1] = (code >> 12 & 0x3f) as u8 | TAG_CONT;
49        dst[2] = (code >> 6 & 0x3f) as u8 | TAG_CONT;
50        dst[3] = (code & 0x3f) as u8 | TAG_CONT;
51        Some(4)
52    } else {
53        None
54    }
55}

57/// Copied from 48d5fe9ec560b53b1f5069219b0d62015e1de5ba^:src/libcore/char.rs
58#[inline]
59fn encode_utf16_raw(mut ch: u32, dst: &mut [u16]) -> Option<usize> {
60    // Marked #[inline] to allow llvm optimizing it away
61    if (ch & 0xffff) == ch && !dst.is_empty() {
62        // The BMP falls through (assuming non-surrogate, as it should)
63        dst[0] = ch as u16;
64        Some(1)
65    } else if dst.len() >= 2 {
66        // Supplementary planes break into surrogates.
67        ch -= 0x1_0000;
68        dst[0] = 0xd800 | ((ch >> 10) as u16);
69        dst[1] = 0xdc00 | ((ch as u16) & 0x3ff);
70        Some(2)
71    } else {
72        None
73    }
74}

76/// Copied from core::str::next_code_point
77#[inline]
78pub fn next_code_point(bytes: &mut slice::Iter<u8>) -> Option<u32> {
79    // Decode UTF-8
80    let x = match bytes.next() {
81        None => return None,
82        Some(&next_byte) if next_byte < 128 => return Some(next_byte as u32),
83        Some(&next_byte) => next_byte,
84    };
85
86    // Multibyte case follows
87    // Decode from a byte combination out of: [[[x y] z] w]
88    // NOTE: Performance is sensitive to the exact formulation here
89    let init = utf8_first_byte(x, 2);
90    let y = unwrap_or_0(bytes.next());
91    let mut ch = utf8_acc_cont_byte(init, y);
92    if x >= 0xe0 {
93        // [[x y z] w] case
94        // 5th bit in 0xE0 .. 0xEF is always clear, so `init` is still valid
95        let z = unwrap_or_0(bytes.next());
96        let y_z = utf8_acc_cont_byte((y & CONT_MASK) as u32, z);
97        ch = init << 12 | y_z;
98        if x >= 0xf0 {
99            // [x y z w] case
100            // use only the lower 3 bits of `init`
101            let w = unwrap_or_0(bytes.next());
102            ch = (init & 7) << 18 | utf8_acc_cont_byte(y_z, w);
103        }
104    }
105
106    Some(ch)
107}

109#[inline]
110fn utf8_first_byte(byte: u8, width: u32) -> u32 {
111    (byte & (0x7f >> width)) as u32
112}
113
114/// Return the value of `ch` updated with continuation byte `byte`.
115#[inline]
116fn utf8_acc_cont_byte(ch: u32, byte: u8) -> u32 {
117    (ch << 6) | (byte & CONT_MASK) as u32
118}
119
120#[inline]
121fn unwrap_or_0(opt: Option<&u8>) -> u8 {
122    match opt {
123        Some(&byte) => byte,
124        None => 0,
125    }
126}

128/// Mask of the value bits of a continuation byte
129const CONT_MASK: u8 = 0b0011_1111;

131/// Copied from String::push
132/// This does **not** include the WTF-8 concatenation check.
133#[inline]
134pub fn push_code_point(string: &mut Wtf8Buf, code_point: CodePoint) {
135    let cur_len = string.len();
136    // This may use up to 4 bytes.
137    string.reserve(4);
138
139    unsafe {
140        // Attempt to not use an intermediate buffer by just pushing bytes
141        // directly onto this string.
142        let slice = slice::from_raw_parts_mut(string.bytes.as_mut_ptr().add(cur_len), 4);
143        let used = encode_utf8_raw(code_point.to_u32(), slice).unwrap_or(0);
144        string.bytes.set_len(cur_len + used);
145    }
146}

148/// Copied from core::str::StrPrelude::is_char_boundary
149#[inline]
150pub fn is_code_point_boundary(slice: &Wtf8, index: usize) -> bool {
151    if index == slice.len() {
152        return true;
153    }
154    match slice.bytes.get(index) {
155        None => false,
156        Some(&b) => !(128u8..192u8).contains(&b),
157    }
158}

160/// Copied from core::str::raw::slice_unchecked
161#[inline]
162pub unsafe fn slice_unchecked(s: &Wtf8, begin: usize, end: usize) -> &Wtf8 {
163    mem::transmute(slice::from_raw_parts(
164        s.bytes.as_ptr().add(begin),
165        end - begin,
166    ))
167}

170#[inline(never)]
171pub fn slice_error_fail(s: &Wtf8, begin: usize, end: usize) -> ! {
172    assert!(begin <= end);

176/// Copied from core::str::Utf16CodeUnits::next
177pub fn next_utf16_code_unit(iter: &mut IllFormedUtf16CodeUnits) -> Option<u16> {
178    if iter.extra != 0 {
179        let tmp = iter.extra;
180        iter.extra = 0;
181        return Some(tmp);
182    }
183
184    let mut buf = [0u16; 2];
185    iter.code_points.next().map(|code_point| {
186        let n = encode_utf16_raw(code_point.to_u32(), &mut buf).unwrap_or(0);
187        if n == 2 {
188            iter.extra = buf[1];
189        }
190        buf[0]
191    })
192}

205pub fn decode_utf16<I: IntoIterator<Item = u16>>(iterable: I) -> DecodeUtf16<I::IntoIter> {

208        buf: None,

212/// Copied from src/librustc_unicode/char.rs
213impl<I: Iterator<Item = u16>> Iterator for DecodeUtf16<I> {
214    type Item = Result<char, u16>;
215
216    fn next(&mut self) -> Option<Result<char, u16>> {
217        let u = match self.buf.take() {
218            Some(buf) => buf,
219            None => self.iter.next()?,
220        };
221
222        if !(0xd800..=0xdfff).contains(&u) {
223            // not a surrogate
224            Some(Ok(unsafe { char::from_u32_unchecked(u as u32) }))
225        } else if u >= 0xdc00 {
226            // a trailing surrogate
227            Some(Err(u))
228        } else {
229            let u2 = match self.iter.next() {
230                Some(u2) => u2,
231                // eof
232                None => return Some(Err(u)),
233            };
234            if !(0xdc00..=0xdfff).contains(&u2) {
235                // not a trailing surrogate so we're not a valid
236                // surrogate pair, so rewind to redecode u2 next time.
237                self.buf = Some(u2);
238                return Some(Err(u));
239            }
240
241            // all ok, so lets decode it.
242            let c = (((u - 0xd800) as u32) << 10 | (u2 - 0xdc00) as u32) + 0x1_0000;
243            Some(Ok(unsafe { char::from_u32_unchecked(c) }))
244        }
245    }
246
247    #[inline]
248    fn size_hint(&self) -> (usize, Option<usize>) {
249        let (low, high) = self.iter.size_hint();
250        // we could be entirely valid surrogates (2 elements per
251        // char), or entirely non-surrogates (1 element per char)
252        (low / 2, high)
253    }
254}
