1use swc_common::{
2    comments::{Comment, Comments, SingleThreadedComments},
3    BytePos, Span, DUMMY_SP,
4};
5use swc_ecma_ast::{Module, Pass, Script};
6use swc_ecma_visit::{noop_visit_mut_type, visit_mut_pass, VisitMut, VisitMutWith};
7
8/// Preserves comments that would otherwise be dropped.
9///
10/// If during compilation an ast node associated with
11/// a comment is dropped, the comment will not appear in the final emitted
12/// output. This can create problems in the JavaScript ecosystem, particularly
13/// around istanbul coverage and other tooling that relies on comment
14/// directives.
15///
16/// This transformer shifts orphaned comments to the next closest known span
17/// while making a best-effort to preserve the "general orientation" of
18/// comments.
19pub fn dropped_comments_preserver(comments: Option<SingleThreadedComments>) -> impl Pass {
20    visit_mut_pass(DroppedCommentsPreserver {
21        comments,
22        is_first_span: true,
23        known_spans: Vec::new(),
24    })
25}
26
27struct DroppedCommentsPreserver {
28    comments: Option<SingleThreadedComments>,
29    is_first_span: bool,
30    known_spans: Vec<Span>,
31}
32
33type CommentEntries = Vec<(BytePos, Vec<Comment>)>;
34
35impl VisitMut for DroppedCommentsPreserver {
36    noop_visit_mut_type!(fail);
37
38    fn visit_mut_module(&mut self, module: &mut Module) {
39        module.visit_mut_children_with(self);
40        self.known_spans
41            .sort_by(|span_a, span_b| span_a.lo.cmp(&span_b.lo));
42        self.shift_comments_to_known_spans();
43    }
44
45    fn visit_mut_script(&mut self, script: &mut Script) {
46        script.visit_mut_children_with(self);
47        self.known_spans
48            .sort_by(|span_a, span_b| span_a.lo.cmp(&span_b.lo));
49        self.shift_comments_to_known_spans();
50    }
51
52    fn visit_mut_span(&mut self, span: &mut Span) {
53        if span.is_dummy() || self.is_first_span {
54            self.is_first_span = false;
55            return;
56        }
57
58        self.known_spans.push(*span);
59        span.visit_mut_children_with(self)
60    }
61}
62
63impl DroppedCommentsPreserver {
64    fn shift_comments_to_known_spans(&self) {
65        if let Some(comments) = &self.comments {
66            let trailing_comments = self.shift_leading_comments(comments);
67
68            self.shift_trailing_comments(trailing_comments);
69        }
70    }
71
72    /// We'll be shifting all comments to known span positions, so drain the
73    /// current comments first to limit the amount of look ups needed into
74    /// the hashmaps.
75    ///
76    /// This way, we only need to take the comments once, and then add them back
77    /// once.
78    fn collect_existing_comments(&self, comments: &SingleThreadedComments) -> CommentEntries {
79        let (mut leading_comments, mut trailing_comments) = comments.borrow_all_mut();
80        let mut existing_comments: CommentEntries = leading_comments
81            .drain()
82            .chain(trailing_comments.drain())
83            .collect();
84
85        existing_comments.sort_by(|(bp_a, _), (bp_b, _)| bp_a.cmp(bp_b));
86
87        existing_comments
88    }
89
90    /// Shift all comments to known leading positions.
91    /// This prevents trailing comments from ending up associated with
92    /// nodes that will not emit trailing comments, while
93    /// preserving any comments that might show up after all code positions.
94    ///
95    /// This maintains the highest fidelity between existing comment positions
96    /// of pre and post compiled code.
97    fn shift_leading_comments(&self, comments: &SingleThreadedComments) -> CommentEntries {
98        let mut existing_comments = self.collect_existing_comments(comments);
99
100        existing_comments.sort_by(|(bp_a, _), (bp_b, _)| bp_a.cmp(bp_b));
101
102        for span in self.known_spans.iter() {
103            let cut_point = existing_comments.partition_point(|(bp, _)| *bp <= span.lo);
104            let collected_comments = existing_comments
105                .drain(..cut_point)
106                .flat_map(|(_, c)| c)
107                .collect::<Vec<Comment>>();
108            self.comments
109                .add_leading_comments(span.lo, collected_comments)
110        }
111
112        existing_comments
113    }
114
115    /// These comments trail all known span lo byte positions.
116    /// Therefore, by shifting them to trail the highest known hi position, we
117    /// ensure that any remaining trailing comments are emitted in a
118    /// similar location
119    fn shift_trailing_comments(&self, remaining_comment_entries: CommentEntries) {
120        let last_trailing = self
121            .known_spans
122            .iter()
123            .max_by_key(|span| span.hi)
124            .cloned()
125            .unwrap_or(DUMMY_SP);
126
127        self.comments.add_trailing_comments(
128            last_trailing.hi,
129            remaining_comment_entries
130                .into_iter()
131                .flat_map(|(_, c)| c)
132                .collect(),
133        );
134    }
135}
