7/// A trait for finding the next match in a regex.
8/// This is broken out from Executor to avoid needing to thread lifetimes
9/// around.
10pub trait MatchProducer: core::fmt::Debug {
11    /// The position type of our indexer.
12    type Position: PositionType;
13
14    /// \return an initial position for the given start offset.
15    fn initial_position(&self, offset: usize) -> Option<Self::Position>;
16
17    /// Attempt to match at the given location.
18    /// \return either the Match and the position to start looking for the next
19    /// match, or None on failure.
20    fn next_match(
21        &mut self,
22        pos: Self::Position,
23        next_start: &mut Option<Self::Position>,
24    ) -> Option<Match>;
25}
26
27/// A trait for executing a regex.
28pub trait Executor<'r, 't>: MatchProducer {
29    /// The ASCII variant.
30    type AsAscii: Executor<'r, 't>;
31
32    /// Construct a new Executor.
33    fn new(re: &'r CompiledRegex, text: &'t str) -> Self;
34}
35
36/// A struct which enables iteration over matches.
37#[derive(Debug)]
38pub struct Matches<Producer: MatchProducer> {
39    mp: Producer,
40    position: Option<Producer::Position>,
41}
42
43impl<Producer: MatchProducer> Matches<Producer> {
44    pub fn new(mp: Producer, start: usize) -> Self {
45        let position = mp.initial_position(start);
46        Matches { mp, position }
47    }
48}
49
50impl<Producer: MatchProducer> Iterator for Matches<Producer> {
51    type Item = Match;
52    fn next(&mut self) -> Option<Self::Item> {
53        let pos = self.position?;
54        self.mp.next_match(pos, &mut self.position)
55    }
