10/// When unrolling a loop, the largest minimum count we will unroll.
11const LOOP_UNROLL_THRESHOLD: usize = 5;
12
13/// Things that a Pass may do.
14pub enum PassAction {
15    // Do nothing to the given node.
16    Keep,
17
18    // Notes that we modified the node in-place.
19    Modified,
20
21    // Remove the given node outright, effectively replacing it with empty.
22    Remove,
23
24    /// Replace the given node with a new Node.
25    Replace(Node),
26}
27
28#[derive(Debug)]
29struct Pass<'a, F>
30where
31    F: FnMut(&mut Node, &Walk) -> PassAction,
32{
33    // The function.
34    func: &'a mut F,
35
36    // Whether this pass has changed anything.
37    changed: bool,
38
39    // If the regex is in unicode mode.
40    unicode: bool,
41}
42
43impl<'a, F> Pass<'a, F>
44where
45    F: FnMut(&mut Node, &Walk) -> PassAction,
46{
47    fn new(func: &'a mut F, unicode: bool) -> Self {
48        Pass {
49            func,
50            changed: false,
51            unicode,
52        }
53    }
54
55    fn run_postorder(&mut self, start: &mut Node) {
56        walk_mut(
57            true,
58            self.unicode,
59            start,
60            &mut |n: &mut Node, walk: &mut Walk| match (self.func)(n, walk) {
61                PassAction::Keep => {}
62                PassAction::Modified => {
63                    self.changed = true;
64                }
65                PassAction::Remove => {
66                    *n = Node::Empty;
67                    self.changed = true;
68                }
69                PassAction::Replace(newnode) => {
70                    *n = newnode;
71                    self.changed = true;
72                }
73            },
74        )
75    }
76
77    fn run_to_fixpoint(&mut self, n: &mut Node) {
78        debug_assert!(!self.changed, "Pass has already been run");
79        loop {
80            self.changed = false;
81            self.run_postorder(n);
82            if !self.changed {
83                break;
84            }
85        }
86    }
87}

89/// Run a "pass" on a regex, which is a function that takes a Node and maybe
90/// returns a new node. \return true if something changed, false if nothing did.
91fn run_pass<F>(r: &mut Regex, func: &mut F) -> bool
92where
93    F: FnMut(&mut Node, &Walk) -> PassAction,
94{
95    let mut p = Pass::new(func, r.flags.unicode);
96    p.run_to_fixpoint(&mut r.node);
97    p.changed
98}

102// Remove empty Nodes.
103fn remove_empties(n: &mut Node, _w: &Walk) -> PassAction {
104    match n {
105        Node::Empty | Node::Goal | Node::Char { .. } => PassAction::Keep,
106        Node::ByteSequence(v) => {
107            if v.is_empty() {
108                PassAction::Remove
109            } else {
110                PassAction::Keep
111            }
112        }
113
114        // Note: do not remove empty sets. These always match against one character; an empty set
115        // should just fail.
116        Node::ByteSet(..) | Node::CharSet(..) => PassAction::Keep,
117        Node::Cat(nodes) => {
118            let blen = nodes.len();
119            nodes.retain(|nn| !nn.is_empty());
120            if nodes.len() == blen {
121                // Nothing was removed.
122                PassAction::Keep
123            } else {
124                match nodes.len() {
125                    0 => PassAction::Remove,
126                    1 => PassAction::Replace(nodes.pop().unwrap()),
127                    _ => PassAction::Modified,
128                }
129            }
130        }
131        Node::Alt(left, right) => {
132            // Empty alt may match the empty string.
133            // Remove it only if both sides are empty.
134            if left.is_empty() && right.is_empty() {
135                PassAction::Remove
136            } else {
137                PassAction::Keep
138            }
139        }
140        Node::MatchAny | Node::MatchAnyExceptLineTerminator | Node::Anchor { .. } => {
141            PassAction::Keep
142        }
143        Node::Loop {
144            quant,
145            loopee,
146            enclosed_groups,
147        } => {
148            // A loop is empty if it has an empty body, or 0 max iters.
149            // But do not remove contained capture groups.
150            if loopee.is_empty()
151                || (quant.max == Some(0) && enclosed_groups.start == enclosed_groups.end)
152            {
153                PassAction::Remove
154            } else {
155                PassAction::Keep
156            }
157        }
158        Node::Loop1CharBody { .. } => PassAction::Keep,
159        Node::CaptureGroup(..) | Node::NamedCaptureGroup(..) => {
160            // Capture groups could in principle be optimized if they only match empties.
161            PassAction::Keep
162        }
163        Node::WordBoundary { .. } | Node::BackRef { .. } | Node::Bracket { .. } => PassAction::Keep,
164        Node::LookaroundAssertion {
165            negate, contents, ..
166        } => {
167            // Negative arounds that match empties could in principle be optimized to always
168            // fail. Here we only optimize positive ones.
169            if !*negate && contents.is_empty() {
170                PassAction::Remove
171            } else {
172                PassAction::Keep
173            }
174        }
175    }
176}

178/// Check if a node contains any capture groups (direct or nested)
179fn contains_capture_groups(node: &Node) -> bool {
180    match node {
181        Node::CaptureGroup(_, _) | Node::NamedCaptureGroup(_, _, _) => true,
182        Node::Cat(nodes) => nodes.iter().any(contains_capture_groups),
183        Node::Alt(left, right) => contains_capture_groups(left) || contains_capture_groups(right),
184        Node::Loop { loopee, .. } => contains_capture_groups(loopee),
185        Node::LookaroundAssertion { contents, .. } => contains_capture_groups(contents),
186        _ => false,
187    }
188}

190// If a node can never match, replace it with an always fails node.
191fn propagate_early_fails(n: &mut Node, _w: &Walk) -> PassAction {
192    // Don't optimize nodes containing capture groups to preserve user-visible group numbers
193    if contains_capture_groups(n) {
194        return PassAction::Keep;
195    }
196
197    match n {
198        Node::Cat(nodes) => {
199            // If any child is an early fail, we are an early fail.
200            // Note this assumes that there is no node after a Goal node.
201            if nodes.iter().any(|nn| nn.match_always_fails()) {
202                PassAction::Replace(Node::make_always_fails())
203            } else {
204                PassAction::Keep
205            }
206        }
207        Node::Alt(left, right) => {
208            // If both sides are early fails, we are an early fail.
209            let left_fails = left.match_always_fails();
210            let right_fails = right.match_always_fails();
211            match (left_fails, right_fails) {
212                (true, true) => PassAction::Replace(Node::make_always_fails()),
213                (false, false) => PassAction::Keep,
214                (true, false) | (false, true) => {
215                    // Here either our left or right node always fails.
216                    // "Steal" the other and return it, replacing us.
217                    let mut new_node = Node::Empty;
218                    core::mem::swap(
219                        &mut new_node,
220                        if left_fails { &mut *right } else { &mut *left },
221                    );
222                    PassAction::Replace(new_node)
223                }
224            }
225        }
226        Node::Loop {
227            loopee,
228            quant,
229            enclosed_groups,
230        } => {
231            if enclosed_groups.start < enclosed_groups.end {
232                return PassAction::Keep;
233            }
234            // If the loop body always fails, we always fail.
235            if quant.min > 0 && loopee.match_always_fails() {
236                PassAction::Replace(Node::make_always_fails())
237            } else {
238                PassAction::Keep
239            }
240        }
241        _ => PassAction::Keep,
242    }
243}

245// Remove excess cats.
246fn decat(n: &mut Node, _w: &Walk) -> PassAction {
247    match n {
248        Node::Cat(nodes) => {
249            if nodes.is_empty() {
250                PassAction::Remove
251            } else if nodes.len() == 1 {
252                PassAction::Replace(nodes.pop().unwrap())
253            } else if nodes.iter().any(|nn| nn.is_cat()) {
254                // Flatmap child cats.
255                // Unfortunately we can't use flatmap() because there's no single iterator type
256                // we can return.
257                // Avoid copying nodes by switching them into owned vec.
258                let mut catted = Vec::new();
259                core::mem::swap(nodes, &mut catted);
260
261                // Decat them.
262                let mut decatted = Vec::new();
263                for nn in catted {
264                    match nn {
265                        Node::Cat(mut nnodes) => {
266                            decatted.append(&mut nnodes);
267                        }
268                        _ => decatted.push(nn),
269                    }
270                }
271                PassAction::Replace(Node::Cat(decatted))
272            } else {
273                PassAction::Keep
274            }
275        }
276        _ => PassAction::Keep,
277    }
278}

280/// Unfold icase chars.
281/// That means for case-insensitive characters, figure out everything that they
282/// could match.
283/// TODO: evaluate unfolding performance and consider a cache within the optimizer.
284fn unfold_icase_chars(n: &mut Node, w: &Walk) -> PassAction {
285    match *n {
286        Node::Char { c, icase } if icase && !w.unicode => {
287            let unfolded = unicode::unfold_uppercase_char(c);
288            debug_assert!(
289                unfolded.contains(&c),
290                "Char should always unfold to at least itself"
291            );
292            match unfolded.len() {
293                0 => panic!("Char should always unfold to at least itself"),
294                1 => {
295                    // Character does not fold or unfold at all.
296                    PassAction::Replace(Node::Char { c, icase: false })
297                }
298                2..=MAX_BYTE_SET_LENGTH => {
299                    // We unfolded to 2+ characters.
300                    PassAction::Replace(Node::CharSet(unfolded))
301                }
302                _ => panic!("Unfolded to more characters than we believed possible"),
303            }
304        }
305        Node::Char { c, icase } if icase => {
306            let unfolded = unicode::unfold_char(c);
307            debug_assert!(
308                unfolded.contains(&c),
309                "Char should always unfold to at least itself"
310            );
311            match unfolded.len() {
312                0 => panic!("Char should always unfold to at least itself"),
313                1 => {
314                    // Character does not fold or unfold at all.
315                    PassAction::Replace(Node::Char { c, icase: false })
316                }
317                2..=MAX_BYTE_SET_LENGTH => {
318                    // We unfolded to 2+ characters.
319                    PassAction::Replace(Node::CharSet(unfolded))
320                }
321                _ => panic!("Unfolded to more characters than we believed possible"),
322            }
323        }
324        _ => PassAction::Keep,
325    }
326}

328// Perform simple unrolling of loops that have a minimum.
329fn unroll_loops(n: &mut Node, _w: &Walk) -> PassAction {
330    match n {
331        Node::Loop {
332            loopee,
333            quant,
334            enclosed_groups,
335        } => {
336            // TODO: consider ignoring loops with nested sub-loops?
337
338            // Do not unroll loops with enclosed groups.
339            if enclosed_groups.start < enclosed_groups.end {
340                return PassAction::Keep;
341            }
342            // Do not unroll large loops, or loops which may execute zero times.
343            if quant.min == 0 || quant.min > LOOP_UNROLL_THRESHOLD {
344                return PassAction::Keep;
345            }
346
347            // We made it through. Replace us with a cat.
348            let mut unrolled = Vec::new();
349            for _ in 0..quant.min {
350                let Some(node) = loopee.try_duplicate(0) else {
351                    return PassAction::Keep;
352                };
353                unrolled.push(node);
354            }
355
356            // We unrolled 'min' elements.
357            // Maybe our loop is now empty.
358            quant.max = quant.max.map(|v| v - quant.min);
359            quant.min = 0;
360            if quant.max != Some(0) {
361                // Move the loop to the end of unrolled.
362                let mut loop_node = Node::Empty;
363                core::mem::swap(&mut loop_node, n);
364                unrolled.push(loop_node);
365            }
366            *n = Node::Cat(unrolled);
367            PassAction::Modified
368        }
369        _ => PassAction::Keep,
370    }
371}

373/// Replace Loops with 1Char loops whenever possible.
374fn promote_1char_loops(n: &mut Node, _w: &Walk) -> PassAction {
375    match n {
376        Node::Loop {
377            loopee,
378            quant,
379            enclosed_groups,
380        } => {
381            // Must be 1Char.
382            if !loopee.matches_exactly_one_char() {
383                return PassAction::Keep;
384            }
385
386            // The above check should be sufficient to ensure we have no enclosed groups.
387            assert!(
388                enclosed_groups.start >= enclosed_groups.end,
389                "Should have no enclosed groups"
390            );
391
392            // This feels hackish?
393            let mut new_loopee = Box::new(Node::Empty);
394            core::mem::swap(&mut new_loopee, loopee);
395
396            *n = Node::Loop1CharBody {
397                loopee: new_loopee,
398                quant: *quant,
399            };
400            PassAction::Modified
401        }
402        _ => PassAction::Keep,
403    }
404}

406/// Replace Cat(Char) with ByteSeq.
407/// Also replace chars with literal bytes.
408/// Don't do this in utf16 mode because UTF-16 should never match against bytes.
409/// TODO: this seems to do too much; consider breaking this up.
410#[cfg(not(feature = "utf16"))]
411fn form_literal_bytes(n: &mut Node, walk: &Walk) -> PassAction {
412    // Helper to return a mutable reference to the nodes of a literal bytes.
413    fn get_literal_bytes(n: &mut Node) -> Option<&mut Vec<u8>> {
414        match n {
415            Node::ByteSequence(v) => Some(v),
416            _ => None,
417        }
418    }
419    match n {
420        Node::Char { c, icase } if !*icase => {
421            if let Some(c) = char::from_u32(*c) {
422                let mut buff = [0; 4];
423                PassAction::Replace(Node::ByteSequence(
424                    c.encode_utf8(&mut buff).as_bytes().to_vec(),
425                ))
426            } else {
427                PassAction::Keep
428            }
429        }
430        Node::CharSet(chars) if chars.iter().all(|&c| c <= 0x7F) => {
431            // All of our chars are ASCII; use a byte set instead.
432            PassAction::Replace(Node::ByteSet(chars.iter().map(|&c| c as u8).collect()))
433        }
434        Node::Cat(nodes) => {
435            // Find and merge adjacent ByteSeq.
436            let mut modified = false;
437            for idx in 1..nodes.len() {
438                let (prev_slice, curr_slice) = nodes.split_at_mut(idx);
439                match (
440                    get_literal_bytes(prev_slice.last_mut().unwrap()),
441                    get_literal_bytes(curr_slice.first_mut().unwrap()),
442                ) {
443                    (Some(prev_bytes), Some(curr_bytes))
444                        if !prev_bytes.is_empty() && !curr_bytes.is_empty() =>
445                    {
446                        if walk.in_lookbehind {
447                            // Our characters were already reversed; we need to reverse them again
448                            // as literal bytes are always forwards. For
449                            // example, if we have (<=ab), then we will get Cat(b, a) but we want
450                            // literal bytes "ab".
451                            curr_bytes.append(prev_bytes);
452                        } else {
453                            prev_bytes.append(curr_bytes);
454                            core::mem::swap(prev_bytes, curr_bytes);
455                        }
456                        modified = true;
457                    }
458                    _ => (),
459                }
460            }
461            if modified {
462                PassAction::Modified
463            } else {
464                PassAction::Keep
465            }
466        }
467        _ => PassAction::Keep,
468    }
469}

471/// Try to reduce a bracket to something simpler.
472fn try_reduce_bracket(bc: &BracketContents) -> Option<Node> {
473    if bc.invert {
474        // Give up.
475        return None;
476    }
477
478    // Count the number of code points.
479    let mut cps_count = 0;
480    for iv in bc.cps.intervals() {
481        cps_count += iv.count_codepoints();
482    }
483    if cps_count > MAX_CHAR_SET_LENGTH {
484        // Too many code points.
485        return None;
486    }
487    // Ok, we want to make a char set.
488    // Note we cannot make a char out of surrogates; should char conversion fail we
489    // just give up.
490    let mut res = Vec::new();
491    for iv in bc.cps.intervals() {
492        for cp in iv.codepoints() {
493            res.push(cp);
494        }
495    }
496    debug_assert!(res.len() <= MAX_CHAR_SET_LENGTH, "Unexpectedly many chars");
497    Some(Node::CharSet(res))
498}

500/// Optimize brackets like [a-b].
501/// Optimize certain stupid brackets like `[a]` to a single char.
502/// Invert a bracket if it would *reduce* the number of ranges.
503/// Note we only run this once.
504fn simplify_brackets(n: &mut Node, _walk: &Walk) -> PassAction {
505    match n {
506        Node::Bracket(bc) => {
507            if let Some(new_node) = try_reduce_bracket(bc) {
508                return PassAction::Replace(new_node);
509            }
510
511            // TODO: does this ever help anything?
512            if bc.cps.intervals().len() > bc.cps.inverted_interval_count() {
513                bc.cps = bc.cps.inverted();
514                bc.invert = !bc.invert;
515                PassAction::Modified
516            } else {
517                PassAction::Keep
518            }
519        }
520        _ => PassAction::Keep,
521    }
522}

524pub fn optimize(r: &mut Regex) {
525    run_pass(r, &mut simplify_brackets);
526    loop {
527        let mut changed = false;
528        changed |= run_pass(r, &mut decat);
529        if r.flags.icase {
530            changed |= run_pass(r, &mut unfold_icase_chars);
531        }
532        changed |= run_pass(r, &mut unroll_loops);
533        changed |= run_pass(r, &mut promote_1char_loops);
534        #[cfg(not(feature = "utf16"))]
535        {
536            changed |= run_pass(r, &mut form_literal_bytes);
537        }
538        changed |= run_pass(r, &mut remove_empties);
539        changed |= run_pass(r, &mut propagate_early_fails);
540        if !changed {
541            break;
542        }
543    }
