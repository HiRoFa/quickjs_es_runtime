7/// Facilities for searching bytes.
8pub trait ByteSearcher {
9    /// Search for ourselves in a slice of bytes.
10    /// The length of the slice is unspecified and may be 0.
11    /// \return the next index of ourselves in the slice, or None.
12    fn find_in(&self, rhs: &[u8]) -> Option<usize>;
13}

15/// Helper trait that describes matching against literal bytes.
16pub trait ByteSeq: core::fmt::Debug + Copy + Clone {
17    /// Number of bytes.
18    const LENGTH: usize;
19
20    /// Test if a slice is equal.
21    /// The slice must have exactly LENGTH bytes.
22    fn equals_known_len(&self, rhs: &[u8]) -> bool;
23}
24
25unsafe extern "C" {
26    fn memcmp(s1: *const u8, s2: *const u8, n: usize) -> i32;
27}
28
29impl<const N: usize> ByteSeq for [u8; N] {
30    const LENGTH: usize = N;
31
32    #[inline(always)]
33    fn equals_known_len(&self, rhs: &[u8]) -> bool {
34        debug_assert!(rhs.len() == Self::LENGTH, "Slice has wrong length");
35        if cfg!(feature = "prohibit-unsafe") {
36            // Here's what we would like to do. However this will emit an unnecessary length compare, and an unnecessary pointer compare.
37            self == rhs
38        } else {
39            // Warning: this is delicate. We intend for the compiler to emit optimized bytewise comparisons of unaligned LENGTH bytes,
40            // where LENGTH is a compile-time constant. Rust's default == on slices will perform a pointer comparison which will always be false,
41            // and kill any vectorization.
42            // memcmp() will be optimized to the builtin.
43            unsafe { memcmp(self.as_ptr(), rhs.as_ptr(), Self::LENGTH) == 0 }
44        }
45    }
46}

161/// A helper function for formatting bitmaps, using - ranges.
162fn format_bitmap<Func>(name: &str, f: &mut fmt::Formatter<'_>, contains: Func) -> fmt::Result
163where
164    Func: Fn(u8) -> bool,
165{
166    write!(f, "{}[", name)?;
167    let mut idx = 0;
168    let mut maybe_space = "";
169    while idx <= 256 {
170        // Compute the next value not contained.
171        let mut end = idx;
172        while end <= 256 && contains(end as u8) {
173            end += 1;
174        }
175        match end - idx {
176            0 => (),
177            1 => write!(f, "{}{}", maybe_space, idx)?,
178            _ => write!(f, "{}{}-{}", maybe_space, idx, end - 1)?,
179        };
180        if end > idx {
181            maybe_space = " ";
182        }
183        idx = end + 1
184    }
185    write!(f, "]")?;
186    Ok(())
187}

189/// A bitmap covering ASCII characters.
190#[derive(Default, Copy, Clone)]
191#[repr(align(4))]
192pub struct AsciiBitmap(pub [u8; 16]);
193
194impl AsciiBitmap {
195    /// Set a byte val in this bitmap.
196    #[inline(always)]
197    pub fn set(&mut self, val: u8) {
198        debug_assert!(val <= 127, "Value should be ASCII");
199        self.0[(val >> 3) as usize] |= 1 << (val & 0x7);
200    }
201}
202
203impl fmt::Debug for AsciiBitmap {
204    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
205        format_bitmap("AsciiBitmap", f, |v| self.contains(v))
206    }
207}

209impl ByteSet for AsciiBitmap {
210    /// \return whether this bitmap contains a given value.
211    /// The value does NOT have to be ASCII.
212    #[inline(always)]
213    fn contains(&self, val: u8) -> bool {
214        // Delicate tricks to avoid branches.
215        // In general we want to compute the byte via /8, and then mask into the
216        // byte. But if the value is not ASCII then the byte could be too large.
217        // So mask off the MSB so that the byte is always in range.
218        let byte = (val & 0x7F) >> 3;
219        let bit = val & 0x7;
220
221        // Now probe the bitmap. If our sign bit was set, we want the mask to be 0;
222        // otherwise we want to set only the 'bit' offset.
223        // Invert the sign bit and reuse it.
224        let mask = ((val >> 7) ^ 1) << bit;
225
226        // Probe the bitmap. We expect the compiler to elide the bounds check.
227        (self.0[byte as usize] & mask) != 0
228    }
229}

231/// A bitmap covering all bytes.
232#[derive(Default, Copy, Clone, PartialEq, Eq)]
233#[repr(align(4))]
234pub struct ByteBitmap([u16; 16]);
235
236// TODO: the codegen here is pretty horrible; LLVM is emitting a sequence of
237// halfword instructions. Consider using a union?
238impl ByteBitmap {
239    /// Construct from a sequence of bytes.
240    pub fn new(bytes: &[u8]) -> ByteBitmap {
241        let mut bb = ByteBitmap::default();
242        for &b in bytes {
243            bb.set(b)
244        }
245        bb
246    }
247
248    /// \return whether this bitmap contains a given byte val.
249    #[inline(always)]
250    pub fn contains(&self, val: u8) -> bool {
251        let byte = val >> 4;
252        let bit = val & 0xF;
253        (self.0[byte as usize] & (1 << bit)) != 0
254    }
255
256    /// Set a bit in this bitmap.
257    #[inline(always)]
258    pub fn set(&mut self, val: u8) {
259        let byte = val >> 4;
260        let bit = val & 0xF;
261        self.0[byte as usize] |= 1 << bit;
262    }
263
264    /// Update ourselves from another bitmap, in place.
265    pub fn bitor(&mut self, rhs: &ByteBitmap) {
266        for idx in 0..self.0.len() {
267            self.0[idx] |= rhs.0[idx];
268        }
269    }
270
271    /// Invert our bits, in place.
272    pub fn bitnot(&mut self) -> &mut Self {
273        for val in self.0.iter_mut() {
274            *val = !*val;
275        }
276        self
277    }
278
279    /// Count number of set bits.
280    pub fn count_bits(&self) -> u32 {
281        self.0.iter().map(|v| v.count_ones()).sum()
282    }
283
284    /// Return ourselves as an array of a fixed length.
285    /// Panics if the array is not large enough.
286    #[allow(clippy::wrong_self_convention)]
287    #[inline(always)]
288    pub fn as_array<const N: usize>(&self) -> [u8; N] {
289        let mut array = [0u8; N];
290        let mut idx = 0;
291        for byte in 0..=255 {
292            if self.contains(byte) {
293                array[idx] = byte;
294                idx += 1;
295            }
296        }
297        array
298    }
299
300    /// \return the index of the first byte in the slice that is present in this
301    /// bitmap, using some unsafe tricks.
302    #[inline(always)]
303    fn unsafe_find_in_slice(&self, bytes: &[u8]) -> Option<usize> {
304        type Chunk = u32;
305        let bm = &self.0;
306
307        let mut offset = 0;
308        let (prefix, body, suffix) = unsafe { bytes.align_to::<Chunk>() };
309        for &byte in prefix.iter() {
310            if self.contains(byte) {
311                return Some(offset);
312            }
313            offset += 1;
314        }
315
316        for &chunk in body {
317            // Use LE. Here index 0 is the earliest address.
318            let byte_idxs = ((chunk >> 4) & 0x0F0F0F0F).to_le_bytes();
319            let bit_idxs = (chunk & 0x0F0F0F0F).to_le_bytes();
320            if (bm[byte_idxs[0] as usize] & (1 << bit_idxs[0])) != 0 {
321                return Some(offset);
322            }
323            if (bm[byte_idxs[1] as usize] & (1 << bit_idxs[1])) != 0 {
324                return Some(offset + 1);
325            }
326            if (bm[byte_idxs[2] as usize] & (1 << bit_idxs[2])) != 0 {
327                return Some(offset + 2);
328            }
329            if (bm[byte_idxs[3] as usize] & (1 << bit_idxs[3])) != 0 {
330                return Some(offset + 3);
331            }
332            offset += 4;
333        }
334
335        for &byte in suffix.iter() {
336            if self.contains(byte) {
337                return Some(offset);
338            }
339            offset += 1;
340        }
341        None
342    }
343}

345impl ByteSearcher for ByteBitmap {
346    #[inline(always)]
347    fn find_in(&self, bytes: &[u8]) -> Option<usize> {
348        if cfg!(feature = "prohibit-unsafe") {
349            for (idx, byte) in bytes.iter().enumerate() {
350                if self.contains(*byte) {
351                    return Some(idx);
352                }
353            }
354            None
355        } else {
356            self.unsafe_find_in_slice(bytes)
357        }
358    }
359}
