7/// Facilities for searching bytes.
8pub trait ByteSearcher {
9    /// Search for ourselves in a slice of bytes.
10    /// The length of the slice is unspecified and may be 0.
11    /// \return the next index of ourselves in the slice, or None.
12    fn find_in(&self, rhs: &[u8]) -> Option<usize>;
13}

117// CharSet helper. Avoid branching in the loop to get good unrolling.
118#[allow(unused_parens)]
119#[inline(always)]
120pub fn charset_contains(set: &[u32; MAX_CHAR_SET_LENGTH], c: u32) -> bool {
121    let mut result = false;
122    for &v in set.iter() {
123        result |= (v == c);
124    }
125    result
126}

128/// A helper function for formatting bitmaps, using - ranges.
129fn format_bitmap<Func>(name: &str, f: &mut fmt::Formatter<'_>, contains: Func) -> fmt::Result
130where
131    Func: Fn(u8) -> bool,
132{
133    write!(f, "{}[", name)?;
134    let mut idx = 0;
135    let mut maybe_space = "";
136    while idx <= 256 {
137        // Compute the next value not contained.
138        let mut end = idx;
139        while end <= 256 && contains(end as u8) {
140            end += 1;
141        }
142        match end - idx {
143            0 => (),
144            1 => write!(f, "{}{}", maybe_space, idx)?,
145            _ => write!(f, "{}{}-{}", maybe_space, idx, end - 1)?,
146        };
147        if end > idx {
148            maybe_space = " ";
149        }
150        idx = end + 1
151    }
152    write!(f, "]")?;
153    Ok(())
154}

156/// A bitmap covering ASCII characters.
157#[derive(Default, Copy, Clone)]
158#[repr(align(4))]
159pub struct AsciiBitmap(pub [u8; 16]);
160
161impl AsciiBitmap {
162    /// Set a byte val in this bitmap.
163    #[inline(always)]
164    pub fn set(&mut self, val: u8) {
165        debug_assert!(val <= 127, "Value should be ASCII");
166        self.0[(val >> 3) as usize] |= 1 << (val & 0x7);
167    }
168}
169
170impl fmt::Debug for AsciiBitmap {
171    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
172        format_bitmap("AsciiBitmap", f, |v| self.contains(v))
173    }
174}
175
176impl ByteSet for AsciiBitmap {
177    /// \return whether this bitmap contains a given value.
178    /// The value does NOT have to be ASCII.
179    #[inline(always)]
180    fn contains(&self, val: u8) -> bool {
181        // Delicate tricks to avoid branches.
182        // In general we want to compute the byte via /8, and then mask into the
183        // byte. But if the value is not ASCII then the byte could be too large.
184        // So mask off the MSB so that the byte is always in range.
185        let byte = (val & 0x7F) >> 3;
186        let bit = val & 0x7;
187
188        // Now probe the bitmap. If our sign bit was set, we want the mask to be 0;
189        // otherwise we want to set only the 'bit' offset.
190        // Invert the sign bit and reuse it.
191        let mask = ((val >> 7) ^ 1) << bit;
192
193        // Probe the bitmap. We expect the compiler to elide the bounds check.
194        (self.0[byte as usize] & mask) != 0
195    }
196}

198/// A bitmap covering all bytes.
199#[derive(Default, Copy, Clone, PartialEq, Eq)]
200#[repr(align(4))]
201pub struct ByteBitmap([u16; 16]);
202
203// TODO: the codegen here is pretty horrible; LLVM is emitting a sequence of
204// halfword instructions. Consider using a union?
205impl ByteBitmap {
206    /// Construct from a sequence of bytes.
207    pub fn new(bytes: &[u8]) -> ByteBitmap {
208        let mut bb = ByteBitmap::default();
209        for &b in bytes {
210            bb.set(b)
211        }
212        bb
213    }
214
215    /// \return whether this bitmap contains a given byte val.
216    #[inline(always)]
217    pub fn contains(&self, val: u8) -> bool {
218        let byte = val >> 4;
219        let bit = val & 0xF;
220        (self.0[byte as usize] & (1 << bit)) != 0
221    }
222
223    /// Set a bit in this bitmap.
224    #[inline(always)]
225    pub fn set(&mut self, val: u8) {
226        let byte = val >> 4;
227        let bit = val & 0xF;
228        self.0[byte as usize] |= 1 << bit;
229    }
230
231    /// Update ourselves from another bitmap, in place.
232    pub fn bitor(&mut self, rhs: &ByteBitmap) {
233        for idx in 0..self.0.len() {
234            self.0[idx] |= rhs.0[idx];
235        }
236    }
237
238    /// Invert our bits, in place.
239    pub fn bitnot(&mut self) -> &mut Self {
240        for val in self.0.iter_mut() {
241            *val = !*val;
242        }
243        self
244    }
245
246    /// Count number of set bits.
247    pub fn count_bits(&self) -> u32 {
248        self.0.iter().map(|v| v.count_ones()).sum()
249    }
250
251    /// Return ourselves as an array of a fixed length.
252    /// Panics if the array is not large enough.
253    #[allow(clippy::wrong_self_convention)]
254    #[inline(always)]
255    pub fn as_array<const N: usize>(&self) -> [u8; N] {
256        let mut array = [0u8; N];
257        let mut idx = 0;
258        for byte in 0..=255 {
259            if self.contains(byte) {
260                array[idx] = byte;
261                idx += 1;
262            }
263        }
264        array
265    }
266
267    /// \return the index of the first byte in the slice that is present in this
268    /// bitmap, using some unsafe tricks.
269    #[inline(always)]
270    fn unsafe_find_in_slice(&self, bytes: &[u8]) -> Option<usize> {
271        type Chunk = u32;
272        let bm = &self.0;
273
274        let mut offset = 0;
275        let (prefix, body, suffix) = unsafe { bytes.align_to::<Chunk>() };
276        for &byte in prefix.iter() {
277            if self.contains(byte) {
278                return Some(offset);
279            }
280            offset += 1;
281        }
282
283        for &chunk in body {
284            // Use LE. Here index 0 is the earliest address.
285            let byte_idxs = ((chunk >> 4) & 0x0F0F0F0F).to_le_bytes();
286            let bit_idxs = (chunk & 0x0F0F0F0F).to_le_bytes();
287            if (bm[byte_idxs[0] as usize] & (1 << bit_idxs[0])) != 0 {
288                return Some(offset);
289            }
290            if (bm[byte_idxs[1] as usize] & (1 << bit_idxs[1])) != 0 {
291                return Some(offset + 1);
292            }
293            if (bm[byte_idxs[2] as usize] & (1 << bit_idxs[2])) != 0 {
294                return Some(offset + 2);
295            }
296            if (bm[byte_idxs[3] as usize] & (1 << bit_idxs[3])) != 0 {
297                return Some(offset + 3);
298            }
299            offset += 4;
300        }
301
302        for &byte in suffix.iter() {
303            if self.contains(byte) {
304                return Some(offset);
305            }
306            offset += 1;
307        }
308        None
309    }
310}

312impl ByteSearcher for ByteBitmap {
313    #[inline(always)]
314    fn find_in(&self, bytes: &[u8]) -> Option<usize> {
315        if cfg!(feature = "prohibit-unsafe") {
316            for (idx, byte) in bytes.iter().enumerate() {
317                if self.contains(*byte) {
318                    return Some(idx);
319                }
320            }
321            None
322        } else {
323            self.unsafe_find_in_slice(bytes)
324        }
325    }
326}
