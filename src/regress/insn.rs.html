15type JumpTarget = u32;

17/// The maximum size of a byte sequence instruction.
18pub const MAX_BYTE_SEQ_LENGTH: usize = 16;
19
20/// The maximum size of an array-type-byteset instruction.
21pub const MAX_BYTE_SET_LENGTH: usize = 4;

23/// The maximum size of an array-type-charset instruction.
24/// This also happens to be the maximum number of characters in case-insensitive
25/// equivalence classes.
26pub const MAX_CHAR_SET_LENGTH: usize = 4;

28#[derive(Debug, Clone)]
29pub struct LoopFields {
30    pub loop_id: LoopID,
31    pub min_iters: usize,
32    pub max_iters: usize,
33    pub greedy: bool,
34    pub exit: JumpTarget,
35}

37#[derive(Debug, Clone)]
38/// The list of bytecode instructions.
39pub enum Insn {
40    /// The match was successful.
41    Goal,
42
43    /// Match a single char.
44    Char(u32),
45
46    /// Match a single char, case-insensitive.
47    CharICase(u32),
48
49    /// Match the start of a line (if multiline); emitted by '^'
50    StartOfLine {
51        multiline: bool,
52    },
53
54    /// Match the end of a line; emitted by '$'
55    EndOfLine {
56        multiline: bool,
57    },
58
59    /// Match any character except a line terminator; emitted by '.' only when
60    /// the dot_all flag is set to true.
61    MatchAny,
62
63    /// Match any character except a line terminator; emitted by '.'
64    MatchAnyExceptLineTerminator,
65
66    /// Enter a loop from "outside".
67    EnterLoop(LoopFields),
68
69    /// Re-enter a loop.
70    LoopAgain {
71        begin: JumpTarget,
72    },
73
74    /// The next instruction is a "1Char" instruction which always matches one
75    /// character. Attempt to match it [min, max] times.
76    Loop1CharBody {
77        min_iters: usize,
78        max_iters: usize,
79        greedy: bool,
80    },
81
82    /// Set the IP to a new value.
83    Jump {
84        target: JumpTarget,
85    },
86
87    /// The next instruction is the primary branch.
88    /// If it fails to match, jump to secondary.
89    Alt {
90        secondary: JumpTarget,
91    },
92
93    /// Enter a capture group.
94    BeginCaptureGroup(CaptureGroupID),
95
96    /// Exit a capture group.
97    EndCaptureGroup(CaptureGroupID),
98
99    /// Clear a capture group.
100    ResetCaptureGroup(CaptureGroupID),
101
102    /// Perform a backreference match.
103    BackRef(u32),
104
105    /// Match the next character against the bracket contents, stored at the given index in the CompiledRegex.
106    Bracket(usize),
107
108    /// A simple bitmap bracket for ASCII.
109    /// It contains a bitmap of the range [0, 127].
110    AsciiBracket(AsciiBitmap),
111
112    /// Perform a lookahead assertion.
113    Lookahead {
114        negate: bool,
115        start_group: CaptureGroupID,
116        end_group: CaptureGroupID,
117        continuation: JumpTarget,
118    },
119
120    /// Perform a lookbehind assertion.
121    Lookbehind {
122        negate: bool,
123        start_group: CaptureGroupID,
124        end_group: CaptureGroupID,
125        continuation: JumpTarget,
126    },
127
128    /// \w or \W word boundaries.
129    WordBoundary {
130        invert: bool,
131    },
132
133    /// Match any of the contained chars
134    /// There is no length field; characters are simply duplicated as necessary.
135    CharSet([u32; MAX_CHAR_SET_LENGTH]),
136
137    /// Match the next byte against some possibilities.
138    ByteSet2(ByteArraySet<[u8; 2]>),
139    ByteSet3(ByteArraySet<[u8; 3]>),
140    ByteSet4(ByteArraySet<[u8; 4]>),
141
142    /// Match a sequence of literal bytes.
143    ByteSeq1([u8; 1]),
144    ByteSeq2([u8; 2]),
145    ByteSeq3([u8; 3]),
146    ByteSeq4([u8; 4]),
147    ByteSeq5([u8; 5]),
148    ByteSeq6([u8; 6]),
149    ByteSeq7([u8; 7]),
150    ByteSeq8([u8; 8]),
151    ByteSeq9([u8; 9]),
152    ByteSeq10([u8; 10]),
153    ByteSeq11([u8; 11]),
154    ByteSeq12([u8; 12]),
155    ByteSeq13([u8; 13]),
156    ByteSeq14([u8; 14]),
157    ByteSeq15([u8; 15]),
158    ByteSeq16([u8; 16]),
159
160    /// An instruction that always fails, which may be produced in weird cases
161    /// like an inverted bracket which matches everything.
162    JustFail,
163}

165/// The peeled prefix start predicate.
166/// This is a fast way of locating the first potential match.
167#[derive(Debug, Clone)]
168pub enum StartPredicate {
169    /// May match an arbitrary sequence.
170    Arbitrary,
171
172    /// Look for the first instance of any of the given bytes.
173    ByteSet1([u8; 1]),
174    ByteSet2([u8; 2]),
175    ByteSet3([u8; 3]),
176
177    /// Look for a byte sequence.
178    ByteSeq(Box<memmem::Finder<'static>>),
179
180    /// Look for a byte which matches the bitmap.
181    ByteBracket(ByteBitmap),
182
183    /// The regex is anchored to the start of the line/string.
184    /// This avoids string searching entirely and only tries matching at the beginning.
185    StartAnchored,
186}

188#[derive(Debug, Clone)]
189pub struct CompiledRegex {
190    // Sequence of instructions.
191    pub insns: Vec<Insn>,
192
193    // The bracket contents, indexed by the value of the `Bracket` instruction.
194    pub brackets: Vec<BracketContents>,
195
196    // Predicate to rapidly find the first potential match.
197    pub start_pred: StartPredicate,
198
199    // Number of loops, used to populate loop data.
200    pub loops: u32,
201
202    // Number of capture groups, used to populate capture group data.
203    pub groups: u32,
204
205    // A list of capture group names. This is either:
206    //   - Empty, if there were no named capture groups.
207    //   - A list of names with length `groups`, corresponding to the capture
208    //     group names in order. Groups without names have an empty string.
209    pub group_names: Box<[Box<str>]>,
210
211    // Flags controlling matching.
212    pub flags: api::Flags,
213}
