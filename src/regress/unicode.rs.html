13// CodePointRange packs a code point and a length together into a u32.
14// We currently do not need to store any information about code points in plane 16 (U+100000),
15// which are private use, so we only need 20 bits of code point storage;
16// the remaining 12 can be the length.
17// The length is stored with a bias of -1, so the last codepoint may be obtained by adding the "length" and the first code point.
18const CODE_POINT_BITS: u32 = 20;
19const LENGTH_BITS: u32 = 32 - CODE_POINT_BITS;
20
21#[derive(Copy, Clone, Debug)]
22pub struct CodePointRange(u32);
23
24// This will trigger an error in const functions if $x is false.
25macro_rules! const_assert_true {
26    ($x:expr $(,)*) => {
27        [()][!$x as usize];
28    };
29}
30
31impl CodePointRange {
32    #[inline(always)]
33    pub const fn from(start: u32, len: u32) -> Self {
34        const_assert_true!(start < (1 << CODE_POINT_BITS));
35        const_assert_true!(len > 0 && len <= (1 << LENGTH_BITS));
36        const_assert_true!((start + len - 1) < ((1 << CODE_POINT_BITS) - 1));
37        CodePointRange((start << LENGTH_BITS) | (len - 1))
38    }
39
40    #[inline(always)]
41    const fn len_minus_1(self) -> u32 {
42        self.0 & ((1 << LENGTH_BITS) - 1)
43    }
44
45    // \return the first codepoint in the range.
46    #[inline(always)]
47    pub const fn first(self) -> u32 {
48        self.0 >> LENGTH_BITS
49    }
50
51    // \return the last codepoint in the range.
52    #[inline(always)]
53    pub const fn last(self) -> u32 {
54        self.first() + self.len_minus_1()
55    }
56}

58// The "extra" field contains a predicate mask in the low bits and a signed delta amount in the high bits.
59// A code point only transforms if its difference from the range base is 0 once masked.
60const PREDICATE_MASK_BITS: u32 = 4;
61
62pub(crate) struct FoldRange {
63    /// The range of codepoints.
64    pub(crate) range: CodePointRange,
65
66    /// Combination of the signed delta amount and predicate mask.
67    pub(crate) extra: i32,
68}
69
70impl FoldRange {
71    #[inline(always)]
72    pub const fn from(start: u32, length: u32, delta: i32, modulo: u8) -> Self {
73        const_assert_true!(modulo.is_power_of_two());
74        let mask = (modulo - 1) as i32;
75        const_assert_true!(mask < (1 << PREDICATE_MASK_BITS));
76        const_assert_true!(((delta << PREDICATE_MASK_BITS) >> PREDICATE_MASK_BITS) == delta);
77        let extra = mask | (delta << PREDICATE_MASK_BITS);
78        FoldRange {
79            range: CodePointRange::from(start, length),
80            extra,
81        }
82    }
83    #[inline(always)]
84    fn first(&self) -> u32 {
85        self.range.first()
86    }
87
88    #[inline(always)]
89    fn last(&self) -> u32 {
90        self.range.last()
91    }
92
93    #[inline(always)]
94    fn delta(&self) -> i32 {
95        self.extra >> PREDICATE_MASK_BITS
96    }
97
98    #[inline(always)]
99    fn predicate_mask(&self) -> u32 {
100        (self.extra as u32) & ((1 << PREDICATE_MASK_BITS) - 1)
101    }
102
103    fn add_delta(&self, cu: u32) -> u32 {
104        let cs = (cu as i32) + self.delta();
105        core::debug_assert!(0 <= cs && cs <= 0x10FFFF);
106        cs as u32
107    }
108
109    /// \return the Interval of transformed-to code points.
110    fn transformed_to(&self) -> Interval {
111        Interval {
112            first: self.add_delta(self.first()),
113            last: self.add_delta(self.last()),
114        }
115    }
116
117    /// \return the Interval of transformed-from code points.
118    fn transformed_from(&self) -> Interval {
119        Interval {
120            first: self.first(),
121            last: self.last(),
122        }
123    }
124
125    fn can_apply(&self, cu: u32) -> bool {
126        self.transformed_from().contains(cu)
127    }
128
129    fn apply(&self, cu: u32) -> u32 {
130        debug_assert!(self.can_apply(cu), "Cannot apply to this code point");
131        let offset = cu - self.first();
132        if (offset & self.predicate_mask()) == 0 {
133            self.add_delta(cu)
134        } else {
135            cu
136        }
137    }
138}

150pub fn fold(cu: u32) -> u32 {
151    let searched = FOLDS.binary_search_by(|fr| {
152        if fr.first() > cu {
153            Ordering::Greater
154        } else if fr.last() < cu {
155            Ordering::Less
156        } else {
157            Ordering::Equal
158        }
159    });
160    if let Ok(index) = searched {
161        let fr: &FoldRange = if cfg!(feature = "prohibit-unsafe") {
162            unsafe { FOLDS.get_unchecked(index) }
163        } else {
164            FOLDS.get(index).expect("Invalid index")
165        };
166        fr.apply(cu)
167    } else {
168        cu
169    }
170}

178        } else {

191    }

194// Add all folded characters in the given interval to the given code point set.
195// This skips characters which fold to themselves.
196fn fold_interval(iv: Interval, recv: &mut CodePointSet) {
197    let overlaps = FOLDS.equal_range_by(|tr| {
198        if tr.first() > iv.last {
199            Ordering::Greater
200        } else if tr.last() < iv.first {
201            Ordering::Less
202        } else {
203            Ordering::Equal
204        }
205    });
206    for fr in &FOLDS[overlaps] {
207        debug_assert!(
208            fr.transformed_from().overlaps(iv),
209            "Interval does not overlap transform"
210        );
211        // Find the (inclusive) range of our interval that this transform covers.
212        let first_trans = core::cmp::max(fr.first(), iv.first);
213        let last_trans = core::cmp::min(fr.last(), iv.last);
214
215        let modulo = fr.predicate_mask() + 1;
216        if modulo == 1 {
217            // Optimization: when modulo is 1, every character in range gets transformed
218            for cu in first_trans..(last_trans + 1) {
219                let cs = fr.add_delta(cu);
220                if cs != cu {
221                    recv.add_one(cs);
222                }
223            }
224        } else {
225            // Optimization: walk by modulo amount instead of checking every character
226            let offset_start = first_trans - fr.first();
227            let start_aligned = first_trans + ((modulo - (offset_start % modulo)) % modulo);
228
229            let mut cu = start_aligned;
230            while cu <= last_trans {
231                let cs = fr.add_delta(cu);
232                recv.add_one(cs);
233                cu += modulo;
234            }
235        }
236    }
237}

239/// Find all characters that fold into the given interval and add them to the given code point set.
240/// This skips characters which fold to themselves.
241fn unfold_interval(iv: Interval, recv: &mut CodePointSet) {
242    // Note: We still need to check all ranges because the relationship between
243    // transformed_from and transformed_to intervals can be complex
244    for tr in FOLDS.iter() {
245        if !iv.overlaps(tr.transformed_to()) {
246            continue;
247        }
248
249        let modulo = tr.predicate_mask() + 1;
250        let first_source = tr.first();
251        let last_source = tr.last();
252
253        let mut process_cp = |cp| {
254            let tcp = tr.apply(cp);
255            if tcp != cp && iv.contains(tcp) {
256                recv.add_one(cp);
257            }
258        };
259
260        if modulo == 1 {
261            // Optimization: when modulo is 1, every character in range gets transformed
262            for cp in first_source..(last_source + 1) {
263                process_cp(cp);
264            }
265        } else {
266            // Walk by modulo amount instead of checking every character
267            let mut cp = first_source;
268            while cp <= last_source {
269                process_cp(cp);
270                cp += modulo;
271            }
272        }
273    }
274}

276/// \return all the characters which fold to c's fold.
277/// This is a linear search across all ranges.
278/// The result always contains c.
279pub fn unfold_char(c: u32) -> Vec<u32> {
280    let mut res = vec![c];
281    let fcp = fold(c);
282    if fcp != c {
283        res.push(fcp);
284    }
285    // TODO: optimize ASCII case.
286    for tr in FOLDS.iter() {
287        if !tr.transformed_to().contains(fcp) {
288            continue;
289        }
290        for cp in tr.transformed_from().codepoints() {
291            // TODO: this can be optimized.
292            let tcp = tr.apply(cp);
293            if tcp == fcp {
294                res.push(cp);
295            }
296        }
297    }
298    res.sort_unstable();
299    res.dedup();
300    res
301}

303pub(crate) fn unfold_uppercase_char(c: u32) -> Vec<u32> {
304    let mut res = vec![c];
305    let fcp = uppercase(c);
306    if fcp != c {
307        res.push(fcp);
308    }
309    for tr in TO_UPPERCASE.iter() {
310        if !tr.transformed_to().contains(fcp) {
311            continue;
312        }
313        for cp in tr.transformed_from().codepoints() {
314            let tcp = tr.apply(cp);
315            if tcp == fcp {
316                res.push(cp);
317            }
318        }
319    }
320    res.sort_unstable();
321    res.dedup();
322    res
323}

325// Fold every character in \p input, then find all the prefolds.
326pub fn add_icase_code_points(mut input: CodePointSet) -> CodePointSet {
327    let mut folded = input.clone();
328    for iv in input.intervals() {
329        fold_interval(*iv, &mut folded)
330    }
331
332    // Reuse input storage.
333    input.clone_from(&folded);
334    for iv in folded.intervals() {
335        unfold_interval(*iv, &mut input);
336    }
337    input
338}

363pub(crate) fn unicode_property_from_str(
364    s: &str,
365    name: Option<UnicodePropertyName>,
366    unicode_sets: bool,
367) -> Option<PropertyEscapeKind> {
368    match name {
369        Some(UnicodePropertyName::GeneralCategory) => Some(PropertyEscapeKind::CharacterClass(
370            general_category_property_value_ranges(
371                &unicode_property_value_general_category_from_str(s)?,
372            ),
373        )),
374        Some(UnicodePropertyName::Script) => Some(PropertyEscapeKind::CharacterClass(
375            script_value_ranges(&unicode_property_value_script_from_str(s)?),
376        )),
377        Some(UnicodePropertyName::ScriptExtensions) => Some(PropertyEscapeKind::CharacterClass(
378            script_extensions_value_ranges(&unicode_property_value_script_from_str(s)?),
379        )),
380        None => {
381            if let Some(value) = unicode_property_binary_from_str(s) {
382                return Some(PropertyEscapeKind::CharacterClass(binary_property_ranges(
383                    &value,
384                )));
385            }
386            if unicode_sets {
387                if let Some(value) = unicode_string_property_from_str(s) {
388                    return Some(PropertyEscapeKind::StringSet(string_property_sets(&value)));
389                }
390            }
391            Some(PropertyEscapeKind::CharacterClass(
392                general_category_property_value_ranges(
393                    &unicode_property_value_general_category_from_str(s)?,
394                ),
395            ))
396        }
397    }

405    // Map from folded char to the chars that folded to it.
406    // If an entry is missing, it means either nothing folds to the char,
407    // or it folds exclusively to itself; this can be determined by comparing
408    // the char to its fold.
409    fn get_unfold_map() -> HashMap<u32, Vec<u32>> {
410        let mut unfold_map: HashMap<u32, Vec<u32>> = HashMap::new();
411        for c in 0..=0x10FFFF {
412            let fc = fold(c);
413            if fc != c {
414                unfold_map.entry(fc).or_default().push(c);
415            }
416        }
417
418        // We neglected self-folds - add them now, but only for entries
419        // where something else folds to it, else our map would be quite large.
420        // Also sort them all.
421        for (&k, v) in unfold_map.iter_mut() {
422            assert_eq!(k, fold(k), "folds should be idempotent");
423            v.push(k);
424            v.sort_unstable();
425        }
426        unfold_map
427    }
