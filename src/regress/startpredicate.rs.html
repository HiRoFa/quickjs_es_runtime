12/// Check if a node is anchored to the start of the line/string.
13/// Returns true if the node begins with a StartOfLine anchor.
14fn is_start_anchored(n: &Node) -> bool {
15    match n {
16        Node::Anchor {
17            anchor_type: ir::AnchorType::StartOfLine,
18            multiline,
19        } => !multiline,
20        Node::Cat(nodes) => {
21            // For concatenation, check if the first node is start-anchored
22            nodes.first().is_some_and(is_start_anchored)
23        }
24        Node::CaptureGroup(child, ..) | Node::NamedCaptureGroup(child, ..) => {
25            is_start_anchored(child)
26        }
27        // Other nodes are not anchored
28        _ => false,
29    }
30}

32/// Convert the code point set to a first-byte bitmap.
33/// That is, make a list of all of the possible first bytes of every contained
34/// code point, and store that in a bitmap.
35fn cps_to_first_byte_bitmap(input: &codepointset::CodePointSet) -> Box<ByteBitmap> {
36    let mut bitmap = Box::<ByteBitmap>::default();
37    for iv in input.intervals() {
38        add_utf8_first_bytes_to_bitmap(*iv, &mut bitmap);
39    }
40    bitmap
41}

43/// The "IR" for a start predicate.
44enum AbstractStartPredicate {
45    /// No predicate.
46    Arbitrary,
47
48    /// Sequence of non-empty bytes.
49    Sequence(Vec<u8>),
50
51    /// Set of bytes.
52    Set(Box<ByteBitmap>),
53}
54
55impl AbstractStartPredicate {
56    /// \return the disjunction of two predicates.
57    /// That is, a predicate that matches x OR y.
58    fn disjunction(x: Self, y: Self) -> Self {
59        match (x, y) {
60            (Self::Arbitrary, _) => Self::Arbitrary,
61            (_, Self::Arbitrary) => Self::Arbitrary,
62
63            (Self::Sequence(s1), Self::Sequence(s2)) => {
64                // Compute the length of the shared prefix.
65                let shared_len = s1.iter().zip(s2.iter()).take_while(|(a, b)| a == b).count();
66                debug_assert!(s1[..shared_len] == s2[..shared_len]);
67                if shared_len > 0 {
68                    // Use the shared prefix.
69                    Self::Sequence(s1[..shared_len].to_vec())
70                } else {
71                    // Use a set of their first byte.
72                    Self::Set(Box::new(ByteBitmap::new(&[s1[0], s2[0]])))
73                }
74            }
75
76            (Self::Set(mut s1), Self::Set(s2)) => {
77                s1.bitor(s2.as_ref());
78                Self::Set(s1)
79            }
80
81            (Self::Set(mut s1), Self::Sequence(s2)) => {
82                // Add first byte to set.
83                s1.set(s2[0]);
84                Self::Set(s1)
85            }
86
87            (Self::Sequence(s1), Self::Set(mut s2)) => {
88                s2.set(s1[0]);
89                Self::Set(s2)
90            }
91        }
92    }
93
94    /// Resolve ourselves to a concrete start predicate.
95    fn resolve_to_insn(self) -> StartPredicate {
96        match self {
97            Self::Arbitrary => StartPredicate::Arbitrary,
98            Self::Sequence(vals) => match vals.len() {
99                0 => StartPredicate::Arbitrary,
100                1 => StartPredicate::ByteSet1([vals[0]]),
101                _ => StartPredicate::ByteSeq(Box::new(memmem::Finder::new(&vals).into_owned())),
102            },
103            Self::Set(bm) => match bm.count_bits() {
104                0 => StartPredicate::Arbitrary,
105                1 => StartPredicate::ByteSet1(bm.as_array()),
106                2 => StartPredicate::ByteSet2(bm.as_array()),
107                3 => StartPredicate::ByteSet3(bm.as_array()),
108                _ => StartPredicate::ByteBracket(*bm),
109            },
110        }
111    }
112}

114/// Compute any start-predicate for a node..
115/// If this returns None, then the instruction is conceptually zero-width (e.g.
116/// lookahead assertion) and does not contribute to the predicate.
117/// If this returns StartPredicate::Arbitrary, then there is no predicate.
118fn compute_start_predicate(n: &Node) -> Option<AbstractStartPredicate> {
119    let arbitrary = Some(AbstractStartPredicate::Arbitrary);
120    match n {
121        Node::ByteSequence(bytevec) => Some(AbstractStartPredicate::Sequence(bytevec.clone())),
122        Node::ByteSet(bytes) => Some(AbstractStartPredicate::Set(Box::new(ByteBitmap::new(
123            bytes,
124        )))),
125
126        Node::Empty => arbitrary,
127        Node::Goal => arbitrary,
128        Node::BackRef(..) => arbitrary,
129
130        Node::CharSet(chars) => {
131            // Pick the first bytes out.
132            let bytes = chars
133                .iter()
134                .map(|&c| utf8_first_byte(c))
135                .collect::<Vec<_>>();
136            Some(AbstractStartPredicate::Set(Box::new(ByteBitmap::new(
137                &bytes,
138            ))))
139        }
140
141        // We assume that most char nodes have been optimized to ByteSeq or AnyBytes2, so skip
142        // these.
143        // TODO: we could support icase through bitmap of de-folded first bytes.
144        Node::Char { .. } => arbitrary,
145
146        // Cats return the first non-None value, if any.
147        Node::Cat(nodes) => nodes.iter().filter_map(compute_start_predicate).next(),
148
149        // MatchAny (aka .) is too common to do a fast prefix search for.
150        Node::MatchAny => arbitrary,
151
152        // MatchAnyExceptLineTerminator (aka .) is too common to do a fast prefix search for.
153        Node::MatchAnyExceptLineTerminator => arbitrary,
154
155        // TODO: can probably exploit some of these.
156        Node::Anchor { .. } => arbitrary,
157        Node::WordBoundary { .. } => arbitrary,
158
159        // Capture groups delegate to their contents.
160        Node::CaptureGroup(child, ..) | Node::NamedCaptureGroup(child, ..) => {
161            compute_start_predicate(child)
162        }
163
164        // Zero-width assertions are one of the few instructions that impose no start predicate.
165        Node::LookaroundAssertion { .. } => None,
166
167        Node::Loop { loopee, quant, .. } => {
168            // TODO: we could try to join two predicates if the loop were optional.
169            if quant.min > 0 {
170                compute_start_predicate(loopee)
171            } else {
172                arbitrary
173            }
174        }
175
176        Node::Loop1CharBody { loopee, quant } => {
177            // TODO: we could try to join two predicates if the loop were optional.
178            if quant.min > 0 {
179                compute_start_predicate(loopee)
180            } else {
181                arbitrary
182            }
183        }
184
185        // This one is interesting - we compute the disjunction of the predicates of our two arms.
186        Node::Alt(left, right) => {
187            if let (Some(x), Some(y)) = (
188                compute_start_predicate(left),
189                compute_start_predicate(right),
190            ) {
191                Some(AbstractStartPredicate::disjunction(x, y))
192            } else {
193                // This indicates that one of our branches could match the empty string.
194                arbitrary
195            }
196        }
197
198        // Brackets get a bitmap.
199        Node::Bracket(bc) => {
200            // If our bracket is inverted, construct the set of code points not contained.
201            let storage;
202            let cps = if bc.invert {
203                storage = bc.cps.inverted();
204                &storage
205            } else {
206                &bc.cps
207            };
208            let bitmap = cps_to_first_byte_bitmap(cps);
209            Some(AbstractStartPredicate::Set(bitmap))
210        }
211    }
212}

214/// \return the start predicate for a Regex.
215pub fn predicate_for_re(re: &ir::Regex) -> StartPredicate {
216    // Check if the regex is anchored to the start - if so, we can optimize
217    // by avoiding string searching entirely. However, only do this when
218    // multiline mode is disabled, since in multiline mode ^ can match
219    // at the beginning of any line, not just the string start.
220    if is_start_anchored(&re.node) && !re.flags.multiline {
221        return StartPredicate::StartAnchored;
222    }
223
224    compute_start_predicate(&re.node)
225        .unwrap_or(AbstractStartPredicate::Arbitrary)
226        .resolve_to_insn()
227}
