26// Each element here is matching the `sources` of the outer SourceMap.

32#[derive(Serialize, Deserialize, PartialEq, Debug)]
33pub struct RawSourceMap {

35    #[serde(skip_serializing_if = "Option::is_none")]
36    pub file: Option<Value>,
37    pub sources: Option<Vec<Option<String>>>,
38    #[serde(rename = "sourceRoot", skip_serializing_if = "Option::is_none")]
39    pub source_root: Option<String>,
40    #[serde(rename = "sourcesContent", skip_serializing_if = "Option::is_none")]
41    pub sources_content: Option<Vec<Option<String>>>,
42    #[serde(skip_serializing_if = "Option::is_none")]
43    pub sections: Option<Vec<RawSection>>,
44    #[serde(skip_serializing_if = "Option::is_none")]
45    pub names: Option<Vec<Value>>,
46    #[serde(rename = "rangeMappings", skip_serializing_if = "Option::is_none")]
47    pub range_mappings: Option<String>,
48    #[serde(skip_serializing_if = "Option::is_none")]
49    pub mappings: Option<String>,
50    #[serde(rename = "ignoreList", skip_serializing_if = "Option::is_none")]
51    pub ignore_list: Option<Vec<u32>>,
52    #[serde(skip_serializing_if = "Option::is_none")]
53    pub x_facebook_offsets: Option<Vec<Option<u32>>>,
54    #[serde(skip_serializing_if = "Option::is_none")]
55    pub x_metro_module_paths: Option<Vec<String>>,
56    #[serde(skip_serializing_if = "Option::is_none")]
57    pub x_facebook_sources: FacebookSources,
58    #[serde(skip_serializing_if = "Option::is_none")]
59    pub debug_id: Option<DebugId>,
60    // This field only exists to be able to deserialize from "debugId" keys

62    #[serde(skip_serializing_if = "Option::is_none", rename = "debugId")]
63    pub(crate) _debug_id_new: Option<DebugId>,
64}
