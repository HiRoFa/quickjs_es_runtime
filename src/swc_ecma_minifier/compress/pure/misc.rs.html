23fn is_definitely_string(expr: &Expr) -> bool {
24    match expr {
25        Expr::Lit(Lit::Str(_)) => true,
26        Expr::Tpl(_) => true,
27        Expr::Bin(BinExpr {
28            op: BinaryOp::Add,
29            left,
30            right,
31            ..
32        }) => is_definitely_string(left) || is_definitely_string(right),
33        Expr::Paren(ParenExpr { expr, .. }) => is_definitely_string(expr),
34        _ => false,
35    }
36}

38/// Check whether we can compress `new RegExp(…)` to `RegExp(…)`. That's sound
39/// unless the first argument is already a RegExp object and the second is
40/// undefined. We check for the common case where we can prove one of the first
41/// two arguments is a string.
42fn can_compress_new_regexp(args: Option<&[ExprOrSpread]>) -> bool {
43    if let Some(args) = args {
44        if let Some(first) = args.first() {
45            if first.spread.is_some() {
46                false
47            } else if is_definitely_string(&first.expr) {
48                true
49            } else if let Some(second) = args.get(1) {
50                second.spread.is_none() && is_definitely_string(&second.expr)
51            } else {
52                false
53            }
54        } else {
55            true
56        }
57    } else {
58        true
59    }
60}

62fn collect_exprs_from_object(obj: &mut ObjectLit) -> Vec<Box<Expr>> {
63    let mut exprs = Vec::new();
64
65    for prop in obj.props.take() {
66        if let PropOrSpread::Prop(p) = prop {
67            match *p {
68                Prop::Shorthand(p) => {
69                    exprs.push(p.into());
70                }
71                Prop::KeyValue(p) => {
72                    if let PropName::Computed(e) = p.key {
73                        exprs.push(e.expr);
74                    }
75
76                    exprs.push(p.value);
77                }
78                Prop::Getter(p) => {
79                    if let PropName::Computed(e) = p.key {
80                        exprs.push(e.expr);
81                    }
82                }
83                Prop::Setter(p) => {
84                    if let PropName::Computed(e) = p.key {
85                        exprs.push(e.expr);
86                    }
87                }
88                Prop::Method(p) => {
89                    if let PropName::Computed(e) = p.key {
90                        exprs.push(e.expr);
91                    }
92                }
93                _ => {}
94            }
95        }
96    }
97
98    exprs
99}

109    pub(super) fn compress_bin_assignment_to_left(&mut self, e: &mut AssignExpr) {
110        if e.op != op!("=") {
111            return;
112        }
113
114        // TODO: Handle pure properties.
115        let lhs = match &e.left {
116            AssignTarget::Simple(SimpleAssignTarget::Ident(i)) => i,
117            _ => return,
118        };
119
120        // If left operand of a binary expression is not same as lhs, this method has
121        // nothing to do.
122        let (op, right) = match &mut *e.right {
123            Expr::Bin(BinExpr {
124                left, op, right, ..
125            }) => match &**left {
126                Expr::Ident(r) if lhs.sym == r.sym && lhs.ctxt == r.ctxt => (op, right),
127                _ => return,
128            },
129            _ => return,
130        };
131
132        // Don't break code for old browsers.
133        match op {
134            BinaryOp::LogicalOr => return,
135            BinaryOp::LogicalAnd => return,
136            BinaryOp::Exp => return,
137            BinaryOp::NullishCoalescing => return,
138            _ => {}
139        }
140
141        let op = match op {
142            BinaryOp::In | BinaryOp::InstanceOf => return,
143
144            BinaryOp::EqEq | BinaryOp::NotEq | BinaryOp::EqEqEq | BinaryOp::NotEqEq => {
145                // TODO(kdy1): Check if this is optimizable.
146                return;
147            }
148
149            BinaryOp::Lt | BinaryOp::LtEq | BinaryOp::Gt | BinaryOp::GtEq => return,
150
151            BinaryOp::LShift => op!("<<="),
152            BinaryOp::RShift => {
153                op!(">>=")
154            }
155            BinaryOp::ZeroFillRShift => {
156                op!(">>>=")
157            }
158            BinaryOp::Add => {
159                op!("+=")
160            }
161            BinaryOp::Sub => {
162                op!("-=")
163            }
164            BinaryOp::Mul => {
165                op!("*=")
166            }
167            BinaryOp::Div => {
168                op!("/=")
169            }
170            BinaryOp::Mod => {
171                op!("%=")
172            }
173            BinaryOp::BitOr => {
174                op!("|=")
175            }
176            BinaryOp::BitXor => {
177                op!("^=")
178            }
179            BinaryOp::BitAnd => {
180                op!("&=")
181            }
182            BinaryOp::LogicalOr => {
183                op!("||=")
184            }
185            BinaryOp::LogicalAnd => {
186                op!("&&=")
187            }
188            BinaryOp::Exp => {
189                op!("**=")
190            }
191            BinaryOp::NullishCoalescing => {
192                op!("??=")
193            }
194        };
195
196        e.op = op;
197        e.right = right.take();
198        // Now we can compress it to an assignment
199    }

201    /// This method does
202    ///
203    /// - `x *= 3` => `x = 3 * x`
204    /// - `x = 3 | x` `x |= 3`
205    /// - `x = 3 & x` => `x &= 3;`
206    /// - `x ^= 3` => `x = 3 ^ x`
207    pub(super) fn compress_bin_assignment_to_right(&mut self, e: &mut AssignExpr) {
208        if e.op != op!("=") {
209            return;
210        }
211
212        // TODO: Handle pure properties.
213        let lhs = match &e.left {
214            AssignTarget::Simple(SimpleAssignTarget::Ident(i)) => i,
215            _ => return,
216        };
217
218        let (op, left) = match &mut *e.right {
219            Expr::Bin(BinExpr {
220                left, op, right, ..
221            }) => match &**right {
222                Expr::Ident(r) if lhs.sym == r.sym && lhs.ctxt == r.ctxt => {
223                    // We need this check because a function call like below can change value of
224                    // operand.
225                    //
226                    // x = g() * x;
227
228                    match &**left {
229                        Expr::This(..) | Expr::Ident(..) | Expr::Lit(..) => {}
230                        _ => return,
231                    }
232
233                    (op, left)
234                }
235                _ => return,
236            },
237            _ => return,
238        };
239
240        let op = match op {
241            BinaryOp::Mul => {
242                op!("*=")
243            }
244            BinaryOp::BitOr => {
245                op!("|=")
246            }
247            BinaryOp::BitXor => {
248                op!("^=")
249            }
250            BinaryOp::BitAnd => {
251                op!("&=")
252            }
253            _ => return,
254        };
255
256        report_change!("Compressing: `e = 3 & e` => `e &= 3`");
257
258        self.changed = true;
259        e.op = op;
260        e.right = left.take();
261    }

263    pub(super) fn eval_spread_object(&mut self, e: &mut ObjectLit) {
264        fn should_skip(p: &PropOrSpread, expr_ctx: ExprCtx) -> bool {
265            match p {
266                PropOrSpread::Prop(p) => match &**p {
267                    Prop::KeyValue(KeyValueProp { key, value, .. }) => {
268                        key.is_computed() || value.may_have_side_effects(expr_ctx)
269                    }
270                    Prop::Assign(AssignProp { value, .. }) => value.may_have_side_effects(expr_ctx),
271                    Prop::Method(method) => method.key.is_computed(),
272
273                    Prop::Getter(..) | Prop::Setter(..) => true,
274
275                    _ => false,
276                },
277
278                PropOrSpread::Spread(SpreadElement { expr, .. }) => match &**expr {
279                    Expr::Object(ObjectLit { props, .. }) => {
280                        props.iter().any(|p| should_skip(p, expr_ctx))
281                    }
282                    _ => false,
283                },
284            }
285        }
286
287        if e.props.iter().any(|p| should_skip(p, self.expr_ctx))
288            || !e.props.iter().any(|p| match p {
289                PropOrSpread::Spread(SpreadElement { expr, .. }) => {
290                    expr.is_object() || expr.is_null()
291                }
292                _ => false,
293            })
294        {
295            return;
296        }
297
298        let mut new_props = Vec::with_capacity(e.props.len());
299
300        for prop in e.props.take() {
301            match prop {
302                PropOrSpread::Spread(SpreadElement { expr, .. })
303                    if expr.is_object() || expr.is_null() =>
304                {
305                    match *expr {
306                        Expr::Object(ObjectLit { props, .. }) => {
307                            for p in props {
308                                new_props.push(p);
309                            }
310                        }
311
312                        Expr::Lit(Lit::Null(_)) => {}
313
314                        _ => {}
315                    }
316                }
317
318                _ => {
319                    new_props.push(prop);
320                }
321            }
322        }
323
324        e.props = new_props;
325    }

329        if !args

333            return;
334        }

344                        for elem in elems {

363                    }
364                },
365                arg => new_args.push(arg),
366            }
367        }
368
369        self.changed = true;
370        report_change!("Compressing spread array");
371
372        *args = new_args;

426    pub(super) fn remove_invalid(&mut self, e: &mut Expr) {
427        match e {
428            Expr::Seq(seq) => {
429                for e in &mut seq.exprs {
430                    self.remove_invalid(e);
431                }
432
433                if seq.exprs.len() == 1 {
434                    *e = *seq.exprs.pop().unwrap();
435                }
436            }
437
438            Expr::Bin(BinExpr { left, right, .. }) => {
439                self.remove_invalid(left);
440                self.remove_invalid(right);
441
442                if left.is_invalid() {
443                    *e = *right.take();
444                    self.remove_invalid(e);
445                } else if right.is_invalid() {
446                    *e = *left.take();
447                    self.remove_invalid(e);
448                }
449            }
450
451            _ => {}
452        }
453    }

455    pub(super) fn compress_array_join(&mut self, e: &mut Expr) {
456        let call = match e {
457            Expr::Call(e) => e,
458            _ => return,
459        };
460
461        let callee = match &mut call.callee {
462            Callee::Super(_) | Callee::Import(_) => return,
463            Callee::Expr(callee) => &mut **callee,
464        };
465
466        let separator = if call.args.is_empty() {
467            atom!(",")
468        } else if call.args.len() == 1 {
469            if call.args[0].spread.is_some() {
470                return;
471            }
472
473            if is_pure_undefined(self.expr_ctx, &call.args[0].expr) {
474                atom!(",")
475            } else {
476                match &*call.args[0].expr {
477                    Expr::Lit(Lit::Str(s)) => s.value.clone(),
478                    Expr::Lit(Lit::Null(..)) => atom!("null"),
479                    _ => return,
480                }
481            }
482        } else {
483            return;
484        };
485
486        let arr = match callee {
487            Expr::Member(MemberExpr {
488                obj,
489                prop: MemberProp::Ident(IdentName { sym, .. }),
490                ..
491            }) if *sym == *"join" => {
492                if let Expr::Array(arr) = &mut **obj {
493                    arr
494                } else {
495                    return;
496                }
497            }
498            _ => return,
499        };
500
501        if arr.elems.iter().any(|elem| {
502            matches!(
503                elem,
504                Some(ExprOrSpread {
505                    spread: Some(..),
506                    ..
507                })
508            )
509        }) {
510            return;
511        }
512
513        // Handle empty array case first
514        if arr.elems.is_empty() {
515            report_change!("Compressing empty array.join()");
516            self.changed = true;
517            *e = Lit::Str(Str {
518                span: call.span,
519                raw: None,
520                value: atom!(""),
521            })
522            .into();
523            return;
524        }
525
526        let cannot_join_as_str_lit = arr
527            .elems
528            .iter()
529            .filter_map(|v| v.as_ref())
530            .any(|v| match &*v.expr {
531                e if is_pure_undefined(self.expr_ctx, e) => false,
532                Expr::Lit(lit) => !matches!(lit, Lit::Str(..) | Lit::Num(..) | Lit::Null(..)),
533                _ => true,
534            });
535
536        if cannot_join_as_str_lit {
537            if let Some(new_expr) =
538                self.compress_array_join_as_tpl(arr.span, &mut arr.elems, &separator)
539            {
540                self.changed = true;
541                *e = new_expr;
542                return;
543            }
544
545            // Try partial optimization (grouping consecutive literals)
546            if let Some(new_expr) =
547                self.compress_array_join_partial(arr.span, &mut arr.elems, &separator)
548            {
549                self.changed = true;
550                report_change!("Compressing array.join() with partial optimization");
551                *e = new_expr;
552                return;
553            }
554
555            if !self.options.unsafe_passes {
556                return;
557            }
558
559            if arr
560                .elems
561                .iter()
562                .filter_map(|v| v.as_ref())
563                .any(|v| match &*v.expr {
564                    e if is_pure_undefined(self.expr_ctx, e) => false,
565                    Expr::Lit(lit) => !matches!(lit, Lit::Str(..) | Lit::Num(..) | Lit::Null(..)),
566                    // All other expressions can potentially be null/undefined
567                    _ => true,
568                })
569            {
570                return;
571            }
572
573            let sep: Box<Expr> = Lit::Str(Str {
574                span: DUMMY_SP,
575                raw: None,
576                value: separator,
577            })
578            .into();
579            let mut res = Lit::Str(Str {
580                span: DUMMY_SP,
581                raw: None,
582                value: atom!(""),
583            })
584            .into();
585
586            fn add(to: &mut Expr, right: Box<Expr>) {
587                let lhs = to.take();
588                *to = BinExpr {
589                    span: DUMMY_SP,
590                    left: Box::new(lhs),
591                    op: op!(bin, "+"),
592                    right,
593                }
594                .into();
595            }
596
597            for (last, elem) in arr.elems.take().into_iter().identify_last() {
598                if let Some(ExprOrSpread { spread: None, expr }) = elem {
599                    match &*expr {
600                        e if is_pure_undefined(self.expr_ctx, e) => {
601                            // null and undefined should become empty strings in
602                            // join
603                            // Don't add anything for empty string join
604                        }
605                        Expr::Lit(Lit::Null(..)) => {
606                            // null and undefined should become empty strings in
607                            // join
608                            // Don't add anything for empty string join
609                        }
610                        _ => {
611                            add(&mut res, expr);
612                        }
613                    }
614                }
615
616                if !last {
617                    add(&mut res, sep.clone());
618                }
619            }
620
621            *e = res;
622
623            return;
624        }
625
626        let mut res = String::new();
627        for (last, elem) in arr.elems.iter().identify_last() {
628            if let Some(elem) = elem {
629                debug_assert_eq!(elem.spread, None);
630
631                match &*elem.expr {
632                    Expr::Lit(Lit::Str(s)) => {
633                        res.push_str(&s.value);
634                    }
635                    Expr::Lit(Lit::Num(n)) => {
636                        write!(res, "{}", n.value).unwrap();
637                    }
638                    e if is_pure_undefined(self.expr_ctx, e) => {}
639                    Expr::Lit(Lit::Null(..)) => {}
640                    _ => {
641                        unreachable!(
642                            "Expression {:#?} cannot be joined and it should be filtered out",
643                            elem.expr
644                        )
645                    }
646                }
647            }
648
649            if !last {
650                res.push_str(&separator);
651            }
652        }
653
654        report_change!("Compressing array.join()");
655
656        self.changed = true;
657        *e = Lit::Str(Str {
658            span: call.span,
659            raw: None,
660            value: res.into(),
661        })
662        .into()
663    }

665    /// Performs partial optimization on array.join() when there are mixed
666    /// literals and expressions. Groups consecutive literals into string
667    /// concatenations.
668    fn compress_array_join_partial(
669        &mut self,
670        _span: Span,
671        elems: &mut Vec<Option<ExprOrSpread>>,
672        separator: &str,
673    ) -> Option<Expr> {
674        if !self.options.evaluate {
675            return None;
676        }
677
678        // Check if we have any non-literal elements
679        let has_non_literals = elems.iter().flatten().any(|elem| match &*elem.expr {
680            Expr::Lit(Lit::Str(..) | Lit::Num(..) | Lit::Null(..)) => false,
681            e if is_pure_undefined(self.expr_ctx, e) => false,
682            _ => true,
683        });
684
685        if !has_non_literals {
686            return None; // Pure literal case will be handled elsewhere
687        }
688
689        // For non-empty separators, only optimize if we have at least 2 consecutive
690        // literals This prevents infinite loop and ensures meaningful
691        // optimization
692        if !separator.is_empty() {
693            let mut consecutive_literals = 0;
694            let mut max_consecutive = 0;
695
696            for elem in elems.iter().flatten() {
697                let is_literal = match &*elem.expr {
698                    Expr::Lit(Lit::Str(..) | Lit::Num(..) | Lit::Null(..)) => true,
699                    e if is_pure_undefined(self.expr_ctx, e) => true,
700                    _ => false,
701                };
702
703                if is_literal {
704                    consecutive_literals += 1;
705                    max_consecutive = max_consecutive.max(consecutive_literals);
706                } else {
707                    consecutive_literals = 0;
708                }
709            }
710
711            if max_consecutive < 2 {
712                return None;
713            }
714
715            // Only optimize for single-character separators to avoid bloating the code
716            // Long separators like "really-long-separator" should not be optimized
717            if separator.len() > 1 {
718                return None;
719            }
720
721            // For comma separator, require a higher threshold to avoid infinite loops
722            if separator == "," && max_consecutive < 6 {
723                return None;
724            }
725        } else {
726            // For empty string joins, optimize more aggressively since we're
727            // doing string concatenation We can always optimize
728            // these as long as there are mixed expressions and literals
729        }
730
731        // Group consecutive literals and create a string concatenation expression
732        let mut groups = Vec::new();
733        let mut current_group = Vec::new();
734
735        for elem in elems.iter().flatten() {
736            let is_literal = match &*elem.expr {
737                Expr::Lit(Lit::Str(..) | Lit::Num(..) | Lit::Null(..)) => true,
738                e if is_pure_undefined(self.expr_ctx, e) => true,
739                _ => false,
740            };
741
742            if is_literal {
743                current_group.push(elem);
744            } else {
745                if !current_group.is_empty() {
746                    groups.push(GroupType::Literals(current_group));
747                    current_group = Vec::new();
748                }
749                groups.push(GroupType::Expression(elem));
750            }
751        }
752
753        if !current_group.is_empty() {
754            groups.push(GroupType::Literals(current_group));
755        }
756
757        // If we don't have any grouped literals, no optimization possible
758        if groups.iter().all(|g| matches!(g, GroupType::Expression(_))) {
759            return None;
760        }
761
762        // Handle different separators
763        let is_string_concat = separator.is_empty();
764
765        if is_string_concat {
766            // Convert to string concatenation
767            let mut result_parts = Vec::new();
768
769            // Only add empty string prefix when the first element is a non-string
770            // expression that needs coercion to string AND there's no string
771            // literal early enough to provide coercion
772            let needs_empty_string_prefix = match groups.first() {
773                Some(GroupType::Expression(first_expr)) => {
774                    // Check if the first expression is already a string concatenation
775                    let first_needs_coercion = match &*first_expr.expr {
776                        Expr::Bin(BinExpr {
777                            op: op!(bin, "+"), ..
778                        }) => false, // Already string concat
779                        Expr::Lit(Lit::Str(..)) => false, // Already a string literal
780                        Expr::Call(_call) => {
781                            // Function calls may return any type and need string coercion
782                            true
783                        }
784                        _ => true, // Other expressions need string coercion
785                    };
786
787                    // If the first element needs coercion, check if the second element is a string
788                    // literal that can provide the coercion
789                    if first_needs_coercion {
790                        match groups.get(1) {
791                            Some(GroupType::Literals(_)) => false, /* String literals will */
792                            // provide coercion
793                            _ => true, // No string literal to provide coercion
794                        }
795                    } else {
796                        false
797                    }
798                }
799                _ => false,
800            };
801
802            if needs_empty_string_prefix {
803                result_parts.push(Box::new(Expr::Lit(Lit::Str(Str {
804                    span: DUMMY_SP,
805                    raw: None,
806                    value: atom!(""),
807                }))));
808            }
809
810            for group in groups {
811                match group {
812                    GroupType::Literals(literals) => {
813                        let mut joined = String::new();
814                        for literal in literals.iter() {
815                            match &*literal.expr {
816                                Expr::Lit(Lit::Str(s)) => joined.push_str(&s.value),
817                                Expr::Lit(Lit::Num(n)) => write!(joined, "{}", n.value).unwrap(),
818                                Expr::Lit(Lit::Null(..)) => {
819                                    // For string concatenation, null becomes
820                                    // empty string
821                                }
822                                e if is_pure_undefined(self.expr_ctx, e) => {
823                                    // undefined becomes empty string in string
824                                    // context
825                                }
826                                _ => unreachable!(),
827                            }
828                        }
829
830                        result_parts.push(Box::new(Expr::Lit(Lit::Str(Str {
831                            span: DUMMY_SP,
832                            raw: None,
833                            value: joined.into(),
834                        }))));
835                    }
836                    GroupType::Expression(expr) => {
837                        result_parts.push(expr.expr.clone());
838                    }
839                }
840            }
841
842            // Create string concatenation expression
843            if result_parts.len() == 1 {
844                return Some(*result_parts.into_iter().next().unwrap());
845            }
846
847            let mut result = *result_parts.remove(0);
848            for part in result_parts {
849                result = Expr::Bin(BinExpr {
850                    span: DUMMY_SP,
851                    left: Box::new(result),
852                    op: op!(bin, "+"),
853                    right: part,
854                });
855            }
856
857            Some(result)
858        } else {
859            // For non-empty separator, create a more compact array
860            let mut new_elems = Vec::new();
861
862            for group in groups {
863                match group {
864                    GroupType::Literals(literals) => {
865                        let mut joined = String::new();
866                        for (idx, literal) in literals.iter().enumerate() {
867                            if idx > 0 {
868                                joined.push_str(separator);
869                            }
870
871                            match &*literal.expr {
872                                Expr::Lit(Lit::Str(s)) => joined.push_str(&s.value),
873                                Expr::Lit(Lit::Num(n)) => write!(joined, "{}", n.value).unwrap(),
874                                Expr::Lit(Lit::Null(..)) => {
875                                    // null becomes empty string
876                                }
877                                e if is_pure_undefined(self.expr_ctx, e) => {
878                                    // undefined becomes empty string
879                                }
880                                _ => unreachable!(),
881                            }
882                        }
883
884                        new_elems.push(Some(ExprOrSpread {
885                            spread: None,
886                            expr: Box::new(Expr::Lit(Lit::Str(Str {
887                                span: DUMMY_SP,
888                                raw: None,
889                                value: joined.into(),
890                            }))),
891                        }));
892                    }
893                    GroupType::Expression(expr) => {
894                        new_elems.push(Some(ExprOrSpread {
895                            spread: None,
896                            expr: expr.expr.clone(),
897                        }));
898                    }
899                }
900            }
901
902            // Create a new array.join() call with the original separator
903            let new_array = Expr::Array(ArrayLit {
904                span: _span,
905                elems: new_elems,
906            });
907
908            // For comma separator, use .join() without arguments (shorter)
909            let args = if separator == "," {
910                vec![]
911            } else {
912                vec![ExprOrSpread {
913                    spread: None,
914                    expr: Box::new(Expr::Lit(Lit::Str(Str {
915                        span: DUMMY_SP,
916                        raw: None,
917                        value: separator.into(),
918                    }))),
919                }]
920            };
921
922            Some(Expr::Call(CallExpr {
923                span: _span,
924                ctxt: Default::default(),
925                callee: Callee::Expr(Box::new(Expr::Member(MemberExpr {
926                    span: _span,
927                    obj: Box::new(new_array),
928                    prop: MemberProp::Ident(IdentName::new(atom!("join"), _span)),
929                }))),
930                args,
931                ..Default::default()
932            }))
933        }
934    }

958    /// `new RegExp("([Sap]+)", "ig")` => `/([Sap]+)/gi`
959    fn optimize_regex(&mut self, args: &mut Vec<ExprOrSpread>, span: &mut Span) -> Option<Expr> {
960        fn valid_pattern(pattern: &Expr) -> Option<Atom> {
961            if let Expr::Lit(Lit::Str(s)) = pattern {
962                if s.value.contains(|c: char| {
963                    // whitelist
964                    !c.is_ascii_alphanumeric()
965                        && !matches!(c, '$' | '[' | ']' | '(' | ')' | '{' | '}' | '-' | '+' | '_')
966                }) {
967                    None
968                } else {
969                    Some(s.value.clone())
970                }
971            } else {
972                None
973            }
974        }
975        fn valid_flag(flag: &Expr, es_version: EsVersion) -> Option<Atom> {
976            if let Expr::Lit(Lit::Str(s)) = flag {
977                let mut set = FxHashSet::default();
978                for c in s.value.chars() {
979                    if !(matches!(c, 'g' | 'i' | 'm')
980                        || (es_version >= EsVersion::Es2015 && matches!(c, 'u' | 'y'))
981                        || (es_version >= EsVersion::Es2018 && matches!(c, 's')))
982                        || (es_version >= EsVersion::Es2022 && matches!(c, 'd'))
983                    {
984                        return None;
985                    }
986
987                    if !set.insert(c) {
988                        return None;
989                    }
990                }
991
992                Some(s.value.clone())
993            } else {
994                None
995            }
996        }
997
998        let (pattern, flag) = match args.as_slice() {
999            [ExprOrSpread { spread: None, expr }] => (valid_pattern(expr)?, atom!("")),
1000            [ExprOrSpread {
1001                spread: None,
1002                expr: pattern,
1003            }, ExprOrSpread {
1004                spread: None,
1005                expr: flag,
1006            }] => (
1007                valid_pattern(pattern)?,
1008                valid_flag(flag, self.options.ecma)?,
1009            ),
1010            _ => return None,
1011        };
1012
1013        if pattern.is_empty() {
1014            // For some expressions `RegExp()` and `RegExp("")`
1015            // Theoretically we can use `/(?:)/` to achieve shorter code
1016            // But some browsers released in 2015 don't support them yet.
1017            return None;
1018        }
1019
1020        report_change!("Optimized regex");
1021
1022        Some(
1023            Lit::Regex(Regex {
1024                span: *span,
1025                exp: pattern,
1026                flags: {
1027                    let flag = flag.to_string();
1028                    let mut bytes = flag.into_bytes();
1029                    bytes.sort_unstable();
1030
1031                    String::from_utf8(bytes).unwrap().into()
1032                },
1033            })
1034            .into(),
1035        )
1036    }

1347    fn compress_array_join_as_tpl(
1348        &mut self,
1349        span: Span,
1350        elems: &mut Vec<Option<ExprOrSpread>>,
1351        sep: &str,
1352    ) -> Option<Expr> {
1353        if !self.options.evaluate {
1354            return None;
1355        }
1356
1357        if elems.iter().flatten().any(|elem| match &*elem.expr {
1358            Expr::Tpl(t) => t.quasis.iter().any(|q| q.cooked.is_none()),
1359            Expr::Lit(Lit::Str(..)) => false,
1360            _ => true,
1361        }) {
1362            return None;
1363        }
1364
1365        self.changed = true;
1366        report_change!("Compressing array.join() as template literal");
1367
1368        let mut new_tpl = Tpl {
1369            span,
1370            quasis: Vec::new(),
1371            exprs: Vec::new(),
1372        };
1373        let mut cur_raw = String::new();
1374        let mut cur_cooked = String::new();
1375        let mut first = true;
1376
1377        for elem in elems.take().into_iter().flatten() {
1378            if first {
1379                first = false;
1380            } else {
1381                cur_raw.push_str(sep);
1382                cur_cooked.push_str(sep);
1383            }
1384
1385            match *elem.expr {
1386                Expr::Tpl(mut tpl) => {
1387                    //
1388                    for idx in 0..(tpl.quasis.len() + tpl.exprs.len()) {
1389                        if idx % 2 == 0 {
1390                            // quasis
1391                            let e = tpl.quasis[idx / 2].take();
1392
1393                            cur_cooked.push_str(&e.cooked.unwrap());
1394                            cur_raw.push_str(&e.raw);
1395                        } else {
1396                            new_tpl.quasis.push(TplElement {
1397                                span: DUMMY_SP,
1398                                tail: false,
1399                                cooked: Some((&*cur_cooked).into()),
1400                                raw: (&*cur_raw).into(),
1401                            });
1402
1403                            cur_raw.clear();
1404                            cur_cooked.clear();
1405
1406                            let e = tpl.exprs[idx / 2].take();
1407
1408                            new_tpl.exprs.push(e);
1409                        }
1410                    }
1411                }
1412                Expr::Lit(Lit::Str(s)) => {
1413                    cur_cooked.push_str(&convert_str_value_to_tpl_cooked(&s.value));
1414                    cur_raw.push_str(&convert_str_value_to_tpl_raw(&s.value));
1415                }
1416                _ => {
1417                    unreachable!()
1418                }
1419            }
1420        }
1421
1422        new_tpl.quasis.push(TplElement {
1423            span: DUMMY_SP,
1424            tail: false,
1425            cooked: Some(cur_cooked.into()),
1426            raw: cur_raw.into(),
1427        });
1428
1429        Some(new_tpl.into())
1430    }

1487        span: Span,
1488        exprs: impl Iterator<Item = Box<Expr>>,
1489    ) -> Option<Expr> {
1490        let mut exprs = exprs
1491            .filter_map(|mut e| {
1492                self.ignore_return_value(
1493                    &mut e,
1494                    DropOpts::DROP_GLOBAL_REFS_IF_UNUSED
1495                        .union(DropOpts::DROP_NUMBER)
1496                        .union(DropOpts::DROP_STR_LIT),
1497                );
1498
1499                if let Expr::Invalid(..) = &*e {
1500                    None
1501                } else {
1502                    Some(e)
1503                }
1504            })
1505            .collect::<Vec<_>>();
1506
1507        if exprs.is_empty() {
1508            return None;
1509        }
1510        if exprs.len() == 1 {
1511            let mut new = *exprs.remove(0);
1512            new.set_span(span);
1513            return Some(new);
1514        }
1515
1516        Some(
1517            SeqExpr {
1518                span: DUMMY_SP,
1519                exprs,
1520            }
1521            .into(),
1522        )
1523    }

1525    /// Calls [`Self::ignore_return_value`] on the arguments of return
1526    /// statemetns.
1527    ///
1528    /// This function is recursive but does not go into nested scopes.
1529    pub(super) fn ignore_return_value_of_return_stmt(&mut self, s: &mut Stmt, opts: DropOpts) {
1530        match s {
1531            Stmt::Return(s) => {
1532                if let Some(arg) = &mut s.arg {
1533                    self.ignore_return_value(arg, opts);
1534                    if arg.is_invalid() {
1535                        report_change!(
1536                            "Dropped the argument of a return statement because the return value \
1537                             is ignored"
1538                        );
1539                        s.arg = None;
1540                    }
1541                }
1542            }
1543
1544            Stmt::Block(s) => {
1545                for stmt in &mut s.stmts {
1546                    self.ignore_return_value_of_return_stmt(stmt, opts);
1547                }
1548            }
1549
1550            Stmt::If(s) => {
1551                self.ignore_return_value_of_return_stmt(&mut s.cons, opts);
1552                if let Some(alt) = &mut s.alt {
1553                    self.ignore_return_value_of_return_stmt(alt, opts);
1554                }
1555            }
1556
1557            Stmt::Switch(s) => {
1558                for case in &mut s.cases {
1559                    for stmt in &mut case.cons {
1560                        self.ignore_return_value_of_return_stmt(stmt, opts);
1561                    }
1562                }
1563            }
1564
1565            _ => {}
1566        }
1567    }

1569    pub(super) fn ignore_return_value(&mut self, e: &mut Expr, opts: DropOpts) {
1570        if e.is_invalid() {
1571            return;
1572        }
1573
1574        if self.ctx.contains(Ctx::IN_DELETE) {
1575            return;
1576        }
1577
1578        debug_assert_valid(e);
1579
1580        self.optimize_expr_in_bool_ctx(e, true);
1581
1582        match e {
1583            Expr::Seq(seq) => {
1584                if let Some(last) = seq.exprs.last_mut() {
1585                    self.ignore_return_value(last, opts);
1586                }
1587
1588                seq.exprs.retain(|expr| !expr.is_invalid());
1589
1590                if seq.exprs.is_empty() {
1591                    e.take();
1592                    return;
1593                }
1594                if seq.exprs.len() == 1 {
1595                    *e = *seq.exprs.remove(0);
1596                }
1597                return;
1598            }
1599
1600            Expr::Call(CallExpr {
1601                span, ctxt, args, ..
1602            }) if ctxt.has_mark(self.marks.pure) => {
1603                report_change!("ignore_return_value: Dropping a pure call");
1604                self.changed = true;
1605
1606                let new =
1607                    self.make_ignored_expr(*span, args.take().into_iter().map(|arg| arg.expr));
1608
1609                *e = new.unwrap_or(Invalid { span: DUMMY_SP }.into());
1610                return;
1611            }
1612
1613            Expr::Call(CallExpr {
1614                span,
1615                callee: Callee::Expr(callee),
1616                args,
1617                ..
1618            }) if callee.is_pure_callee(self.expr_ctx) => {
1619                report_change!("ignore_return_value: Dropping a pure call (callee is pure)");
1620                self.changed = true;
1621
1622                let new =
1623                    self.make_ignored_expr(*span, args.take().into_iter().map(|arg| arg.expr));
1624
1625                *e = new.unwrap_or(Invalid { span: DUMMY_SP }.into());
1626                return;
1627            }
1628
1629            Expr::TaggedTpl(TaggedTpl {
1630                span, ctxt, tpl, ..
1631            }) if ctxt.has_mark(self.marks.pure) => {
1632                report_change!("ignore_return_value: Dropping a pure call");
1633                self.changed = true;
1634
1635                let new = self.make_ignored_expr(*span, tpl.exprs.take().into_iter());
1636
1637                *e = new.unwrap_or(Invalid { span: DUMMY_SP }.into());
1638                return;
1639            }
1640
1641            Expr::New(NewExpr {
1642                callee,
1643                span,
1644                ctxt,
1645                args,
1646                ..
1647            }) if callee.is_pure_callee(self.expr_ctx) || ctxt.has_mark(self.marks.pure) => {
1648                report_change!("ignore_return_value: Dropping a pure call");
1649                self.changed = true;
1650
1651                let new = self.make_ignored_expr(
1652                    *span,
1653                    args.take().into_iter().flatten().map(|arg| arg.expr),
1654                );
1655
1656                *e = new.unwrap_or(Invalid { span: DUMMY_SP }.into());
1657                return;
1658            }
1659
1660            _ => {}
1661        }
1662
1663        match e {
1664            Expr::Call(CallExpr {
1665                span,
1666                callee: Callee::Expr(callee),
1667                args,
1668                ..
1669            }) => {
1670                if callee.is_pure_callee(self.expr_ctx) {
1671                    self.changed = true;
1672                    report_change!("Dropping pure call as callee is pure");
1673                    *e = self
1674                        .make_ignored_expr(*span, args.take().into_iter().map(|arg| arg.expr))
1675                        .unwrap_or(Invalid { span: DUMMY_SP }.into());
1676                    return;
1677                }
1678            }
1679
1680            Expr::TaggedTpl(TaggedTpl {
1681                span,
1682                tag: callee,
1683                tpl,
1684                ..
1685            }) => {
1686                if callee.is_pure_callee(self.expr_ctx) {
1687                    self.changed = true;
1688                    report_change!("Dropping pure tag tpl as callee is pure");
1689                    *e = self
1690                        .make_ignored_expr(*span, tpl.exprs.take().into_iter())
1691                        .unwrap_or(Invalid { span: DUMMY_SP }.into());
1692                    return;
1693                }
1694            }
1695            _ => (),
1696        }
1697
1698        if self.options.conditionals || self.options.expr {
1699            if let Expr::Cond(CondExpr {
1700                span,
1701                test,
1702                cons,
1703                alt,
1704                ..
1705            }) = e
1706            {
1707                self.ignore_return_value(cons, opts);
1708                self.ignore_return_value(alt, opts);
1709
1710                if cons.is_invalid() && alt.is_invalid() {
1711                    report_change!("Dropping a conditional expression");
1712                    *e = *test.take();
1713                    self.changed = true;
1714                    return;
1715                }
1716
1717                if cons.is_invalid() {
1718                    *e = Expr::Bin(BinExpr {
1719                        span: *span,
1720                        op: op!("||"),
1721                        left: test.take(),
1722                        right: alt.take(),
1723                    });
1724                    report_change!("Dropping the `then` branch of a conditional expression");
1725                    self.changed = true;
1726                    return;
1727                }
1728
1729                if alt.is_invalid() {
1730                    *e = Expr::Bin(BinExpr {
1731                        span: *span,
1732                        op: op!("&&"),
1733                        left: test.take(),
1734                        right: cons.take(),
1735                    });
1736                    report_change!("Dropping the `else` branch of a conditional expression");
1737                    self.changed = true;
1738                    return;
1739                }
1740            }
1741        }
1742
1743        if opts.contains(DropOpts::DROP_NUMBER) {
1744            if let Expr::Lit(Lit::Num(n)) = e {
1745                // Skip 0
1746                if n.value != 0.0 && n.value.classify() == FpCategory::Normal {
1747                    report_change!("Dropping a number");
1748                    *e = Invalid { span: DUMMY_SP }.into();
1749                    return;
1750                }
1751            }
1752        }
1753
1754        if let Expr::Ident(i) = e {
1755            // If it's not a top level, it's a reference to a declared variable.
1756            if i.ctxt.outer() == self.marks.unresolved_mark {
1757                if self.options.side_effects
1758                    || (self.options.unused && opts.contains(DropOpts::DROP_GLOBAL_REFS_IF_UNUSED))
1759                {
1760                    if is_global_var_with_pure_property_access(&i.sym) {
1761                        report_change!("Dropping a reference to a global variable");
1762                        *e = Invalid { span: DUMMY_SP }.into();
1763                        return;
1764                    }
1765                }
1766            } else {
1767                report_change!("Dropping an identifier as it's declared");
1768                *e = Invalid { span: DUMMY_SP }.into();
1769                return;
1770            }
1771        }
1772
1773        if self.options.side_effects
1774            || self.options.dead_code
1775            || self.options.collapse_vars
1776            || self.options.expr
1777        {
1778            match e {
1779                Expr::Unary(UnaryExpr {
1780                    op: op!("!"), arg, ..
1781                }) => {
1782                    self.ignore_return_value(
1783                        arg,
1784                        DropOpts::DROP_GLOBAL_REFS_IF_UNUSED
1785                            .union(DropOpts::DROP_NUMBER)
1786                            .union(DropOpts::DROP_STR_LIT),
1787                    );
1788
1789                    if arg.is_invalid() {
1790                        report_change!("Dropping an unary expression");
1791                        *e = Invalid { span: DUMMY_SP }.into();
1792                        return;
1793                    }
1794                }
1795
1796                Expr::Unary(UnaryExpr {
1797                    op: op!("void") | op!(unary, "+") | op!(unary, "-") | op!("~"),
1798                    arg,
1799                    ..
1800                }) => {
1801                    self.ignore_return_value(
1802                        arg,
1803                        DropOpts::DROP_GLOBAL_REFS_IF_UNUSED
1804                            .union(DropOpts::DROP_NUMBER)
1805                            .union(DropOpts::DROP_STR_LIT),
1806                    );
1807
1808                    if arg.is_invalid() {
1809                        report_change!("Dropping an unary expression");
1810                        *e = Invalid { span: DUMMY_SP }.into();
1811                        return;
1812                    }
1813
1814                    report_change!("Dropping an unary operator");
1815                    *e = *arg.take();
1816                    return;
1817                }
1818
1819                Expr::Bin(
1820                    be @ BinExpr {
1821                        op: op!("||") | op!("&&"),
1822                        ..
1823                    },
1824                ) => {
1825                    self.ignore_return_value(&mut be.right, opts);
1826
1827                    if be.right.is_invalid() {
1828                        report_change!("Dropping the RHS of a binary expression ('&&' / '||')");
1829                        *e = *be.left.take();
1830                        return;
1831                    }
1832                }
1833
1834                Expr::Tpl(tpl) => {
1835                    self.changed = true;
1836                    report_change!("Dropping a template literal");
1837
1838                    for expr in tpl.exprs.iter_mut() {
1839                        self.ignore_return_value(&mut *expr, opts);
1840                    }
1841                    tpl.exprs.retain(|expr| !expr.is_invalid());
1842                    if tpl.exprs.is_empty() {
1843                        e.take();
1844                    } else {
1845                        if tpl.exprs.len() == 1 {
1846                            *e = *tpl.exprs.remove(0);
1847                        } else {
1848                            *e = SeqExpr {
1849                                span: tpl.span,
1850                                exprs: tpl.exprs.take(),
1851                            }
1852                            .into();
1853                        }
1854                    }
1855
1856                    return;
1857                }
1858
1859                Expr::Member(MemberExpr {
1860                    obj,
1861                    prop: MemberProp::Ident(prop),
1862                    ..
1863                }) => {
1864                    if obj.is_ident_ref_to("arguments") {
1865                        if &*prop.sym == "callee" {
1866                            return;
1867                        }
1868                        e.take();
1869                        return;
1870                    }
1871                }
1872
1873                _ => {}
1874            }
1875        }
1876
1877        match e {
1878            Expr::Lit(Lit::Num(n)) => {
1879                if n.value == 0.0 && opts.contains(DropOpts::DROP_NUMBER) {
1880                    report_change!("Dropping a zero number");
1881                    *e = Invalid { span: DUMMY_SP }.into();
1882                    return;
1883                }
1884            }
1885
1886            Expr::Ident(i) => {
1887                if i.ctxt.outer() != self.marks.unresolved_mark {
1888                    report_change!("Dropping an identifier as it's declared");
1889
1890                    self.changed = true;
1891                    *e = Invalid { span: DUMMY_SP }.into();
1892                    return;
1893                }
1894            }
1895
1896            Expr::Lit(Lit::Null(..) | Lit::BigInt(..) | Lit::Bool(..) | Lit::Regex(..))
1897            | Expr::This(..) => {
1898                report_change!("Dropping meaningless values");
1899
1900                self.changed = true;
1901                *e = Expr::dummy();
1902                return;
1903            }
1904
1905            Expr::Bin(
1906                bin @ BinExpr {
1907                    op:
1908                        op!(bin, "+")
1909                        | op!(bin, "-")
1910                        | op!("*")
1911                        | op!("%")
1912                        | op!("**")
1913                        | op!("^")
1914                        | op!("&")
1915                        | op!("|")
1916                        | op!(">>")
1917                        | op!("<<")
1918                        | op!(">>>")
1919                        | op!("===")
1920                        | op!("!==")
1921                        | op!("==")
1922                        | op!("!=")
1923                        | op!("<")
1924                        | op!("<=")
1925                        | op!(">")
1926                        | op!(">="),
1927                    ..
1928                },
1929            ) => {
1930                self.ignore_return_value(
1931                    &mut bin.left,
1932                    DropOpts::DROP_GLOBAL_REFS_IF_UNUSED
1933                        .union(DropOpts::DROP_NUMBER)
1934                        .union(DropOpts::DROP_STR_LIT),
1935                );
1936                self.ignore_return_value(
1937                    &mut bin.right,
1938                    DropOpts::DROP_GLOBAL_REFS_IF_UNUSED
1939                        .union(DropOpts::DROP_NUMBER)
1940                        .union(DropOpts::DROP_STR_LIT),
1941                );
1942
1943                if bin.left.is_invalid() && bin.right.is_invalid() {
1944                    *e = Invalid { span: DUMMY_SP }.into();
1945                    return;
1946                } else if bin.right.is_invalid() {
1947                    *e = *bin.left.take();
1948                    return;
1949                } else if bin.left.is_invalid() {
1950                    *e = *bin.right.take();
1951                    return;
1952                }
1953
1954                if matches!(*bin.left, Expr::Await(..) | Expr::Update(..)) {
1955                    self.changed = true;
1956                    report_change!("ignore_return_value: Compressing binary as seq");
1957                    *e = SeqExpr {
1958                        span: bin.span,
1959                        exprs: vec![bin.left.take(), bin.right.take()],
1960                    }
1961                    .into();
1962                    return;
1963                }
1964            }
1965
1966            Expr::Assign(assign @ AssignExpr { op: op!("="), .. }) => {
1967                // Convert `a = a` to `a`.
1968                if let Some(l) = assign.left.as_ident() {
1969                    if let Expr::Ident(r) = &*assign.right {
1970                        if l.ctxt == r.ctxt
1971                            && l.ctxt != self.expr_ctx.unresolved_ctxt
1972                            && l.sym == r.sym
1973                        {
1974                            self.changed = true;
1975                            *e = *assign.right.take();
1976                        }
1977                    }
1978                }
1979            }
1980
1981            Expr::Array(arr) => {
1982                for elem in arr.elems.iter_mut().flatten() {
1983                    if elem.spread.is_none() {
1984                        self.ignore_return_value(
1985                            &mut elem.expr,
1986                            DropOpts::DROP_NUMBER.union(DropOpts::DROP_STR_LIT),
1987                        );
1988                    }
1989                }
1990
1991                arr.elems
1992                    .retain(|e| e.as_ref().is_some_and(|e| !e.expr.is_invalid()));
1993
1994                if arr.elems.is_empty() {
1995                    *e = Expr::dummy();
1996                    report_change!("Dropping an empty array literal");
1997                    self.changed = true;
1998                    return;
1999                }
2000            }
2001
2002            Expr::Object(obj) => {
2003                let mut has_spread = false;
2004                for prop in obj.props.iter_mut() {
2005                    match prop {
2006                        PropOrSpread::Spread(..) => {
2007                            has_spread = true;
2008                            break;
2009                        }
2010                        PropOrSpread::Prop(p) => match &mut **p {
2011                            Prop::KeyValue(kv) => {
2012                                if !kv.key.is_computed()
2013                                    && !kv.value.may_have_side_effects(self.expr_ctx)
2014                                {
2015                                    **p = Prop::Shorthand(Ident::dummy());
2016                                    self.changed = true;
2017                                    report_change!(
2018                                        "Dropping a key-value pair in an object literal"
2019                                    );
2020                                }
2021                            }
2022
2023                            Prop::Shorthand(i) => {
2024                                if i.ctxt.outer() != self.marks.unresolved_mark {
2025                                    *i = Ident::dummy();
2026                                    self.changed = true;
2027                                    report_change!(
2028                                        "Dropping a shorthand property in an object literal"
2029                                    );
2030                                }
2031                            }
2032
2033                            _ => {}
2034                        },
2035                    }
2036                }
2037
2038                obj.props.retain(|p| match p {
2039                    PropOrSpread::Prop(prop) => match &**prop {
2040                        Prop::Shorthand(i) => !i.is_dummy(),
2041                        _ => true,
2042                    },
2043                    _ => true,
2044                });
2045
2046                if !has_spread {
2047                    if obj.props.is_empty() {
2048                        *e = Expr::dummy();
2049                        report_change!("Dropping an empty object literal");
2050                        self.changed = true;
2051                        return;
2052                    }
2053                }
2054            }
2055
2056            _ => {}
2057        }
2058
2059        match e {
2060            Expr::Lit(Lit::Str(s)) => {
2061                if (self.options.directives && !matches!(&*s.value, "use strict" | "use asm"))
2062                    || opts.contains(DropOpts::DROP_STR_LIT)
2063                    || (s.value.starts_with("@swc/helpers")
2064                        || s.value.starts_with("@babel/helpers"))
2065                {
2066                    self.changed = true;
2067                    *e = Invalid { span: DUMMY_SP }.into();
2068
2069                    return;
2070                }
2071            }
2072
2073            Expr::Seq(seq) => {
2074                self.drop_useless_ident_ref_in_seq(seq);
2075
2076                if let Some(last) = seq.exprs.last_mut() {
2077                    // Non-last elements are already processed.
2078                    self.ignore_return_value(last, opts);
2079                }
2080
2081                let len = seq.exprs.len();
2082                seq.exprs.retain(|e| !e.is_invalid());
2083                if seq.exprs.len() != len {
2084                    self.changed = true;
2085                }
2086
2087                if seq.exprs.len() == 1 {
2088                    *e = *seq.exprs.remove(0);
2089                }
2090                return;
2091            }
2092
2093            Expr::Call(CallExpr {
2094                callee: Callee::Expr(callee),
2095                ..
2096            }) if callee.is_fn_expr() => match &mut **callee {
2097                Expr::Fn(callee) => {
2098                    if callee.ident.is_none() {
2099                        if let Some(body) = &mut callee.function.body {
2100                            if self.options.side_effects {
2101                                self.drop_return_value(&mut body.stmts);
2102                            }
2103                        }
2104                    }
2105                }
2106
2107                _ => {
2108                    unreachable!()
2109                }
2110            },
2111
2112            _ => {}
2113        }
2114
2115        if self.options.side_effects {
2116            if let Expr::Call(CallExpr {
2117                callee: Callee::Expr(callee),
2118                ..
2119            }) = e
2120            {
2121                match &mut **callee {
2122                    Expr::Fn(callee) => {
2123                        if let Some(body) = &mut callee.function.body {
2124                            if let Some(ident) = &callee.ident {
2125                                if IdentUsageFinder::find(ident, body) {
2126                                    return;
2127                                }
2128                            }
2129
2130                            for stmt in &mut body.stmts {
2131                                self.ignore_return_value_of_return_stmt(stmt, opts);
2132                            }
2133                        }
2134                    }
2135                    Expr::Arrow(callee) => match &mut *callee.body {
2136                        BlockStmtOrExpr::BlockStmt(body) => {
2137                            for stmt in &mut body.stmts {
2138                                self.ignore_return_value_of_return_stmt(stmt, opts);
2139                            }
2140                        }
2141                        BlockStmtOrExpr::Expr(body) => {
2142                            self.ignore_return_value(body, opts);
2143
2144                            if body.is_invalid() {
2145                                *body = 0.into();
2146                                return;
2147                            }
2148                        }
2149                    },
2150                    _ => {}
2151                }
2152            }
2153        }
2154
2155        if self.options.side_effects && self.options.pristine_globals {
2156            match e {
2157                Expr::New(NewExpr {
2158                    span, callee, args, ..
2159                }) if callee.is_one_of_global_ref_to(
2160                    self.expr_ctx,
2161                    &[
2162                        "Map", "Set", "Array", "Object", "Boolean", "Number", "String",
2163                    ],
2164                ) =>
2165                {
2166                    report_change!("Dropping a pure new expression");
2167
2168                    self.changed = true;
2169                    *e = self
2170                        .make_ignored_expr(
2171                            *span,
2172                            args.iter_mut().flatten().map(|arg| arg.expr.take()),
2173                        )
2174                        .unwrap_or(Invalid { span: DUMMY_SP }.into());
2175                    return;
2176                }
2177
2178                Expr::Call(CallExpr {
2179                    span,
2180                    callee: Callee::Expr(callee),
2181                    args,
2182                    ..
2183                }) if callee.is_one_of_global_ref_to(
2184                    self.expr_ctx,
2185                    &["Array", "Object", "Boolean", "Number"],
2186                ) =>
2187                {
2188                    report_change!("Dropping a pure call expression");
2189
2190                    self.changed = true;
2191                    *e = self
2192                        .make_ignored_expr(*span, args.iter_mut().map(|arg| arg.expr.take()))
2193                        .unwrap_or(Invalid { span: DUMMY_SP }.into());
2194                    return;
2195                }

2417    ///
2418    /// - `a ? true : false` => `!!a`
2419    pub(super) fn compress_useless_cond_expr(&mut self, expr: &mut Expr) {
2420        let cond = match expr {
2421            Expr::Cond(c) => c,
2422            _ => return,
2423        };
2424
2425        let lt = cond.cons.get_type(self.expr_ctx);
2426        let rt = cond.alt.get_type(self.expr_ctx);
2427        match (lt, rt) {
2428            (Known(Type::Bool), Known(Type::Bool)) => {}
2429            _ => return,
2430        }
2431
2432        let lb = cond.cons.as_pure_bool(self.expr_ctx);
2433        let rb = cond.alt.as_pure_bool(self.expr_ctx);
2434
2435        let lb = match lb {
2436            Value::Known(v) => v,
2437            Value::Unknown => return,
2438        };
2439        let rb = match rb {
2440            Value::Known(v) => v,
2441            Value::Unknown => return,
2442        };
2443
2444        // `cond ? true : false` => !!cond
2445        if lb && !rb {
2446            self.negate(&mut cond.test, false, false);
2447            self.negate(&mut cond.test, false, false);
2448            *expr = *cond.test.take();
2449            return;
2450        }
2451
2452        // `cond ? false : true` => !cond
2453        if !lb && rb {
2454            self.negate(&mut cond.test, false, false);
2455            *expr = *cond.test.take();
2456        }
2457    }
