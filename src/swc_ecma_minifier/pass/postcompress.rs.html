8pub fn postcompress_optimizer(options: &CompressOptions) -> impl '_ + VisitMut {
9    PostcompressOptimizer {
10        options,
11        ctx: Default::default(),
12    }
13}
14
15struct PostcompressOptimizer<'a> {
16    options: &'a CompressOptions,
17
18    ctx: Ctx,
19}
20
21#[derive(Default, Clone, Copy)]
22struct Ctx {
23    is_module: bool,
24    is_top_level: bool,
25}
26
27impl Parallel for PostcompressOptimizer<'_> {
28    fn create(&self) -> Self {
29        Self {
30            options: self.options,
31            ctx: self.ctx,
32        }
33    }
34
35    fn merge(&mut self, _: Self) {}
36}
37
38impl VisitMut for PostcompressOptimizer<'_> {
39    noop_visit_mut_type!(fail);
40
41    fn visit_mut_export_decl(&mut self, export: &mut ExportDecl) {
42        match &mut export.decl {
43            Decl::Var(decl) => {
44                // Don't change constness of exported variables.
45                decl.visit_mut_children_with(self);
46            }
47            _ => {
48                export.decl.visit_mut_with(self);
49            }
50        }
51    }
52
53    fn visit_mut_module_decl(&mut self, m: &mut ModuleDecl) {
54        if let ModuleDecl::ExportDefaultExpr(e) = m {
55            match &mut *e.expr {
56                Expr::Fn(f) => {
57                    if f.ident.is_some() {
58                        if self.options.top_level() {
59                            *m = ExportDefaultDecl {
60                                span: e.span,
61                                decl: DefaultDecl::Fn(f.take()),
62                            }
63                            .into()
64                        }
65                    } else {
66                        *m = ExportDefaultDecl {
67                            span: e.span,
68                            decl: DefaultDecl::Fn(f.take()),
69                        }
70                        .into()
71                    }
72                }
73                Expr::Class(c) => {
74                    if c.ident.is_some() {
75                        if self.options.top_level() {
76                            *m = ExportDefaultDecl {
77                                span: e.span,
78                                decl: DefaultDecl::Class(c.take()),
79                            }
80                            .into()
81                        }
82                    } else {
83                        *m = ExportDefaultDecl {
84                            span: e.span,
85                            decl: DefaultDecl::Class(c.take()),
86                        }
87                        .into()
88                    }
89                }
90                _ => (),
91            }
92        }
93
94        m.visit_mut_children_with(self)
95    }
96
97    fn visit_mut_module_items(&mut self, nodes: &mut Vec<ModuleItem>) {
98        self.ctx.is_module = maybe_par!(
99            nodes
100                .iter()
101                .any(|s| matches!(s, ModuleItem::ModuleDecl(..))),
102            *crate::LIGHT_TASK_PARALLELS
103        );
104        self.ctx.is_top_level = true;
105
106        self.maybe_par(*crate::LIGHT_TASK_PARALLELS, nodes, |v, n| {
107            n.visit_mut_with(v);
108        });
109    }
110
111    fn visit_mut_stmts(&mut self, nodes: &mut Vec<Stmt>) {
112        let old = self.ctx;
113
114        self.ctx.is_top_level = false;
115
116        self.maybe_par(*crate::LIGHT_TASK_PARALLELS, nodes, |v, n| {
117            n.visit_mut_with(v);
118        });
119
120        self.ctx = old;
121    }
122
123    fn visit_mut_var_decl(&mut self, v: &mut VarDecl) {
124        v.visit_mut_children_with(self);
125
126        if self.options.const_to_let {
127            if self.ctx.is_module || !self.ctx.is_top_level {
128                // We don't change constness of top-level variables in a script
129
130                if let VarDeclKind::Const = v.kind {
131                    v.kind = VarDeclKind::Let;
132                }
133            }
134        }
135    }
136
137    fn visit_mut_exprs(&mut self, n: &mut Vec<Box<Expr>>) {
138        self.maybe_par(*LIGHT_TASK_PARALLELS, n, |v, n| {
139            n.visit_mut_with(v);
140        });
141    }
142
143    fn visit_mut_opt_vec_expr_or_spreads(&mut self, n: &mut Vec<Option<ExprOrSpread>>) {
144        self.maybe_par(*LIGHT_TASK_PARALLELS, n, |v, n| {
145            n.visit_mut_with(v);
146        });
147    }
148
149    fn visit_mut_expr_or_spreads(&mut self, n: &mut Vec<ExprOrSpread>) {
150        self.maybe_par(*LIGHT_TASK_PARALLELS, n, |v, n| {
151            n.visit_mut_with(v);
152        });
153    }
154
155    fn visit_mut_var_declarators(&mut self, n: &mut Vec<VarDeclarator>) {
156        self.maybe_par(*LIGHT_TASK_PARALLELS, n, |v, n| {
157            n.visit_mut_with(v);
158        });
159    }
160}
